34#undef _FORTIFY_SOURCE
35#include <limits.h>
36#include <stdio.h>
37#include <string.h>
38#include <stdarg.h>
39#include "local.h"
40
41int
42snprintf(char *str, size_t n, const char *fmt, ...)
43{
44 va_list ap;
45 int ret;
46 char dummy;
47 FILE f;
48 struct __sfileext fext;
49
50 /* While snprintf(3) specifies size_t stdio uses an int internally */
51 if (n > INT_MAX)
52 n = INT_MAX;
53 /* Stdio internals do not deal correctly with zero length buffer */
54 if (n == 0) {
55 str = &dummy;
56 n = 1;
57 }
58 _FILEEXT_SETUP(&f, &fext);
59 f._file = -1;
60 f._flags = __SWR | __SSTR;
61 f._bf._base = f._p = (unsigned char *)str;
62 f._bf._size = f._w = n - 1;
63 va_start(ap, fmt);
64 ret = __vfprintf(&f, fmt, ap);
65 va_end(ap);
66 *f._p = '\0';
67 return (ret);
68}
