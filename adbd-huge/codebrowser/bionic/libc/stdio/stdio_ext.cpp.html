37size_t __fbufsize(FILE* fp) {
38 return fp->_bf._size;
39}
40
41/* For a _SRW stream, we don't know whether we last read or wrote.
42int __freading(FILE* fp) {
43 return (fp->_flags & _SRD) != 0 || ...;
44}
45*/
46
47/* For a _SRW stream, we don't know whether we last read or wrote.
48int __fwriting(FILE*) {
49 return (fp->_flags & _SWR) != 0 || ...;
50}
51*/
52
53int __freadable(FILE* fp) {
54 return (fp->_flags & (__SRD|__SRW)) != 0;
55}
56
57int __fwritable(FILE* fp) {
58 return (fp->_flags & (__SWR|__SRW)) != 0;
59}
60
61int __flbf(FILE* fp) {
62 return (fp->_flags & __SLBF) != 0;
63}

69size_t __fpending(FILE* fp) {

73void _flushlbf() {
74 // If we flush all streams, we know we've flushed all the line-buffered streams.
75 fflush(NULL);
76}

78int __fsetlocking(FILE* fp, int type) {
79 int old_state = _EXT(fp)->_caller_handles_locking ? FSETLOCKING_BYCALLER : FSETLOCKING_INTERNAL;
80 if (type == FSETLOCKING_QUERY) {
81 return old_state;
82 }
83
84 if (type != FSETLOCKING_INTERNAL && type != FSETLOCKING_BYCALLER) {
85 // The API doesn't let us report an error, so blow up.
86 __libc_fatal("Bad type (%d) passed to __fsetlocking", type);
87 }
88
89 _EXT(fp)->_caller_handles_locking = (type == FSETLOCKING_BYCALLER);
90 return old_state;
91}

105int fileno_unlocked(FILE* fp) {
106 int fd = fp->_file;
107 if (fd == -1) {
108 errno = EBADF;
109 return -1;
110 }
111 return fd;
112}
