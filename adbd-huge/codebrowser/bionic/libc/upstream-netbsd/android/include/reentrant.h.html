25//
26// Map NetBSD libc internal locking to pthread locking.
27//
28
29#define MUTEX_INITIALIZER PTHREAD_MUTEX_INITIALIZER
30#define mutex_t pthread_mutex_t
31#define mutex_lock(x) pthread_mutex_lock(x)
32#define mutex_unlock(x) pthread_mutex_unlock(x)
33
34#define RWLOCK_INITIALIZER PTHREAD_RWLOCK_INITIALIZER
35#define rwlock_t pthread_rwlock_t
36#define rwlock_rdlock pthread_rwlock_rdlock
37#define rwlock_unlock pthread_rwlock_unlock
38#define rwlock_wrlock pthread_rwlock_wrlock
