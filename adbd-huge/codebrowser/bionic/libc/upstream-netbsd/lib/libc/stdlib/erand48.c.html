32double
33erand48(unsigned short xseed[3])
34{
35
36 _DIAGASSERT(xseed != NULL);
37
38 __dorand48(xseed);
39 return ldexp((double) xseed[0], -48) +
40 ldexp((double) xseed[1], -32) +
41 ldexp((double) xseed[2], -16);
42}
