29#ifndef _DIRENT_H_
30#define _DIRENT_H_
31
32#include <stdint.h>
33#include <sys/cdefs.h>
34
35__BEGIN_DECLS
36
37#ifndef DT_UNKNOWN
38#define DT_UNKNOWN 0
39#define DT_FIFO 1
40#define DT_CHR 2
41#define DT_DIR 4
42#define DT_BLK 6
43#define DT_REG 8
44#define DT_LNK 10
45#define DT_SOCK 12
46#define DT_WHT 14
47#endif
48
49#define __DIRENT64_BODY \
50 uint64_t d_ino; \
51 int64_t d_off; \
52 unsigned short d_reclen; \
53 unsigned char d_type; \
54 char d_name[256]; \
55
56struct dirent { __DIRENT64_BODY };
57struct dirent64 { __DIRENT64_BODY };
58
59#undef __DIRENT64_BODY
60
61/* glibc compatibility. */
62#undef _DIRENT_HAVE_D_NAMLEN /* Linux doesn't have a d_namlen field. */
63#define _DIRENT_HAVE_D_RECLEN
64#define _DIRENT_HAVE_D_OFF
65#define _DIRENT_HAVE_D_TYPE
66
67#define d_fileno d_ino
68
69typedef struct DIR DIR;
70
71extern DIR* opendir(const char*);
72extern DIR* fdopendir(int);
73extern struct dirent* readdir(DIR*);
74extern struct dirent64* readdir64(DIR*);
75extern int readdir_r(DIR*, struct dirent*, struct dirent**);
76extern int readdir64_r(DIR*, struct dirent64*, struct dirent64**);
77extern int closedir(DIR*);
78extern void rewinddir(DIR*);
79extern void seekdir(DIR*, long);
80extern long telldir(DIR*);
81extern int dirfd(DIR*);
82extern int alphasort(const struct dirent**, const struct dirent**);
83extern int alphasort64(const struct dirent64**, const struct dirent64**);
84extern int scandir64(const char*, struct dirent64***, int (*)(const struct dirent64*), int (*)(const struct dirent64**, const struct dirent64**));
85extern int scandir(const char*, struct dirent***, int (*)(const struct dirent*), int (*)(const struct dirent**, const struct dirent**));
86
87#if defined(__USE_GNU)
88int scandirat64(int, const char*, struct dirent64***, int (*)(const struct dirent64*), int (*)(const struct dirent64**, const struct dirent64**));
89int scandirat(int, const char*, struct dirent***, int (*)(const struct dirent*), int (*)(const struct dirent**, const struct dirent**));
90#endif
91
92__END_DECLS
93
94#endif /* _DIRENT_H_ */
