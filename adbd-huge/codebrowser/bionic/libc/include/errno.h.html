28#ifndef _ERRNO_H
29#define _ERRNO_H
30
31#include <sys/cdefs.h>
32#include <linux/errno.h>
33
34__BEGIN_DECLS
35
36/* on Linux, ENOTSUP and EOPNOTSUPP are defined as the same error code
37 * even if 1000.3 states that they should be different
38 */
39#ifndef ENOTSUP
40#define ENOTSUP EOPNOTSUPP
41#endif
42
43/* internal function returning the address of the thread-specific errno */
44extern volatile int* __errno(void) __pure2;
45
46/* a macro expanding to the errno l-value */
47#define errno (*__errno())
48
49#if __ANDROID_API__ < 21
50#include <android/legacy_errno_inlines.h>
51#endif
52
53__END_DECLS
54
55#endif /* _ERRNO_H */
