28#ifndef _SYS_WAIT_H_
29#define _SYS_WAIT_H_
30
31#include <sys/cdefs.h>
32#include <sys/types.h>
33#include <sys/resource.h>
34#include <linux/wait.h>
35#include <signal.h>
36
37__BEGIN_DECLS
38
39#define WEXITSTATUS(s) (((s) & 0xff00) >> 8)
40#define WCOREDUMP(s) ((s) & 0x80)
41#define WTERMSIG(s) ((s) & 0x7f)
42#define WSTOPSIG(s) WEXITSTATUS(s)
43
44#define WIFEXITED(s) (WTERMSIG(s) == 0)
45#define WIFSTOPPED(s) (WTERMSIG(s) == 0x7f)
46#define WIFSIGNALED(s) (WTERMSIG((s)+1) >= 2)
47#define WIFCONTINUED(s) ((s) == 0xffff)
48
49#define W_EXITCODE(ret, sig) ((ret) << 8 | (sig))
50#define W_STOPCODE(sig) ((sig) << 8 | 0x7f)
51
52extern pid_t wait(int *);
53extern pid_t waitpid(pid_t, int *, int);
54extern pid_t wait4(pid_t, int *, int, struct rusage *);
55
56/* Posix states that idtype_t should be an enumeration type, but
57 * the kernel headers define P_ALL, P_PID and P_PGID as constant macros
58 * instead.
59 */
60typedef int idtype_t;
61
62extern int waitid(idtype_t which, id_t id, siginfo_t *info, int options);
63
64__END_DECLS
65
66#endif /* _SYS_WAIT_H_ */
