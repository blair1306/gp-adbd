44 long swd;
45 long twd;

66};

68 long ebx;
69 long ecx;
70 long edx;
71 long esi;
72 long edi;
73 long ebp;
74 long eax;

77 long xfs;
78 long xgs;
79 long orig_eax;
80 long eip;

84 long xss;
