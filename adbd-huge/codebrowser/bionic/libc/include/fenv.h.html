57/*
58 * The following constant represents the default floating-point environment
59 * (that is, the one installed at program startup) and has type pointer to
60 * const-qualified fenv_t.
61 *
62 * It can be used as an argument to the functions that manage the floating-point
63 * environment, namely fesetenv() and feupdateenv().
64 */
65extern const fenv_t __fe_dfl_env;
66#define FE_DFL_ENV (&__fe_dfl_env)
