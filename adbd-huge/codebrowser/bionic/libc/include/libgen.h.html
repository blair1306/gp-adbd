29#ifndef _LIBGEN_H
30#define _LIBGEN_H
31
32#include <sys/cdefs.h>
33#include <sys/types.h>
34
35
36__BEGIN_DECLS
37
38/*
39 * Including <string.h> will get you the GNU basename, unless <libgen.h> is
40 * included, either before or after including <string.h>.
41 *
42 * Note that this has the wrong argument cv-qualifiers, but doesn't modify its
43 * input and uses thread-local storage for the result if necessary.
44 */
45extern char* __posix_basename(const char*) __RENAME(basename);
46
47#define basename __posix_basename
48
49/* This has the wrong argument cv-qualifiers, but doesn't modify its input and uses thread-local storage for the result if necessary. */
50extern char* dirname(const char*);
51
52#if !defined(__LP64__)
53/* These non-standard functions are not needed on Android; basename and dirname use thread-local storage. */
54extern int dirname_r(const char*, char*, size_t);
55extern int basename_r(const char*, char*, size_t);
56#endif
57
58__END_DECLS
59
60#endif /* _LIBGEN_H */
