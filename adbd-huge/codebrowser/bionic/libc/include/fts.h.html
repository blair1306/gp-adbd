35#ifndef _FTS_H_
36#define _FTS_H_
37
38#include <sys/types.h>
39
40typedef struct {
41 struct _ftsent *fts_cur; /* current node */
42 struct _ftsent *fts_child; /* linked list of children */
43 struct _ftsent **fts_array; /* sort array */
44 dev_t fts_dev; /* starting device # */
45 char *fts_path; /* path for this descent */
46 int fts_rfd; /* fd for root */
47 size_t fts_pathlen; /* sizeof(path) */
48 int fts_nitems; /* elements in the sort array */
49 int (*fts_compar)(); /* compare function */
50
51#define FTS_COMFOLLOW 0x0001 /* follow command line symlinks */
52#define FTS_LOGICAL 0x0002 /* logical walk */
53#define FTS_NOCHDIR 0x0004 /* don't change directories */
54#define FTS_NOSTAT 0x0008 /* don't get stat info */
55#define FTS_PHYSICAL 0x0010 /* physical walk */
56#define FTS_SEEDOT 0x0020 /* return dot and dot-dot */
57#define FTS_XDEV 0x0040 /* don't cross devices */
58#define FTS_OPTIONMASK 0x00ff /* valid user option mask */
59
60#define FTS_NAMEONLY 0x1000 /* (private) child names only */
61#define FTS_STOP 0x2000 /* (private) unrecoverable error */
62 int fts_options; /* fts_open options, global flags */
63} FTS;
64
65typedef struct _ftsent {
66 struct _ftsent *fts_cycle; /* cycle node */
67 struct _ftsent *fts_parent; /* parent directory */
68 struct _ftsent *fts_link; /* next file in directory */
69 long fts_number; /* local numeric value */
70 void *fts_pointer; /* local address value */
71 char *fts_accpath; /* access path */
72 char *fts_path; /* root path */
73 int fts_errno; /* errno for this node */
74 int fts_symfd; /* fd for symlink */
75 size_t fts_pathlen; /* strlen(fts_path) */
76 size_t fts_namelen; /* strlen(fts_name) */
77
78 ino_t fts_ino; /* inode */
79 dev_t fts_dev; /* device */
80 nlink_t fts_nlink; /* link count */
81
82#define FTS_ROOTPARENTLEVEL -1
83#define FTS_ROOTLEVEL 0
84#define FTS_MAXLEVEL 0x7fff
85 short fts_level; /* depth (-1 to N) */
86
87#define FTS_D 1 /* preorder directory */
88#define FTS_DC 2 /* directory that causes cycles */
89#define FTS_DEFAULT 3 /* none of the above */
90#define FTS_DNR 4 /* unreadable directory */
91#define FTS_DOT 5 /* dot or dot-dot */
92#define FTS_DP 6 /* postorder directory */
93#define FTS_ERR 7 /* error; errno is set */
94#define FTS_F 8 /* regular file */
95#define FTS_INIT 9 /* initialized only */
96#define FTS_NS 10 /* stat(2) failed */
97#define FTS_NSOK 11 /* no stat(2) requested */
98#define FTS_SL 12 /* symbolic link */
99#define FTS_SLNONE 13 /* symbolic link without target */
100 unsigned short fts_info; /* user flags for FTSENT structure */
101
102#define FTS_DONTCHDIR 0x01 /* don't chdir .. to the parent */
103#define FTS_SYMFOLLOW 0x02 /* followed a symlink to get here */
104 unsigned short fts_flags; /* private flags for FTSENT structure */
105
106#define FTS_AGAIN 1 /* read node again */
107#define FTS_FOLLOW 2 /* follow symbolic link */
108#define FTS_NOINSTR 3 /* no instructions */
109#define FTS_SKIP 4 /* discard node */
110 unsigned short fts_instr; /* fts_set() instructions */
111
112 struct stat *fts_statp; /* stat(2) information */
113 char fts_name[1]; /* file name */
114} FTSENT;
115
116__BEGIN_DECLS
117FTSENT *fts_children(FTS *, int);
118int fts_close(FTS *);
119FTS *fts_open(char * const *, int,
120 int (*)(const FTSENT **, const FTSENT **));
121FTSENT *fts_read(FTS *);
122int fts_set(FTS *, FTSENT *, int);
123__END_DECLS
124
125#endif /* !_FTS_H_ */
