29#ifndef _IFADDRS_H_
30#define _IFADDRS_H_
31
32#include <sys/cdefs.h>
33#include <netinet/in.h>
34#include <sys/socket.h>
35
36__BEGIN_DECLS
37
38struct ifaddrs {
39 struct ifaddrs* ifa_next;
40 char* ifa_name;
41 unsigned int ifa_flags;
42 struct sockaddr* ifa_addr;
43 struct sockaddr* ifa_netmask;
44 union {
45 struct sockaddr* ifu_broadaddr;
46 struct sockaddr* ifu_dstaddr;
47 } ifa_ifu;
48 void* ifa_data;
49};
50
51#define ifa_broadaddr ifa_ifu.ifu_broadaddr
52#define ifa_dstaddr ifa_ifu.ifu_dstaddr
53
54void freeifaddrs(struct ifaddrs*);
55int getifaddrs(struct ifaddrs**);
56
57__END_DECLS
58
59#endif
