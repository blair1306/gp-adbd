48#ifndef _MACHINE_IEEE_H_
49#define _MACHINE_IEEE_H_
50
51#include <sys/types.h>
52
53__BEGIN_DECLS
54
55#define SNG_EXPBITS 8
56#define SNG_FRACBITS 23
57
58#define SNG_EXP_INFNAN 255
59#define SNG_EXP_BIAS 127
60
61struct ieee_single {
62 unsigned sng_frac:23;
63 unsigned sng_exp:8;
64 unsigned sng_sign:1;
65};
66
67#define DBL_EXPBITS 11
68#define DBL_FRACHBITS 20
69#define DBL_FRACLBITS 32
70#define DBL_FRACBITS 52
71
72#define DBL_EXP_INFNAN 2047
73#define DBL_EXP_BIAS 1023
74
75struct ieee_double {
76 unsigned dbl_fracl;
77 unsigned dbl_frach:20;
78 unsigned dbl_exp:11;
79 unsigned dbl_sign:1;
80};
81
82#if __LP64__
83
84/* 64-bit Android uses ld128 long doubles. */
85
86#define EXT_EXPBITS 15
87#define EXT_FRACHBITS 16
88#define EXT_FRACHMBITS 32
89#define EXT_FRACLMBITS 32
90#define EXT_FRACLBITS 32
91#define EXT_FRACBITS 112
92
93#define EXT_EXP_INFNAN 32767
94#define EXT_EXP_BIAS 16383
95
96#define EXT_IMPLICIT_NBIT
97
98#define EXT_TO_ARRAY32(p, a) do { \
99 (a)[0] = (uint32_t)(p)->ext_fracl; \
100 (a)[1] = (uint32_t)(p)->ext_fraclm; \
101 (a)[2] = (uint32_t)(p)->ext_frachm; \
102 (a)[3] = (uint32_t)(p)->ext_frach; \
103} while(0)
104
105struct ieee_ext {
106 unsigned ext_fracl;
107 unsigned ext_fraclm;
108 unsigned ext_frachm;
109 unsigned ext_frach:16;
110 unsigned ext_exp:15;
111 unsigned ext_sign:1;
112};
113
114#endif
115
116__END_DECLS

118#endif /* _MACHINE_IEEE_H_ */
