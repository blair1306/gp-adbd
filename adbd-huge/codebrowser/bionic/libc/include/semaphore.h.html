29#ifndef _SEMAPHORE_H
30#define _SEMAPHORE_H
31
32#include <sys/cdefs.h>
33
34__BEGIN_DECLS
35
36struct timespec;
37
38typedef struct {
39 unsigned int count;
40#ifdef __LP64__
41 int __reserved[3];
42#endif
43} sem_t;
44
45#define SEM_FAILED NULL
46
47int sem_destroy(sem_t*);
48int sem_getvalue(sem_t*, int*);
49int sem_init(sem_t*, int, unsigned int);
50int sem_post(sem_t*);
51int sem_timedwait(sem_t*, const struct timespec*);
52int sem_trywait(sem_t*);
53int sem_wait(sem_t*);
54
55/* These aren't actually implemented. */
56sem_t* sem_open(const char*, int, ...);
57int sem_close(sem_t*);
58int sem_unlink(const char*);
59
60__END_DECLS
61
62#endif /* _SEMAPHORE_H */
