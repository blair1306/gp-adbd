29#ifndef _SYSLOG_H
30#define _SYSLOG_H
31
32#include <stdio.h>
33#include <sys/cdefs.h>
34#include <stdarg.h>
35
36__BEGIN_DECLS
37
38/* Priorities are translated to Android log priorities as shown. */
39#define LOG_EMERG 0 /* ERROR */
40#define LOG_ALERT 1 /* ERROR */
41#define LOG_CRIT 2 /* ERROR */
42#define LOG_ERR 3 /* ERROR */
43#define LOG_WARNING 4 /* WARN */
44#define LOG_NOTICE 5 /* INFO */
45#define LOG_INFO 6 /* INFO */
46#define LOG_DEBUG 7 /* DEBUG */
47
48#define LOG_PRIMASK 7
49#define LOG_PRI(x) ((x) & LOG_PRIMASK)
50#define LOG_MAKEPRI(fac, pri) ((fac) | (pri))
51
52/* Facilities are currently ignored on Android. */
53#define LOG_KERN 0000
54#define LOG_USER 0010
55#define LOG_MAIL 0020
56#define LOG_DAEMON 0030
57#define LOG_AUTH 0040
58#define LOG_SYSLOG 0050
59#define LOG_LPR 0060
60#define LOG_NEWS 0070
61#define LOG_UUCP 0100
62#define LOG_CRON 0110
63#define LOG_AUTHPRIV 0120
64#define LOG_FTP 0130
65#define LOG_LOCAL0 0200
66#define LOG_LOCAL1 0210
67#define LOG_LOCAL2 0220
68#define LOG_LOCAL3 0230
69#define LOG_LOCAL4 0240
70#define LOG_LOCAL5 0250
71#define LOG_LOCAL6 0260
72#define LOG_LOCAL7 0270
73
74#define LOG_FACMASK 01770
75#define LOG_FAC(x) (((x) >> 3) & (LOG_FACMASK >> 3))
76
77#define LOG_MASK(pri) (1 << (pri))
78#define LOG_UPTO(pri) ((1 << ((pri)+1)) - 1)
79
80/* openlog(3) flags are currently ignored on Android. */
81#define LOG_PID 0x01
82#define LOG_CONS 0x02
83#define LOG_ODELAY 0x04
84#define LOG_NDELAY 0x08
85#define LOG_NOWAIT 0x10
86#define LOG_PERROR 0x20
87
88void closelog(void);
89void openlog(const char*, int, int);
90int setlogmask(int);
91void syslog(int, const char*, ...) __printflike(2, 3);
92void vsyslog(int, const char*, va_list) __printflike(2, 0);
93
94__END_DECLS
95
96#endif /* _SYSLOG_H */
