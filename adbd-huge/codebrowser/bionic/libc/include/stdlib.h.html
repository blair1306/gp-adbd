183extern char* __realpath_real(const char*, char*) __RENAME(realpath);
184__errordecl(__realpath_size_error, "realpath output parameter must be NULL or a >= PATH_MAX bytes buffer");
185
186#if !defined(__clang__)
187__BIONIC_FORTIFY_INLINE
188char* realpath(const char* path, char* resolved) {
189 size_t bos = __bos(resolved);
190
191 /* PATH_MAX is unavailable without polluting the namespace, but it's always 4096 on Linux */
192 if (bos != __BIONIC_FORTIFY_UNKNOWN_SIZE && bos < 4096) {
193 __realpath_size_error();
194 }
195
196 return __realpath_real(path, resolved);
197}
