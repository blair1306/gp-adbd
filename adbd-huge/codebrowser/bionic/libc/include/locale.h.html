28#ifndef _LOCALE_H_
29#define _LOCALE_H_
30
31#include <sys/cdefs.h>
32#include <xlocale.h>
33
34__BEGIN_DECLS
35
36#define LC_CTYPE 0
37#define LC_NUMERIC 1
38#define LC_TIME 2
39#define LC_COLLATE 3
40#define LC_MONETARY 4
41#define LC_MESSAGES 5
42#define LC_ALL 6
43#define LC_PAPER 7
44#define LC_NAME 8
45#define LC_ADDRESS 9
46#define LC_TELEPHONE 10
47#define LC_MEASUREMENT 11
48#define LC_IDENTIFICATION 12
49
50#define LC_CTYPE_MASK (1 << LC_CTYPE)
51#define LC_NUMERIC_MASK (1 << LC_NUMERIC)
52#define LC_TIME_MASK (1 << LC_TIME)
53#define LC_COLLATE_MASK (1 << LC_COLLATE)
54#define LC_MONETARY_MASK (1 << LC_MONETARY)
55#define LC_MESSAGES_MASK (1 << LC_MESSAGES)
56#define LC_PAPER_MASK (1 << LC_PAPER)
57#define LC_NAME_MASK (1 << LC_NAME)
58#define LC_ADDRESS_MASK (1 << LC_ADDRESS)
59#define LC_TELEPHONE_MASK (1 << LC_TELEPHONE)
60#define LC_MEASUREMENT_MASK (1 << LC_MEASUREMENT)
61#define LC_IDENTIFICATION_MASK (1 << LC_IDENTIFICATION)
62
63#define LC_ALL_MASK (LC_CTYPE_MASK | LC_NUMERIC_MASK | LC_TIME_MASK | LC_COLLATE_MASK | \
64 LC_MONETARY_MASK | LC_MESSAGES_MASK | LC_PAPER_MASK | LC_NAME_MASK | \
65 LC_ADDRESS_MASK | LC_TELEPHONE_MASK | LC_MEASUREMENT_MASK | \
66 LC_IDENTIFICATION_MASK)
67
68struct lconv {
69 char* decimal_point;
70 char* thousands_sep;
71 char* grouping;
72 char* int_curr_symbol;
73 char* currency_symbol;
74 char* mon_decimal_point;
75 char* mon_thousands_sep;
76 char* mon_grouping;
77 char* positive_sign;
78 char* negative_sign;
79 char int_frac_digits;
80 char frac_digits;
81 char p_cs_precedes;
82 char p_sep_by_space;
83 char n_cs_precedes;
84 char n_sep_by_space;
85 char p_sign_posn;
86 char n_sign_posn;
87 char int_p_cs_precedes;
88 char int_p_sep_by_space;
89 char int_n_cs_precedes;
90 char int_n_sep_by_space;
91 char int_p_sign_posn;
92 char int_n_sign_posn;
93};
94
95struct lconv* localeconv(void);
96
97locale_t duplocale(locale_t);
98void freelocale(locale_t);
99locale_t newlocale(int, const char*, locale_t);
100char* setlocale(int, const char*);
101locale_t uselocale(locale_t);
102
103#define LC_GLOBAL_LOCALE ((locale_t) -1L)
104
105__END_DECLS
