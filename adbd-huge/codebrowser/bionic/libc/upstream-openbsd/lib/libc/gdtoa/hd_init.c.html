33
34 unsigned char hexdig[256];
35
36 static void
37#ifdef KR_headers
38htinit(h, s, inc) unsigned char *h; unsigned char *s; int inc;
39#else
40htinit(unsigned char *h, unsigned char *s, int inc)
41#endif
42{
43 int i, j;
44 for(i = 0; (j = s[i]) !=0; i++)
45 h[j] = i + inc;
46 }
47
48 void
49hexdig_init_D2A(Void)
50{
51#define USC (unsigned char *)
52 htinit(hexdig, USC "0123456789", 0x10);
53 htinit(hexdig, USC "abcdef", 0x10 + 10);
54 htinit(hexdig, USC "ABCDEF", 0x10 + 10);
55 }
