32#include "gdtoaimp.h"
33
34 double
35ulp
36#ifdef KR_headers
37 (x) U *x;
38#else
39 (U *x)
40#endif
41{
42 Long L;
43 U a;
44
45 L = (word0(x) & Exp_mask) - (P-1)*Exp_msk1;
46#ifndef Sudden_Underflow
47 if (L > 0) {
48#endif
49#ifdef IBM
50 L |= Exp_msk1 >> 4;
51#endif
52 word0(&a) = L;
53 word1(&a) = 0;
54#ifndef Sudden_Underflow
55 }
56 else {
57 L = -L >> Exp_shift;
58 if (L < Exp_shift) {
59 word0(&a) = 0x80000 >> L;
60 word1(&a) = 0;
61 }
62 else {
63 word0(&a) = 0;
64 L -= Exp_shift;
65 word1(&a) = L >= 31 ? 1 : 1 << (31 - L);
66 }
67 }
68#endif
69 return dval(&a);
70 }
