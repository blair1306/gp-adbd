19#include <stdio.h>
20#include <stdlib.h>
21#include <string.h>
22#include <errno.h>
23#include <stdarg.h>
24#include "local.h"
25
26int
27asprintf(char **str, const char *fmt, ...)
28{
29 int ret;
30 va_list ap;
31 FILE f;
32 struct __sfileext fext;
33 unsigned char *_base;
34
35 _FILEEXT_SETUP(&f, &fext);
36 f._file = -1;
37 f._flags = __SWR | __SSTR | __SALC;
38 f._bf._base = f._p = malloc(128);
39 if (f._bf._base == NULL)
40 goto err;
41 f._bf._size = f._w = 127; /* Leave room for the NUL */
42 va_start(ap, fmt);
43 ret = __vfprintf(&f, fmt, ap);
44 va_end(ap);
45 if (ret == -1)
46 goto err;
47 *f._p = '\0';
48 _base = realloc(f._bf._base, ret + 1);
49 if (_base == NULL)
50 goto err;
51 *str = (char *)_base;
52 return (ret);
53
54err:
55 free(f._bf._base);
56 f._bf._base = NULL;
57 *str = NULL;
58 errno = ENOMEM;
59 return (-1);
60}
61DEF_WEAK(asprintf);
