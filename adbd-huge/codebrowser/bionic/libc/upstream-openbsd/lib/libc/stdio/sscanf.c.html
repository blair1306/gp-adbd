40static int
41eofread(void *cookie, char *buf, int len)
42{
43

45}

47/* SCANFLIKE2 */
48int
49sscanf(const char *str, const char *fmt, ...)
50{
51 int ret;
52 va_list ap;
53 FILE f;
54 struct __sfileext fext;
55
56 _FILEEXT_SETUP(&f, &fext);
57 f._flags = __SRD;
58 f._bf._base = f._p = (unsigned char *)str;
59 f._bf._size = f._r = strlen(str);
60 f._read = eofread;
61 f._lb._base = NULL;
62 va_start(ap, fmt);
63 ret = __svfscanf(&f, fmt, ap);
64 va_end(ap);
65 return (ret);
66}
