36#include <stdio.h>
37#include <unistd.h>
38
39__warn_references(tmpnam,
40 "warning: tmpnam() possibly used unsafely; consider using mkstemp()");
41
42extern char *_mktemp(char *);
43
44char *
45tmpnam(char *s)
46{
47 static u_long tmpcount;
48 static char buf[L_tmpnam];
49
50 if (s == NULL)
51 s = buf;
52 (void)snprintf(s, L_tmpnam, "%stmp.%lu.XXXXXXXXX", P_tmpdir, tmpcount);
53 ++tmpcount;
54 return (_mktemp(s));
55}
