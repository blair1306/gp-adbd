41#include "local.h"

43static int
44__dwrite(void *cookie, const char *buf, int n)
45{
46 int *fdp = cookie;
47 return (write(*fdp, buf, n));
48}
49
50int
51vdprintf(int fd, const char * __restrict fmt, va_list ap)
52{
53 FILE f;
54 struct __sfileext fext;
55 unsigned char buf[BUFSIZ];
56 int ret;
57
58 _FILEEXT_SETUP(&f, &fext);
59
60 f._p = buf;
61 f._w = sizeof(buf);
62 f._flags = __SWR;
63 f._file = -1;
64 f._bf._base = buf;
65 f._bf._size = sizeof(buf);
66 f._cookie = &fd;
67 f._write = __dwrite;
68
69 if ((ret = __vfprintf(&f, fmt, ap)) < 0)
70 return ret;
71
72 return fflush(&f) ? EOF : ret;
73}
