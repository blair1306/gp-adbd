29#include <stdarg.h>
30#include <stdio.h>
31#include <wchar.h>
32
33int
34fwscanf(FILE * __restrict fp, const wchar_t * __restrict fmt, ...)
35{
36 va_list ap;
37 int r;
38
39 va_start(ap, fmt);
40 r = vfwscanf(fp, fmt, ap);
41 va_end(ap);
42
43 return (r);
44}
