18#include <string.h>
19#include <stdlib.h>
20
21extern const char *__progname;
22
23void
24setprogname(const char *progname)
25{
26 const char *tmpn;
27
28 tmpn = strrchr(progname, '/');
29 if (tmpn == NULL)
30 __progname = progname;
31 else
32 __progname = tmpn + 1;
33}
