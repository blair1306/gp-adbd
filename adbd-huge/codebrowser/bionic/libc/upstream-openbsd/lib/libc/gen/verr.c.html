31#include <err.h>
32#include <errno.h>
33#include <stdio.h>
34#include <stdlib.h>
35#include <string.h>
36#include <stdarg.h>
37
38extern char *__progname; /* Program name, from crt0. */
39
40__dead void
41verr(int eval, const char *fmt, va_list ap)
42{
43 int sverrno;
44
45 sverrno = errno;
46 (void)fprintf(stderr, "%s: ", __progname);
47 if (fmt != NULL) {
48 (void)vfprintf(stderr, fmt, ap);
49 (void)fprintf(stderr, ": ");
50 }
51 (void)fprintf(stderr, "%s\n", strerror(sverrno));
52 exit(eval);
53}
54DEF_WEAK(verr);
