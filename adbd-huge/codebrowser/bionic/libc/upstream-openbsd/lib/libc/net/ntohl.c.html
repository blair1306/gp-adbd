7#include <sys/types.h>
8#include <endian.h>
9
10#undef ntohl
11
12u_int32_t
13ntohl(u_int32_t x)
14{
15#if BYTE_ORDER == LITTLE_ENDIAN
16 u_char *s = (u_char *)&x;
17 return (u_int32_t)(s[0] << 24 | s[1] << 16 | s[2] << 8 | s[3]);
18#else
19 return x;
20#endif
21}
