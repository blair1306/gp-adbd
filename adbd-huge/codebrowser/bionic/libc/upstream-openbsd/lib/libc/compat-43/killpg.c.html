30#include <sys/types.h>
31#include <signal.h>
32#include <errno.h>
33
34/*
35 * Backwards-compatible killpg().
36 */
37int
38killpg(pid_t pgid, int sig)
39{
40 if (pgid == 1) {
41 errno = ESRCH;
42 return (-1);
43 }
44 return (kill(-pgid, sig));
45}
