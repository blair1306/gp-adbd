17#ifndef ERRNO_RESTORER_H
18#define ERRNO_RESTORER_H
19
20#include <errno.h>
21
22#include "bionic_macros.h"
23
24class ErrnoRestorer {
25 public:
26 explicit ErrnoRestorer() : saved_errno_(errno) {
27 }
28
29 ~ErrnoRestorer() {
30 errno = saved_errno_;
31 }
32
33 void override(int new_errno) {
34 saved_errno_ = new_errno;
35 }
36
37 private:
38 int saved_errno_;
39
40 DISALLOW_COPY_AND_ASSIGN(ErrnoRestorer);
41};
42
43#endif // ERRNO_RESTORER_H
