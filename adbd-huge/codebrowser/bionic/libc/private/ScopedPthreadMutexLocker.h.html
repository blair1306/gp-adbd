17#ifndef SCOPED_PTHREAD_MUTEX_LOCKER_H
18#define SCOPED_PTHREAD_MUTEX_LOCKER_H
19
20#include <pthread.h>
21
22#include "bionic_macros.h"
23
24class ScopedPthreadMutexLocker {
25 public:
26 explicit ScopedPthreadMutexLocker(pthread_mutex_t* mu) : mu_(mu) {
27 pthread_mutex_lock(mu_);
28 }
29
30 ~ScopedPthreadMutexLocker() {
31 pthread_mutex_unlock(mu_);
32 }
33
34 private:
35 pthread_mutex_t* mu_;
36
37 DISALLOW_IMPLICIT_CONSTRUCTORS(ScopedPthreadMutexLocker);
38};
39
40#endif // SCOPED_PTHREAD_MUTEX_LOCKER_H
