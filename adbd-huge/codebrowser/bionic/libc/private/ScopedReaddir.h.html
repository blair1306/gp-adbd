17#ifndef SCOPED_READDIR_H
18#define SCOPED_READDIR_H
19
20#include <dirent.h>
21
22#include "private/bionic_macros.h"
23
24class ScopedReaddir {
25 public:
26 ScopedReaddir(const char* path) : ScopedReaddir(opendir(path)) {
27 }
28
29 ScopedReaddir(DIR* dir) {
30 dir_ = dir;
31 }
32
33 ~ScopedReaddir() {
34 if (dir_ != NULL) {
35 closedir(dir_);
36 }
37 }
38
39 bool IsBad() {
40 return dir_ == NULL;
41 }
42
43 dirent* ReadEntry() {
44 return readdir(dir_);
45 }
46
47 private:
48 DIR* dir_;
49
50 DISALLOW_COPY_AND_ASSIGN(ScopedReaddir);
51};
52
53#endif // SCOPED_READDIR_H
