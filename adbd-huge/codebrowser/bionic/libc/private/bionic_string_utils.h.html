17#ifndef _BIONIC_STRING_UTILS_H_
18#define _BIONIC_STRING_UTILS_H_
19
20#include <string.h>
21
22static inline bool ends_with(const char* s1, const char* s2) {
23 size_t s1_length = strlen(s1);
24 size_t s2_length = strlen(s2);
25 if (s2_length > s1_length) {
26 return false;
27 }
28 return memcmp(s1 + (s1_length - s2_length), s2, s2_length) == 0;
29}

31#endif // _BIONIC_STRING_UTILS_H_
