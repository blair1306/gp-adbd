18#define BIONIC_SYSTRACE_H

20#include "bionic_macros.h"

22// Tracing class for bionic. To begin a trace at a specified point:
23// ScopedTrace("Trace message");
24// The trace will end when the contructor goes out of scope.
25
26class __LIBC_HIDDEN__ ScopedTrace {
27 public:
28 explicit ScopedTrace(const char* message);
29 ~ScopedTrace();
30
31 private:
32 DISALLOW_COPY_AND_ASSIGN(ScopedTrace);
33};
34
35#endif
