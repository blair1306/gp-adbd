47static inline int check_timespec(const timespec* ts, bool null_allowed) {
48 if (null_allowed && ts == nullptr) {
49 return 0;
50 }
51 // glibc just segfaults if you pass a null timespec.
52 // That seems a lot more likely to catch bad code than returning EINVAL.
53 if (ts->tv_nsec < 0 || ts->tv_nsec >= NS_PER_S) {
54 return EINVAL;
55 }
56 if (ts->tv_sec < 0) {
57 return ETIMEDOUT;
58 }
59 return 0;
60}
61
62#if !defined(__LP64__)
63static inline void absolute_timespec_from_timespec(timespec& abs_ts, const timespec& ts, clockid_t clock) {
64 clock_gettime(clock, &abs_ts);
65 abs_ts.tv_sec += ts.tv_sec;
66 abs_ts.tv_nsec += ts.tv_nsec;
67 if (abs_ts.tv_nsec >= NS_PER_S) {
68 abs_ts.tv_nsec -= NS_PER_S;
69 abs_ts.tv_sec++;
70 }
71}
