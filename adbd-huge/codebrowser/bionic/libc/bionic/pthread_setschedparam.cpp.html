29#include <errno.h>
30
31#include "private/ErrnoRestorer.h"
32#include "pthread_internal.h"
33
34int pthread_setschedparam(pthread_t t, int policy, const sched_param* param) {
35 ErrnoRestorer errno_restorer;
36
37 pthread_internal_t* thread = __pthread_internal_find(t);
38 if (thread == NULL) {
39 return ESRCH;
40 }
41
42 int rc = sched_setscheduler(thread->tid, policy, param);
43 if (rc == -1) {
44 return errno;
45 }
46 return 0;
47}
