16#include <sys/cdefs.h>
17
18#include "pthread_internal.h"
19
20class thread_local_dtor {
21 public:
22 void (*func) (void *);
23 void *arg;
24 void *dso_handle; // unused...
25 thread_local_dtor* next;
26};
27
28extern "C" int __cxa_thread_atexit_impl(void (*func) (void *), void *arg, void *dso_handle) {
29 thread_local_dtor* dtor = new thread_local_dtor();
30
31 dtor->func = func;
32 dtor->arg = arg;
33 dtor->dso_handle = dso_handle;
34
35 pthread_internal_t* thread = __get_thread();
36 dtor->next = thread->thread_local_dtors;
37 thread->thread_local_dtors = dtor;
38 return 0;
39}
40
41extern "C" __LIBC_HIDDEN__ void __cxa_thread_finalize() {
42 pthread_internal_t* thread = __get_thread();
43 while (thread->thread_local_dtors != nullptr) {
44 thread_local_dtor* current = thread->thread_local_dtors;
45 thread->thread_local_dtors = current->next;
46
47 current->func(current->arg);
48 delete current;
49 }
50}
