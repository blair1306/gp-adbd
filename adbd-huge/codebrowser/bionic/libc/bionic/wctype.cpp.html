63int iswctype(wint_t wc, wctype_t char_class) {
64 switch (char_class) {
65 case WC_TYPE_ALNUM: return iswalnum(wc);
66 case WC_TYPE_ALPHA: return iswalpha(wc);
67 case WC_TYPE_BLANK: return iswblank(wc);
68 case WC_TYPE_CNTRL: return iswcntrl(wc);
69 case WC_TYPE_DIGIT: return iswdigit(wc);
70 case WC_TYPE_GRAPH: return iswgraph(wc);
71 case WC_TYPE_LOWER: return iswlower(wc);
72 case WC_TYPE_PRINT: return iswprint(wc);
73 case WC_TYPE_PUNCT: return iswpunct(wc);
74 case WC_TYPE_SPACE: return iswspace(wc);
75 case WC_TYPE_UPPER: return iswupper(wc);
76 case WC_TYPE_XDIGIT: return iswxdigit(wc);
77 default: return 0;
78 }
79}

91wctype_t wctype(const char* property) {
92 static const char* const properties[WC_TYPE_MAX] = {
93 "<invalid>",
94 "alnum", "alpha", "blank", "cntrl", "digit", "graph",
95 "lower", "print", "punct", "space", "upper", "xdigit"
96 };
97 for (size_t i = 0; i < WC_TYPE_MAX; ++i) {
98 if (!strcmp(properties[i], property)) {
99 return static_cast<wctype_t>(i);
100 }
101 }
102 return static_cast<wctype_t>(0);
103}

105wctype_t wctype_l(const char* property, locale_t) {
