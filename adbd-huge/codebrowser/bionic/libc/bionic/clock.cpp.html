29#include <time.h>
30#include <sys/sysconf.h>
31#include <sys/times.h>
32
33#include "private/bionic_constants.h"

35// http://pubs.opengroup.org/onlinepubs/9699919799/functions/clock.html
36clock_t clock() {
37 timespec ts;
38 if (clock_gettime(CLOCK_PROCESS_CPUTIME_ID, &ts) == -1) {
39 return -1;
40 }
41 return (ts.tv_sec * CLOCKS_PER_SEC) + (ts.tv_nsec / (NS_PER_S / CLOCKS_PER_SEC));
42}
