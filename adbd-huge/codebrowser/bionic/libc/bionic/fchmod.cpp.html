29#include <fcntl.h>
30#include <sys/stat.h>
31#include <sys/types.h>
32#include <errno.h>
33#include <unistd.h>
34#include <stdio.h>
35
36extern "C" int ___fchmod(int, mode_t);
37
38int fchmod(int fd, mode_t mode) {
39 int saved_errno = errno;
40 int result = ___fchmod(fd, mode);
41
42 if ((result == 0) || (errno != EBADF)) {
43 return result;
44 }
45
46 // fd could be an O_PATH file descriptor, and the kernel
47 // may not directly support fchmod() on such a file descriptor.
48 // Use /proc/self/fd instead to emulate this support.
49 // https://sourceware.org/bugzilla/show_bug.cgi?id=14578
50 //
51 // As of February 2015, there are no kernels which support fchmod
52 // on an O_PATH file descriptor, and "man open" documents fchmod
53 // on O_PATH file descriptors as returning EBADF.
54 int fd_flag = fcntl(fd, F_GETFL);
55 if ((fd_flag == -1) || ((fd_flag & O_PATH) == 0)) {
56 errno = EBADF;
57 return -1;
58 }
59
60 char buf[40];
61 snprintf(buf, sizeof(buf), "/proc/self/fd/%d", fd);
62 errno = saved_errno;
63 result = chmod(buf, mode);
64 if ((result == -1) && (errno == ELOOP)) {
65 // Linux does not support changing the mode of a symlink.
66 // For fchmodat(AT_SYMLINK_NOFOLLOW), POSIX requires a return
67 // value of ENOTSUP. Assume that's true here too.
68 errno = ENOTSUP;
69 }
70
71 return result;
72}
