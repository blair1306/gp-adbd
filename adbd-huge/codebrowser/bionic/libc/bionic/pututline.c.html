28#include <string.h>
29#include <stdio.h>
30#include <utmp.h>
31
32
33void pututline(struct utmp* utmp)
34{
35 FILE* f;
36 struct utmp u;
37 long i;
38
39 if (!(f = fopen(_PATH_UTMP, "w+e")))
40 return;
41
42 while (fread(&u, sizeof(struct utmp), 1, f) == 1)
43 {
44 if (!strncmp(utmp->ut_line, u.ut_line, sizeof(u.ut_line) -1))
45 {
46 if ((i = ftell(f)) < 0)
47 goto ret;
48 if (fseek(f, i - sizeof(struct utmp), SEEK_SET) < 0)
49 goto ret;
50 fwrite(utmp, sizeof(struct utmp), 1, f);
51 goto ret;
52 }
53 }
54
55
56 fclose(f);
57
58 if (!(f = fopen(_PATH_UTMP, "w+e")))
59 return;
60 fwrite(utmp, sizeof(struct utmp), 1, f);
61
62ret:
63 fclose(f);
64}
