29#include <stddef.h>
30#include <sys/cdefs.h>
31#include <sys/auxv.h>
32#include <private/bionic_auxv.h>
33#include <elf.h>
34#include <errno.h>
35
36__LIBC_HIDDEN__ ElfW(auxv_t)* __libc_auxv = NULL;
37
38extern "C" unsigned long int getauxval(unsigned long int type) {
39 for (ElfW(auxv_t)* v = __libc_auxv; v->a_type != AT_NULL; ++v) {
40 if (v->a_type == type) {
41 return v->a_un.a_val;
42 }
43 }
44 errno = ENOENT;
45 return 0;
46}
