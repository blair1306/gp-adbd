29#include <sys/ioctl.h>
30#include <stdarg.h>
31
32extern "C" int __ioctl(int, int, void *);
33
34int ioctl(int fd, int request, ...) {
35 va_list ap;
36 va_start(ap, request);
37 void* arg = va_arg(ap, void*);
38 va_end(ap);
39 return __ioctl(fd, request, arg);
40}
