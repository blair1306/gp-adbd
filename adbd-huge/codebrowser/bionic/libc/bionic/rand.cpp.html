17#include <stdlib.h>
18
19// The BSD rand/srand is very weak. glibc just uses random/srandom instead.
20// Since we're likely to run code intended for glibc, and POSIX doesn't seem
21// to disallow this, we go that route too.
22
23int rand() {
24 return random();
25}
26
27void srand(unsigned int seed) {
28 return srandom(seed);
29}
