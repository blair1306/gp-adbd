33extern "C" int __signalfd4(int fd, kernel_sigset_t* mask, size_t sizemask, int flags);
34
35int signalfd(int fd, const sigset_t* mask, int flags) {
36 kernel_sigset_t in_set(mask);
37 return __signalfd4(fd, &in_set, sizeof(in_set), flags);
38}
