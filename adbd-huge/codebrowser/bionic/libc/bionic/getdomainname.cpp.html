29#include <errno.h>
30#include <string.h>
31#include <unistd.h>
32#include <sys/utsname.h>
33
34int getdomainname(char* name, size_t len) {
35 utsname uts;
36 if (uname(&uts) == -1) return -1;
37
38 // Note: getdomainname()'s behavior varies across implementations when len is
39 // too small. bionic follows the historical libc policy of returning EINVAL,
40 // instead of glibc's policy of copying the first len bytes without a NULL
41 // terminator.
42 if (strlen(uts.domainname) >= len) {
43 errno = EINVAL;
44 return -1;
45 }
46
47 strncpy(name, uts.domainname, len);
48 return 0;
49}
