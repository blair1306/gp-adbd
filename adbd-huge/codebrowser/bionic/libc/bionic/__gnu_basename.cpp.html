29#define _GNU_SOURCE 1
30#include <string.h>
31
32extern "C" const char* __gnu_basename(const char* path) {
33 const char* last_slash = strrchr(path, '/');
34 return (last_slash != NULL) ? last_slash + 1 : path;
35}
