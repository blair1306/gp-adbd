17#include <errno.h>
18#include <sys/cdefs.h>
19#include <unistd.h>
20
21#if !defined(__LP64__)
22static_assert(sizeof(off_t) == 4,
23 "libc can't be built with _FILE_OFFSET_BITS=64.");
24
25// The kernel's implementation of ftruncate uses an unsigned long for the length
26// parameter, so it will not catch negative values. On the other hand
27// ftruncate64 does check for this, so just forward the call.
28int ftruncate(int filedes, off_t length) {
29 return ftruncate64(filedes, length);
30}
31#endif // !defined(__LP64__)
