29#include <errno.h>
30#include <pthread.h>
31
32#include "pthread_internal.h"
33
34int pthread_detach(pthread_t t) {
35 pthread_internal_t* thread = __pthread_internal_find(t);
36 if (thread == NULL) {
37 return ESRCH;
38 }
39
40 ThreadJoinState old_state = THREAD_NOT_JOINED;
41 while (old_state == THREAD_NOT_JOINED &&
42 !atomic_compare_exchange_weak(&thread->join_state, &old_state, THREAD_DETACHED)) {
43 }
44
45 if (old_state == THREAD_NOT_JOINED) {
46 return 0;
47 } else if (old_state == THREAD_EXITED_NOT_JOINED) {
48 // Use pthread_join to clean it up.
49 return pthread_join(t, NULL);
50 }
51 return EINVAL;
52}
