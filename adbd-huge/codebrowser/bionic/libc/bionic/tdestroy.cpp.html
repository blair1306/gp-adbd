21// Destroy a tree and free all allocated resources.
22// This is a GNU extension, not available from BSD.
23void tdestroy(void* root, void (*destroy_func)(void*)) {
24 node_t* root_node = reinterpret_cast<node_t*>(root);
25 if (root_node == NULL) {
26 return;
27 }
28 if (root_node->llink) {
29 tdestroy(root_node->llink, destroy_func);
30 }
31 if (root_node->rlink) {
32 tdestroy(root_node->rlink, destroy_func);
33 }
34 (*destroy_func)(root_node->key);
35 free(root);
36}
