29#include <string.h>
30#include "private/ThreadLocalBuffer.h"
31
32extern "C" const char* __strerror_lookup(int);
33
34static ThreadLocalBuffer<char, NL_TEXTMAX> g_strerror_tls_buffer;
35
36char* strerror(int error_number) {
37 // Just return the original constant in the easy cases.
38 char* result = const_cast<char*>(__strerror_lookup(error_number));
39 if (result != nullptr) {
40 return result;
41 }
42
43 result = g_strerror_tls_buffer.get();
44 strerror_r(error_number, result, g_strerror_tls_buffer.size());
45 return result;
46}
