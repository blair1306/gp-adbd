98 if (supplied_size < 0) {

105size_t __fread_chk(void* __restrict buf, size_t size, size_t count,
106 FILE* __restrict stream, size_t buf_size) {
107 size_t total;
108 if (__predict_false(__size_mul_overflow(size, count, &total))) {
109 // overflow: trigger the error path in fread
110 return fread(buf, size, count, stream);
111 }
112 __check_buffer_access("fread", "write into", total, buf_size);
113 return fread(buf, size, count, stream);
114}

241// This is a variant of __stpncpy_chk, but it also checks to make
242// sure we don't read beyond the end of "src". The code for this is
243// based on the original version of stpncpy, but modified to check
244// how much we read from "src" at the end of the copy operation.
245char* __stpncpy_chk2(char* __restrict dst, const char* __restrict src,
246 size_t n, size_t dst_len, size_t src_len) {
247 __check_buffer_access("stpncpy", "write into", n, dst_len);
248 if (n != 0) {
249 char* d = dst;
250 const char* s = src;
251
252 do {
253 if ((*d++ = *s++) == 0) {
254 // NUL pad the remaining n-1 bytes.
255 while (--n != 0) {
256 *d++ = 0;
257 }
258 break;
259 }
260 } while (--n != 0);
261
262 size_t s_copy_len = static_cast<size_t>(s - src);
263 if (__predict_false(s_copy_len > src_len)) {
264 __fortify_fatal("stpncpy: detected read past end of %zu-byte buffer", src_len);
265 }
266 }
267
268 return dst;
269}

271char* __strchr_chk(const char* p, int ch, size_t s_len) {
272 for (;; ++p, s_len--) {
273 if (__predict_false(s_len == 0)) {
274 __fortify_fatal("strchr: prevented read past end of buffer");
275 }
276 if (*p == static_cast<char>(ch)) {
277 return const_cast<char*>(p);
278 }
279 if (*p == '\0') {
280 return NULL;
281 }
282 }
283}

297size_t __strlen_chk(const char* s, size_t s_len) {
298 // TODO: "prevented" here would be a lie because this strlen can run off the end.
299 // strlen is too important to be expensive, so we wanted to be able to call the optimized
300 // implementation, but I think we need to implement optimized assembler __strlen_chk routines.
301 size_t ret = strlen(s);
302 if (__predict_false(ret >= s_len)) {
303 __fortify_fatal("strlen: detected read past end of buffer");
304 }
305 return ret;
306}

308// Runtime implementation of __builtin____strncat_chk (used directly by compiler, not in headers).
309extern "C" char* __strncat_chk(char* __restrict dst, const char* __restrict src,
310 size_t len, size_t dst_buf_size) {
311 if (len == 0) {
312 return dst;
313 }
314
315 size_t dst_len = __strlen_chk(dst, dst_buf_size);
316 char* d = dst + dst_len;
317 dst_buf_size -= dst_len;
318
319 while (*src != '\0') {
320 *d++ = *src++;
321 len--; dst_buf_size--;
322
323 if (__predict_false(dst_buf_size == 0)) {
324 __fortify_fatal("strncat: prevented write past end of buffer");
325 }
326
327 if (len == 0) {
328 break;
329 }
330 }
331
332 *d = '\0';
333 return dst;
334}

373char* __strrchr_chk(const char* p, int ch, size_t s_len) {
374 for (const char* save = NULL;; ++p, s_len--) {
375 if (s_len == 0) {
376 __fortify_fatal("strrchr: prevented read past end of buffer");
377 }
378 if (*p == static_cast<char>(ch)) {
379 save = p;
380 }
381 if (!*p) {
382 return const_cast<char*>(save);
383 }
384 }
385}
