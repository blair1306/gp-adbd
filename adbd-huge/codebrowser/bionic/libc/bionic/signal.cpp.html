29#include <signal.h>
30
31__LIBC_HIDDEN__ sighandler_t _signal(int signum, sighandler_t handler, int flags) {
32 struct sigaction sa;
33 sigemptyset(&sa.sa_mask);
34 sa.sa_handler = handler;
35 sa.sa_flags = flags;
36
37 if (sigaction(signum, &sa, &sa) == -1) {
38 return SIG_ERR;
39 }
40
41 return sa.sa_handler;
42}
43
44sighandler_t signal(int signum, sighandler_t handler) {
45 return _signal(signum, handler, SA_RESTART);
46}
