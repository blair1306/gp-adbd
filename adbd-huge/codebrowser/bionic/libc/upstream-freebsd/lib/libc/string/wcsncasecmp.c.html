27#include <sys/cdefs.h>

30#include <wchar.h>
31#include <wctype.h>
32
33int
34wcsncasecmp(const wchar_t *s1, const wchar_t *s2, size_t n)
35{
36 wchar_t c1, c2;
37
38 if (n == 0)
39 return (0);
40 for (; *s1; s1++, s2++) {
41 c1 = towlower(*s1);
42 c2 = towlower(*s2);
43 if (c1 != c2)
44 return ((int)c1 - c2);
45 if (--n == 0)
46 return (0);
47 }
48 return (-*s2);
49}
