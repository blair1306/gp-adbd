39#include <wchar.h>

41int
42wcsncmp(const wchar_t *s1, const wchar_t *s2, size_t n)
43{
44
45 if (n == 0)
46 return (0);
47 do {
48 if (*s1 != *s2++) {
49 /* XXX assumes wchar_t = int */
50 return (*(const unsigned int *)s1 -
51 *(const unsigned int *)--s2);
52 }
53 if (*s1++ == 0)
54 break;
55 } while (--n != 0);
56 return (0);
57}
