28#include <sys/cdefs.h>
29#include <netdb.h>
30#include "servent.h"
31
32struct servent *
33getservbyport(int port, const char *proto)
34{
35 res_static rs = __res_get_static();
36
37 if (rs == NULL || proto == NULL) {
38 errno = EINVAL;
39 return NULL;
40 }
41
42 rs->servent_ptr = NULL;
43 while (1) {
44 struct servent* s = getservent_r(rs);
45 if (s == NULL)
46 break;
47 if ( s->s_port == port && !strcmp( s->s_proto, proto ) )
48 return s;
49 }
50
51 return NULL;
52}
