76/*
77 * Revision information. This is the release date in YYYYMMDD format.
78 * It can change every day so the right thing to do with it is use it
79 * in preprocessor commands such as "#if (__RES > 19931104)". Do not
80 * compare for equality; rather, use it to determine whether your resolver
81 * is new enough to contain a certain feature.

86/*
87 * This used to be defined in res_query.c, now it's in herror.c.
88 * [XXX no it's not. It's in irs/irs_data.c]
89 * It was
90 * never extern'd by any *.h file before it was placed here. For thread
91 * aware programs, the last h_errno value set is stored in res->h_errno.
92 *
93 * XXX: There doesn't seem to be a good reason for exposing RES_SET_H_ERRNO
94 * (and __h_errno_set) to the public via <resolv.h>.
95 * XXX: __h_errno_set is really part of IRS, not part of the resolver.
96 * If somebody wants to build and use a resolver that doesn't use IRS,
97 * what do they do? Perhaps something like
98 * #ifdef WANT_IRS
99 * # define RES_SET_H_ERRNO(r,x) __h_errno_set(r,x)
100 * #else
101 * # define RES_SET_H_ERRNO(r,x) (h_errno = (r)->res_h_errno = (x))
102 * #endif
103 */
104
105#define RES_SET_H_ERRNO(r,x) (h_errno = (r)->res_h_errno = (x))

156struct __res_state {
157 unsigned netid; /* NetId: cache key and socket mark */
158 int retrans; /* retransmission time interval */
159 int retry; /* number of times to retransmit */
160#ifdef sun
161 u_int options; /* option flags - see below. */
162#else
163 u_long options; /* option flags - see below. */
164#endif
165 int nscount; /* number of name servers */
166 struct sockaddr_in
167 nsaddr_list[MAXNS]; /* address of name server */
168#define nsaddr nsaddr_list[0] /* for backward compatibility */
169 u_short id; /* current message id */
170 char *dnsrch[MAXDNSRCH+1]; /* components of domain to search */
171 char defdname[256]; /* default domain (deprecated) */
172#ifdef sun
173 u_int pfcode; /* RES_PRF_ flags - see below. */
174#else
175 u_long pfcode; /* RES_PRF_ flags - see below. */
176#endif
177 unsigned ndots:4; /* threshold for initial abs. query */
178 unsigned nsort:4; /* number of elements in sort_list[] */
179 char unused[3];
180 struct {
181 struct in_addr addr;
182 uint32_t mask;
183 } sort_list[MAXRESOLVSORT];
184#ifdef __OLD_RES_STATE
185 char lookups[4];
186#else
187 res_send_qhook qhook; /* query hook */
188 res_send_rhook rhook; /* response hook */
189 int res_h_errno; /* last one set for this context */
190 unsigned _mark; /* If non-0 SET_MARK to _mark on all request sockets */
191 int _vcsock; /* PRIVATE: for res_send VC i/o */
192 u_int _flags; /* PRIVATE: see below */
193 u_int _pad; /* make _u 64 bit aligned */
194 union {
195 /* On an 32-bit arch this means 512b total. */
196 char pad[72 - 4*sizeof (int) - 2*sizeof (void *)];
197 struct {
198 uint16_t nscount;
199 uint16_t nstimes[MAXNS]; /* ms. */
200 int nssocks[MAXNS];
201 struct __res_state_ext *ext; /* extention for IPv6 */
202 } _ext;
203 } _u;
204#endif
205 struct res_static rstatic[1];
206};

241/*
242 * Resolver flags (used to be discrete per-module statics ints).
243 */
244#define RES_F_VC 0x00000001 /* socket is TCP */
245#define RES_F_CONN 0x00000002 /* socket is connected */
246#define RES_F_EDNS0ERR 0x00000004 /* EDNS0 caused errors */
247#define RES_F__UNUSED 0x00000008 /* (unused) */
248#define RES_F_LASTMASK 0x000000F0 /* ordinal server of last res_nsend */
249#define RES_F_LASTSHIFT 4 /* bit position of LASTMASK "flag" */
250#define RES_GETLAST(res) (((res)._flags & RES_F_LASTMASK) >> RES_F_LASTSHIFT)
251
252/* res_findzonecut2() options */
253#define RES_EXHAUSTIVE 0x00000001 /* always do all queries */
254#define RES_IPV4ONLY 0x00000002 /* IPv4 only */
255#define RES_IPV6ONLY 0x00000004 /* IPv6 only */

257/*
258 * Resolver options (keep these in synch with res_debug.c, please)
259 */
260#define RES_INIT 0x00000001 /* address initialized */
261#define RES_DEBUG 0x00000002 /* print debug messages */
262#define RES_AAONLY 0x00000004 /* authoritative answers only (!IMPL)*/
263#define RES_USEVC 0x00000008 /* use virtual circuit */
264#define RES_PRIMARY 0x00000010 /* query primary server only (!IMPL) */
265#define RES_IGNTC 0x00000020 /* ignore trucation errors */
266#define RES_RECURSE 0x00000040 /* recursion desired */
267#define RES_DEFNAMES 0x00000080 /* use default domain name */
268#define RES_STAYOPEN 0x00000100 /* Keep TCP socket open */
269#define RES_DNSRCH 0x00000200 /* search up local domain tree */
270#define RES_INSECURE1 0x00000400 /* type 1 security disabled */
271#define RES_INSECURE2 0x00000800 /* type 2 security disabled */
272#define RES_NOALIASES 0x00001000 /* shuts off HOSTALIASES feature */
273#define RES_USE_INET6 0x00002000 /* use/map IPv6 in gethostbyname() */
274#define RES_ROTATE 0x00004000 /* rotate ns list after each query */
275#define RES_NOCHECKNAME 0x00008000 /* do not check names for sanity. */
276#define RES_KEEPTSIG 0x00010000 /* do not strip TSIG records */
277#define RES_BLAST 0x00020000 /* blast all recursive servers */
278#define RES_NOTLDQUERY 0x00100000 /* don't unqualified name as a tld */
279#define RES_USE_DNSSEC 0x00200000 /* use DNSSEC using OK bit in OPT */
280/* #define RES_DEBUG2 0x00400000 */ /* nslookup internal */
281/* KAME extensions: use higher bit to avoid conflict with ISC use */
282#define RES_USE_DNAME 0x10000000 /* use DNAME */
283#define RES_USE_EDNS0 0x40000000 /* use EDNS0 if configured */
284#define RES_NO_NIBBLE2 0x80000000 /* disable alternate nibble lookup */
285
286#define RES_DEFAULT (RES_RECURSE | RES_DEFNAMES | \
287 RES_DNSRCH | RES_NO_NIBBLE2)
