28#ifndef _RESOLV_CACHE_H_
29#define _RESOLV_CACHE_H_
30
31#include <stddef.h>
32#include <sys/cdefs.h>
33
34struct __res_state;
35
36/* sets the name server addresses to the provided res_state structure. The
37 * name servers are retrieved from the cache which is associated
38 * with the network to which the res_state structure is associated */
39__LIBC_HIDDEN__
40extern void _resolv_populate_res_for_net(struct __res_state* statp);
41
42typedef enum {
43 RESOLV_CACHE_UNSUPPORTED, /* the cache can't handle that kind of queries */
44 /* or the answer buffer is too small */
45 RESOLV_CACHE_NOTFOUND, /* the cache doesn't know about this query */
46 RESOLV_CACHE_FOUND /* the cache found the answer */
47} ResolvCacheStatus;
48
49__LIBC_HIDDEN__
50extern ResolvCacheStatus
51_resolv_cache_lookup( unsigned netid,
52 const void* query,
53 int querylen,
54 void* answer,
55 int answersize,
56 int *answerlen );
57
58/* add a (query,answer) to the cache, only call if _resolv_cache_lookup
59 * did return RESOLV_CACHE_NOTFOUND
60 */
61__LIBC_HIDDEN__
62extern void
63_resolv_cache_add( unsigned netid,
64 const void* query,
65 int querylen,
66 const void* answer,
67 int answerlen );
68
69/* Notify the cache a request failed */
70__LIBC_HIDDEN__
71extern void
72_resolv_cache_query_failed( unsigned netid,
73 const void* query,
74 int querylen);
75
76#endif /* _RESOLV_CACHE_H_ */
