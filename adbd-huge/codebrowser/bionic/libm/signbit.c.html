29#include <math.h>
30
31#include "fpmath.h"
32
33int __signbit(double d)
34{
35 union IEEEd2bits u;
36
37 u.d = d;
38 return (u.bits.sign);
39}
40
41int __signbitf(float f)
42{
43 union IEEEf2bits u;
44
45 u.f = f;
46 return (u.bits.sign);
47}
48
49#ifdef __LP64__
50int __signbitl(long double e)
51{
52 union IEEEl2bits u;
53
54 u.e = e;
55 return (u.bits.sign);
56}
57#else // __LP32__
58__weak_reference(__signbit, __signbitl);
59#endif // __LP64__
