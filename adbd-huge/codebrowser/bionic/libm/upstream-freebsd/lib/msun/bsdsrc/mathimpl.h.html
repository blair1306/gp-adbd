45/*
46 * TRUNC() is a macro that sets the trailing 27 bits in the mantissa of an
47 * IEEE double variable to zero. It must be expression-like for syntactic
48 * reasons, and we implement this expression using an inline function
49 * instead of a pure macro to avoid depending on the gcc feature of
50 * statement-expressions.
51 */
52#define TRUNC(d) (_b_trunc(&(d)))
53
54static __inline void
55_b_trunc(volatile double *_dp)
56{
57 uint32_t _lw;
58
59 GET_LOW_WORD(_lw, *_dp);
60 SET_LOW_WORD(*_dp, _lw & 0xf8000000);
61}

66};
