193long double
194erfl(long double x)
195{
196 long double ax,R,S,P,Q,s,y,z,r;
197 uint64_t lx, llx;
198 int32_t i;
199 uint16_t hx;
200
201 EXTRACT_LDBL128_WORDS(hx, lx, llx, x);
202
203 if((hx & 0x7fff) == 0x7fff) { /* erfl(nan)=nan */
204 i = (hx>>15)<<1;
205 return (1-i)+one/x; /* erfl(+-inf)=+-1 */
206 }
207
208 ax = fabsl(x);
209 if(ax < 0.84375) {
210 if(ax < 0x1p-40L) {
211 if(ax < 0x1p-16373L)
212 return (8*x+efx8*x)/8; /* avoid spurious underflow */
213 return x + efx*x;
214 }
215 z = x*x;
216 r = pp0+z*(pp1+z*(pp2+z*(pp3+z*(pp4+z*(pp5+z*(pp6+z*(pp7+
217 z*(pp8+z*pp9))))))));
218 s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*(qq5+z*(qq6+z*(qq7+
219 z*(qq8+z*qq9))))))));
220 y = r/s;
221 return x + x*y;
222 }
223 if(ax < 1.25) {
224 s = ax-one;
225 P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*(pa6+s*(pa7+
226 s*(pa8+s*(pa9+s*(pa10+s*pa11))))))))));
227 Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*(qa6+s*(qa7+
228 s*(qa8+s*(qa9+s*(qa10+s*(qa11+s*qa12)))))))))));
229 if(x>=0) return (erx + P/Q); else return (-erx - P/Q);
230 }
231 if (ax >= 9) { /* inf>|x|>= 9 */
232 if(x>=0) return (one-tiny); else return (tiny-one);
233 }
234 s = one/(ax*ax);
235 if(ax < 2.85715) { /* |x| < 2.85715 */
236 R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(ra5+s*(ra6+s*(ra7+
237 s*(ra8+s*(ra9+s*(ra10+s*(ra11+s*(ra12+s*(ra13+s*(ra14+
238 s*(ra15+s*ra16)))))))))))))));
239 S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(sa5+s*(sa6+s*(sa7+
240 s*(sa8+s*(sa9+s*(sa10+s*(sa11+s*(sa12+s*(sa13+s*(sa14+
241 s*(sa15+s*sa16)))))))))))))));
242 } else { /* |x| >= 2.85715 */
243 R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(rb5+s*(rb6+s*(rb7+
244 s*(rb8+s*(rb9+s*(rb10+s*(rb11+s*(rb12+s*(rb13+
245 s*rb14)))))))))))));
246 S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(sb5+s*(sb6+s*(sb7+
247 s*(sb8+s*(sb9+s*(sb10+s*(sb11+s*(sb12+s*(sb13+
248 s*sb14)))))))))))));
249 }
250 z = (float)ax;
251 r = expl(-z*z-0.5625)*expl((z-ax)*(z+ax)+R/S);
252 if(x>=0) return (one-r/ax); else return (r/ax-one);
253}

255long double
256erfcl(long double x)
257{
258 long double ax,R,S,P,Q,s,y,z,r;
259 uint64_t lx, llx;
260 uint16_t hx;
261
262 EXTRACT_LDBL128_WORDS(hx, lx, llx, x);
263
264 if((hx & 0x7fff) == 0x7fff) { /* erfcl(nan)=nan */
265 /* erfcl(+-inf)=0,2 */
266 return ((hx>>15)<<1)+one/x;
267 }
268
269 ax = fabsl(x);
270 if(ax < 0.84375L) {
271 if(ax < 0x1p-34L)
272 return one-x;
273 z = x*x;
274 r = pp0+z*(pp1+z*(pp2+z*(pp3+z*(pp4+z*(pp5+z*(pp6+z*(pp7+
275 z*(pp8+z*pp9))))))));
276 s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*(qq5+z*(qq6+z*(qq7+
277 z*(qq8+z*qq9))))))));
278 y = r/s;
279 if(ax < 0.25L) { /* x<1/4 */
280 return one-(x+x*y);
281 } else {
282 r = x*y;
283 r += (x-half);
284 return half - r;
285 }
286 }
287 if(ax < 1.25L) {
288 s = ax-one;
289 P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*(pa6+s*(pa7+
290 s*(pa8+s*(pa9+s*(pa10+s*pa11))))))))));
291 Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*(qa6+s*(qa7+
292 s*(qa8+s*(qa9+s*(qa10+s*(qa11+s*qa12)))))))))));
293 if(x>=0) {
294 z = one-erx; return z - P/Q;
295 } else {
296 z = erx+P/Q; return one+z;
297 }
298 }
299
300 if(ax < 108) { /* |x| < 108 */
301 s = one/(ax*ax);
302 if(ax < 2.85715) { /* |x| < 2.85715 */
303 R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(ra5+s*(ra6+s*(ra7+
304 s*(ra8+s*(ra9+s*(ra10+s*(ra11+s*(ra12+s*(ra13+s*(ra14+
305 s*(ra15+s*ra16)))))))))))))));
306 S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(sa5+s*(sa6+s*(sa7+
307 s*(sa8+s*(sa9+s*(sa10+s*(sa11+s*(sa12+s*(sa13+s*(sa14+
308 s*(sa15+s*sa16)))))))))))))));
309 } else if(ax < 9) {
310 R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(rb5+s*(rb6+s*(rb7+
311 s*(rb8+s*(rb9+s*(rb10+s*(rb11+s*(rb12+s*(rb13+
312 s*rb14)))))))))))));
313 S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(sb5+s*(sb6+s*(sb7+
314 s*(sb8+s*(sb9+s*(sb10+s*(sb11+s*(sb12+s*(sb13+
315 s*sb14)))))))))))));
316 } else {
317 if(x < -9) return two-tiny; /* x < -9 */
318 R=rc0+s*(rc1+s*(rc2+s*(rc3+s*(rc4+s*(rc5+s*(rc6+s*(rc7+
319 s*(rc8+s*rc9))))))));
320 S=one+s*(sc1+s*(sc2+s*(sc3+s*(sc4+s*(sc5+s*(sc6+s*(sc7+
321 s*(sc8+s*sc9))))))));
322 }
323 z = (float)ax;
324 r = expl(-z*z-0.5625)*expl((z-ax)*(z+ax)+R/S);

326 } else {
