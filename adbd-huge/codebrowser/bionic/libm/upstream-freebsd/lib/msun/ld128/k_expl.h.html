232/*
233 * Kernel for expl(x). x must be finite and not tiny or huge.
234 * "tiny" is anything that would make us underflow (|A6*x^6| < ~LDBL_MIN).
235 * "huge" is anything that would make fn*L1 inexact (|x| > ~2**17*ln2).
236 */
237static inline void
238__k_expl(long double x, long double *hip, long double *lop, int *kp)
239{
240 long double q, r, r1, t;
241 double dr, fn, r2;
242 int n, n2;
243
244 /* Reduce x to (k*ln2 + endpoint[n2] + r1 + r2). */
245 /* Use a specialized rint() to get fn. Assume round-to-nearest. */
246 /* XXX assume no extra precision for the additions, as for trig fns. */
247 /* XXX this set of comments is now quadruplicated. */
248 /* XXX but see ../src/e_exp.c for a fix using double_t. */
249 fn = (double)x * INV_L + 0x1.8p52 - 0x1.8p52;
250#if defined(HAVE_EFFICIENT_IRINT)
251 n = irint(fn);
252#else
253 n = (int)fn;
254#endif
255 n2 = (unsigned)n % INTERVALS;
256 /* Depend on the sign bit being propagated: */
257 *kp = n >> LOG2_INTERVALS;
258 r1 = x - fn * L1;
259 r2 = fn * -L2;
260 r = r1 + r2;
261
262 /* Evaluate expl(endpoint[n2] + r1 + r2) = tbl[n2] * expl(r1 + r2). */
263 dr = r;
264 q = r2 + r * r * (A2 + r * (A3 + r * (A4 + r * (A5 + r * (A6 +
265 dr * (A7 + dr * (A8 + dr * (A9 + dr * A10))))))));
266 t = tbl[n2].lo + tbl[n2].hi;
267 *hip = tbl[n2].hi;
268 *lop = tbl[n2].lo + t * (q + r1);
269}
270
271/*
272 * XXX: the rest of the functions are identical for ld80 and ld128.
273 * However, we should use scalbnl() for ld128, since long double
274 * multiplication is very slow on the only supported ld128 arch (sparc64).
275 */
276
277static inline void
278k_hexpl(long double x, long double *hip, long double *lop)
279{
280 float twopkm1;
281 int k;
282
283 __k_expl(x, hip, lop, &k);
284 SET_FLOAT_WORD(twopkm1, 0x3f800000 + ((k - 1) << 23));
285 *hip *= twopkm1;
286 *lop *= twopkm1;
287}
288
289static inline long double
290hexpl(long double x)
291{
292 long double hi, lo, twopkm2;
293 int k;
294
295 twopkm2 = 1;
296 __k_expl(x, &hi, &lo, &k);
297 SET_LDBL_EXPSIGN(twopkm2, BIAS + k - 2);
298 return (lo + hi) * 2 * twopkm2;
299}

305static inline long double complex
306__ldexp_cexpl(long double complex z, int expt)
307{
308 long double exp_x, hi, lo;
309 long double x, y, scale1, scale2;
310 int half_expt, k;
311
312 x = creall(z);
313 y = cimagl(z);
314 __k_expl(x, &hi, &lo, &k);
315
316 exp_x = (lo + hi) * 0x1p16382;
317 expt += k - 16382;
318
319 scale1 = 1;
320 half_expt = expt / 2;
321 SET_LDBL_EXPSIGN(scale1, BIAS + half_expt);
322 scale2 = 1;
323 SET_LDBL_EXPSIGN(scale1, BIAS + expt - half_expt);
324
325 return (CMPLXL(cos(y) * exp_x * scale1 * scale2,
326 sinl(y) * exp_x * scale1 * scale2));
327}
