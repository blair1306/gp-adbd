323/*
324 * exp2l(x): compute the base 2 exponential of x
325 *
326 * Accuracy: Peak error < 0.502 ulp.
327 *
328 * Method: (accurate tables)
329 *
330 * Reduce x:
331 * x = 2**k + y, for integer k and |y| <= 1/2.
332 * Thus we have exp2(x) = 2**k * exp2(y).
333 *
334 * Reduce y:
335 * y = i/TBLSIZE + z - eps[i] for integer i near y * TBLSIZE.
336 * Thus we have exp2(y) = exp2(i/TBLSIZE) * exp2(z - eps[i]),
337 * with |z - eps[i]| <= 2**-8 + 2**-98 for the table used.
338 *
339 * We compute exp2(i/TBLSIZE) via table lookup and exp2(z - eps[i]) via
340 * a degree-10 minimax polynomial with maximum error under 2**-120.
341 * The values in exp2t[] and eps[] are chosen such that
342 * exp2t[i] = exp2(i/TBLSIZE + eps[i]), and eps[i] is a small offset such
343 * that exp2t[i] is accurate to 2**-122.
344 *
345 * Note that the range of i is +-TBLSIZE/2, so we actually index the tables
346 * by i0 = i + TBLSIZE/2.
347 *
348 * This method is due to Gal, with many details due to Gal and Bachelis:
349 *
350 * Gal, S. and Bachelis, B. An Accurate Elementary Mathematical Library
351 * for the IEEE Floating Point Standard. TOMS 17(1), 26-46 (1991).
352 */
353long double
354exp2l(long double x)
355{
356 union IEEEl2bits u, v;
357 long double r, t, twopk, twopkp10000, z;
358 uint32_t hx, ix, i0;
359 int k;
360
361 u.e = x;
362
363 /* Filter out exceptional cases. */
364 hx = u.xbits.expsign;
365 ix = hx & EXPMASK;
366 if (ix >= BIAS + 14) { /* |x| >= 16384 */
367 if (ix == BIAS + LDBL_MAX_EXP) {
368 if (u.xbits.manh != 0
369 || u.xbits.manl != 0
370 || (hx & 0x8000) == 0)
371 return (x + x); /* x is NaN or +Inf */
372 else
373 return (0.0); /* x is -Inf */
374 }
375 if (x >= 16384)
376 return (huge * huge); /* overflow */
377 if (x <= -16495)
378 return (twom10000 * twom10000); /* underflow */
379 } else if (ix <= BIAS - 115) { /* |x| < 0x1p-115 */
380 return (1.0 + x);
381 }
382
383 /*
384 * Reduce x, computing z, i0, and k. The low bits of x + redux
385 * contain the 16-bit integer part of the exponent (k) followed by
386 * TBLBITS fractional bits (i0). We use bit tricks to extract these
387 * as integers, then set z to the remainder.
388 *
389 * Example: Suppose x is 0xabc.123456p0 and TBLBITS is 8.
390 * Then the low-order word of x + redux is 0x000abc12,
391 * We split this into k = 0xabc and i0 = 0x12 (adjusted to
392 * index into the table), then we compute z = 0x0.003456p0.
393 *
394 * XXX If the exponent is negative, the computation of k depends on
395 * '>>' doing sign extension.
396 */
397 u.e = x + redux;
398 i0 = (u.bits.manl & 0xffffffff) + TBLSIZE / 2;
399 k = (int)i0 >> TBLBITS;
400 i0 = i0 & (TBLSIZE - 1);
401 u.e -= redux;
402 z = x - u.e;
403 v.xbits.manh = 0;
404 v.xbits.manl = 0;
405 if (k >= LDBL_MIN_EXP) {
406 v.xbits.expsign = LDBL_MAX_EXP - 1 + k;
407 twopk = v.e;
408 } else {
409 v.xbits.expsign = LDBL_MAX_EXP - 1 + k + 10000;
410 twopkp10000 = v.e;
411 }
412
413 /* Compute r = exp2(y) = exp2t[i0] * p(z - eps[i]). */
414 t = tbl[i0]; /* exp2t[i0] */
415 z -= eps[i0]; /* eps[i0] */
416 r = t + t * z * (P1 + z * (P2 + z * (P3 + z * (P4 + z * (P5 + z * (P6
417 + z * (P7 + z * (P8 + z * (P9 + z * P10)))))))));
418
419 /* Scale by 2**k. */
420 if(k >= LDBL_MIN_EXP) {
421 if (k == LDBL_MAX_EXP)
422 return (r * 2.0 * 0x1p16383L);
423 return (r * twopk);
424 } else {
425 return (r * twopkp10000 * twom10000);
426 }
427}
