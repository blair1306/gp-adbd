19#include "math.h"
20#include "math_private.h"
21
22static const float one = 1.0, huge = 1e30;
23
24static const float zero = 0.0;
25
26float
27__ieee754_atanhf(float x)
28{
29 float t;
30 int32_t hx,ix;
31 GET_FLOAT_WORD(hx,x);
32 ix = hx&0x7fffffff;
33 if (ix>0x3f800000) /* |x|>1 */
34 return (x-x)/(x-x);
35 if(ix==0x3f800000)
36 return x/zero;
37 if(ix<0x31800000&&(huge+x)>zero) return x; /* x<2**-28 */
38 SET_FLOAT_WORD(x,ix);
39 if(ix<0x3f000000) { /* x < 0.5 */
40 t = x+x;
41 t = (float)0.5*log1pf(t+t*x/(one-x));
42 } else
43 t = (float)0.5*log1pf((x+x)/(one-x));
44 if(hx>=0) return t; else return -t;
45}
