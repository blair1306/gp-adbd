16#include <sys/cdefs.h>
17__FBSDID("$FreeBSD$");
18
19#include "math.h"
20#include "math_private.h"
21
22#ifdef _SCALB_INT
23float
24__ieee754_scalbf(float x, int fn)
25#else
26float
27__ieee754_scalbf(float x, float fn)
28#endif
29{
30#ifdef _SCALB_INT
31 return scalbnf(x,fn);
32#else
33 if ((isnanf)(x)||(isnanf)(fn)) return x*fn;
34 if (!finitef(fn)) {
35 if(fn>(float)0.0) return x*fn;
36 else return x/(-fn);
37 }
38 if (rintf(fn)!=fn) return (fn-fn)/(fn-fn);
39 if ( fn > (float)65000.0) return scalbnf(x, 65000);
40 if (-fn > (float)65000.0) return scalbnf(x,-65000);
41 return scalbnf(x,(int)fn);
42#endif
43}
