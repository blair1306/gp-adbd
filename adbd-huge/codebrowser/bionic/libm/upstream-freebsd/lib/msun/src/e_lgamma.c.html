18/* __ieee754_lgamma(x)
19 * Return the logarithm of the Gamma function of x.
20 *
21 * Method: call __ieee754_lgamma_r
22 */
23
24#include <float.h>
25
26#include "math.h"
27#include "math_private.h"
28
29extern int signgam;
30
31double
32__ieee754_lgamma(double x)
33{
34 return __ieee754_lgamma_r(x,&signgam);
35}
