16#include <sys/cdefs.h>
17__FBSDID("$FreeBSD$");
18
19#include "math.h"
20#include "math_private.h"
21
22static const float one = 1.0, shuge = 1.0e37;
23
24float
25__ieee754_sinhf(float x)
26{
27 float t,h;
28 int32_t ix,jx;
29
30 GET_FLOAT_WORD(jx,x);
31 ix = jx&0x7fffffff;
32
33 /* x is INF or NaN */
34 if(ix>=0x7f800000) return x+x;
35
36 h = 0.5;
37 if (jx<0) h = -h;
38 /* |x| in [0,9], return sign(x)*0.5*(E+E/(E+1))) */
39 if (ix < 0x41100000) { /* |x|<9 */
40 if (ix<0x39800000) /* |x|<2**-12 */
41 if(shuge+x>one) return x;/* sinh(tiny) = tiny with inexact */
42 t = expm1f(fabsf(x));
43 if(ix<0x3f800000) return h*((float)2.0*t-t*t/(t+one));
44 return h*(t+t/(t+one));
45 }
46
47 /* |x| in [9, logf(maxfloat)] return 0.5*exp(|x|) */
48 if (ix < 0x42b17217) return h*__ieee754_expf(fabsf(x));
49
50 /* |x| in [logf(maxfloat), overflowthresold] */
51 if (ix<=0x42b2d4fc)
52 return h*2.0F*__ldexp_expf(fabsf(x), -1);
53
54 /* |x| > overflowthresold, sinh(x) overflow */
55 return x*shuge;
56}
