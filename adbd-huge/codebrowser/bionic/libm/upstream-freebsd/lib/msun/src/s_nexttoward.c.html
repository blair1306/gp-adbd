16/*
17 * We assume that a long double has a 15-bit exponent. On systems
18 * where long double is the same as double, nexttoward() is an alias
19 * for nextafter(), so we don't use this routine.
20 */
21
22#include <float.h>
23
24#include "fpmath.h"
25#include "math.h"
26#include "math_private.h"
27
28#if LDBL_MAX_EXP != 0x4000
29#error "Unsupported long double format"
30#endif
31
32double
33nexttoward(double x, long double y)
34{
35 union IEEEl2bits uy;
36 volatile double t;
37 int32_t hx,ix;
38 u_int32_t lx;
39
40 EXTRACT_WORDS(hx,lx,x);
41 ix = hx&0x7fffffff; /* |x| */
42 uy.e = y;
43
44 if(((ix>=0x7ff00000)&&((ix-0x7ff00000)|lx)!=0) ||
45 (uy.bits.exp == 0x7fff &&
46 ((uy.bits.manh&~LDBL_NBIT)|uy.bits.manl) != 0))
47 return x+y; /* x or y is nan */
48 if(x==y) return (double)y; /* x=y, return y */
49 if(x==0.0) {
50 INSERT_WORDS(x,uy.bits.sign<<31,1); /* return +-minsubnormal */
51 t = x*x;
52 if(t==x) return t; else return x; /* raise underflow flag */
53 }
54 if(hx>0.0 ^ x < y) { /* x -= ulp */
55 if(lx==0) hx -= 1;
56 lx -= 1;
57 } else { /* x += ulp */
58 lx += 1;
59 if(lx==0) hx += 1;
60 }
61 ix = hx&0x7ff00000;
62 if(ix>=0x7ff00000) return x+x; /* overflow */
63 if(ix<0x00100000) { /* underflow */
64 t = x*x;
65 if(t!=x) { /* raise underflow flag */
66 INSERT_WORDS(x,hx,lx);
67 return x;
68 }
69 }
70 INSERT_WORDS(x,hx,lx);
71 return x;
72}
