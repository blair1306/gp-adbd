30#include <complex.h>
31#include <math.h>
32
33#include "math_private.h"
34
35float complex
36cprojf(float complex z)
37{
38
39 if (!isinf(crealf(z)) && !isinf(cimagf(z)))
40 return (z);
41 else
42 return (CMPLXF(INFINITY, copysignf(0.0, cimagf(z))));
43}
