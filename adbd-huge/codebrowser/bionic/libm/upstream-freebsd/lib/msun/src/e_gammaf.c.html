19/* __ieee754_gammaf(x)
20 * Return the logarithm of the Gamma function of x.
21 *
22 * Method: call __ieee754_gammaf_r
23 */
24
25#include "math.h"
26#include "math_private.h"
27
28extern int signgam;
29
30float
31__ieee754_gammaf(float x)
32{
33 return __ieee754_gammaf_r(x,&signgam);
34}
