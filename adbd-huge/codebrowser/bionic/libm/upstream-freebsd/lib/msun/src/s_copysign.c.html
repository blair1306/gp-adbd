16/*
17 * copysign(double x, double y)
18 * copysign(x,y) returns a value with the magnitude of x and
19 * with the sign bit of y.
20 */
21
22#include "math.h"
23#include "math_private.h"
24
25double
26copysign(double x, double y)
27{
28 u_int32_t hx,hy;
29 GET_HIGH_WORD(hx,x);
30 GET_HIGH_WORD(hy,y);
31 SET_HIGH_WORD(x,(hx&0x7fffffff)|(hy&0x80000000));
32 return x;
33}
