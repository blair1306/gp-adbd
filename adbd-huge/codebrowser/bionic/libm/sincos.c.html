27#define _GNU_SOURCE 1
28#include <math.h>
29
30// Disable sincos optimization for all functions in this file,
31// otherwise gcc would generate infinite calls.
32// Refer to gcc PR46926.
33// -fno-builtin-sin or -fno-builtin-cos can disable sincos optimization,
34// but these two options do not work inside optimize pragma in-file.
35// Thus we just enforce -O0 when compiling this file.
36#pragma GCC optimize ("O0")
37
38void sincos(double x, double* p_sin, double* p_cos) {
39 *p_sin = sin(x);
40 *p_cos = cos(x);
41}
42
43void sincosf(float x, float* p_sinf, float* p_cosf) {
44 *p_sinf = sinf(x);
45 *p_cosf = cosf(x);
46}
47
48void sincosl(long double x, long double* p_sinl, long double* p_cosl) {
49 *p_sinl = sinl(x);
50 *p_cosl = cosl(x);
51}
