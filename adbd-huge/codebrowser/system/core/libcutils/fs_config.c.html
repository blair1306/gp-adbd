44/* The following structure is stored little endian */
45struct fs_path_config_from_file {
46 uint16_t len;
47 uint16_t mode;
48 uint16_t uid;
49 uint16_t gid;
50 uint64_t capabilities;
51 char prefix[];
52} __attribute__((__aligned__(sizeof(uint64_t))));
53
54/* My kingdom for <endian.h> */
55static inline uint16_t get2LE(const uint8_t* src)
56{
57 return src[0] | (src[1] << 8);
58}
59
60static inline uint64_t get8LE(const uint8_t* src)
61{
62 uint32_t low, high;
63
64 low = src[0] | (src[1] << 8) | (src[2] << 16) | (src[3] << 24);
65 high = src[4] | (src[5] << 8) | (src[6] << 16) | (src[7] << 24);
66 return ((uint64_t) high << 32) | (uint64_t) low;
67}

69#define ALIGN(x, alignment) ( ((x) + ((alignment) - 1)) & ~((alignment) - 1) )

160static int fs_config_open(int dir, const char *target_out_path)
161{
162 int fd = -1;
163
164 if (target_out_path && *target_out_path) {
165 /* target_out_path is the path to the directory holding content of system partition
166 but as we cannot guaranty it ends with '/system' we need this below skip_len logic */
167 char *name = NULL;
168 int target_out_path_len = strlen(target_out_path);
169 int skip_len = strlen("/system");
170
171 if (target_out_path[target_out_path_len] == '/') {
172 skip_len++;
173 }
174 if (asprintf(&name, "%s%s", target_out_path, (dir ? conf_dir : conf_file) + skip_len) != -1) {
175 fd = TEMP_FAILURE_RETRY(open(name, O_RDONLY | O_BINARY));
176 free(name);
177 }
178 }
179 if (fd < 0) {
180 fd = TEMP_FAILURE_RETRY(open(dir ? conf_dir : conf_file, O_RDONLY | O_BINARY));
181 }
182 return fd;
183}

185static bool fs_config_cmp(bool dir, const char *prefix, size_t len,
186 const char *path, size_t plen)
187{
188 if (dir) {
189 if (plen < len) {
190 return false;
191 }
192 } else {
193 /* If name ends in * then allow partial matches. */
194 if (prefix[len - 1] == '*') {
195 return !strncmp(prefix, path, len - 1);
196 }
197 if (plen != len) {
198 return false;
199 }
200 }
201 return !strncmp(prefix, path, len);
202}

204void fs_config(const char *path, int dir, const char *target_out_path,
205 unsigned *uid, unsigned *gid, unsigned *mode, uint64_t *capabilities)
206{
207 const struct fs_path_config *pc;
208 int fd, plen;
209
210 if (path[0] == '/') {
211 path++;
212 }
213
214 plen = strlen(path);
215
216 fd = fs_config_open(dir, target_out_path);
217 if (fd >= 0) {
218 struct fs_path_config_from_file header;
219
220 while (TEMP_FAILURE_RETRY(read(fd, &header, sizeof(header))) == sizeof(header)) {
221 char *prefix;
222 uint16_t host_len = get2LE((const uint8_t *)&header.len);
223 ssize_t len, remainder = host_len - sizeof(header);
224 if (remainder <= 0) {
225 ALOGE("%s len is corrupted", dir ? conf_dir : conf_file);
226 break;
227 }
228 prefix = calloc(1, remainder);
229 if (!prefix) {
230 ALOGE("%s out of memory", dir ? conf_dir : conf_file);
231 break;
232 }
233 if (TEMP_FAILURE_RETRY(read(fd, prefix, remainder)) != remainder) {
234 free(prefix);
235 ALOGE("%s prefix is truncated", dir ? conf_dir : conf_file);
236 break;
237 }
238 len = strnlen(prefix, remainder);
239 if (len >= remainder) { /* missing a terminating null */
240 free(prefix);
241 ALOGE("%s is corrupted", dir ? conf_dir : conf_file);
242 break;
243 }
244 if (fs_config_cmp(dir, prefix, len, path, plen)) {
245 free(prefix);
246 close(fd);
247 *uid = get2LE((const uint8_t *)&(header.uid));
248 *gid = get2LE((const uint8_t *)&(header.gid));
249 *mode = (*mode & (~07777)) | get2LE((const uint8_t *)&(header.mode));
250 *capabilities = get8LE((const uint8_t *)&(header.capabilities));
251 return;
252 }
253 free(prefix);
254 }
255 close(fd);
256 }
257
258 pc = dir ? android_dirs : android_files;
259 for(; pc->prefix; pc++){
260 if (fs_config_cmp(dir, pc->prefix, strlen(pc->prefix), path, plen)) {
261 break;
262 }
263 }
264 *uid = pc->uid;
265 *gid = pc->gid;
266 *mode = (*mode & (~07777)) | pc->mode;
267 *capabilities = pc->capabilities;
268}

270ssize_t fs_config_generate(char *buffer, size_t length, const struct fs_path_config *pc)
271{
272 struct fs_path_config_from_file *p = (struct fs_path_config_from_file *)buffer;
273 size_t len = ALIGN(sizeof(*p) + strlen(pc->prefix) + 1, sizeof(uint64_t));
274
275 if ((length < len) || (len > UINT16_MAX)) {
276 return -ENOSPC;
277 }
278 memset(p, 0, len);
279 uint16_t host_len = len;
280 p->len = get2LE((const uint8_t *)&host_len);
281 p->mode = get2LE((const uint8_t *)&(pc->mode));
282 p->uid = get2LE((const uint8_t *)&(pc->uid));
283 p->gid = get2LE((const uint8_t *)&(pc->gid));
284 p->capabilities = get8LE((const uint8_t *)&(pc->capabilities));
285 strcpy(p->prefix, pc->prefix);
286 return len;
287}
