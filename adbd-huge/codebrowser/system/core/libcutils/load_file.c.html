18#include <stdlib.h>
19#include <unistd.h>
20#include <fcntl.h>
21
22void *load_file(const char *fn, unsigned *_sz)
23{
24 char *data;
25 int sz;
26 int fd;
27
28 data = 0;
29 fd = open(fn, O_RDONLY);
30 if(fd < 0) return 0;
31
32 sz = lseek(fd, 0, SEEK_END);
33 if(sz < 0) goto oops;
34
35 if(lseek(fd, 0, SEEK_SET) != 0) goto oops;
36
37 data = (char*) malloc(sz + 1);
38 if(data == 0) goto oops;
39
40 if(read(fd, data, sz) != sz) goto oops;
41 close(fd);
42 data[sz] = 0;
43
44 if(_sz) *_sz = sz;
45 return data;
46
47oops:
48 close(fd);
49 if(data != 0) free(data);
50 return 0;
51}
