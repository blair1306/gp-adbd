42static void local_socket_close_locked(asocket *s);

44static unsigned local_socket_next_id = 1;
45
46static asocket local_socket_list = {
47 .next = &local_socket_list,
48 .prev = &local_socket_list,
49};
50
51/* the the list of currently closing local sockets.
52** these have no peer anymore, but still packets to
53** write to their fd.
54*/
55static asocket local_socket_closing_list = {
56 .next = &local_socket_closing_list,
57 .prev = &local_socket_closing_list,
58};
59
60// Parse the global list of sockets to find one with id |local_id|.
61// If |peer_id| is not 0, also check that it is connected to a peer
62// with id |peer_id|. Returns an asocket handle on success, NULL on failure.
63asocket *find_local_socket(unsigned local_id, unsigned peer_id)
64{
65 asocket *s;
66 asocket *result = NULL;
67
68 adb_mutex_lock(&socket_list_lock);
69 for (s = local_socket_list.next; s != &local_socket_list; s = s->next) {
70 if (s->id != local_id)
71 continue;
72 if (peer_id == 0 || (s->peer && s->peer->id == peer_id)) {
73 result = s;
74 }
75 break;
76 }
77 adb_mutex_unlock(&socket_list_lock);
78
79 return result;
80}

82static void
83insert_local_socket(asocket* s, asocket* list)
84{
85 s->next = list;
86 s->prev = s->next->prev;
87 s->prev->next = s;
88 s->next->prev = s;
89}

92void install_local_socket(asocket *s)
93{
94 adb_mutex_lock(&socket_list_lock);
95
96 s->id = local_socket_next_id++;
97
98 // Socket ids should never be 0.
99 if (local_socket_next_id == 0)
100 local_socket_next_id = 1;
101
102 insert_local_socket(s, &local_socket_list);
103
104 adb_mutex_unlock(&socket_list_lock);
105}
106
107void remove_socket(asocket *s)
108{
109 // socket_list_lock should already be held
110 if (s->prev && s->next)
111 {
112 s->prev->next = s->next;
113 s->next->prev = s->prev;
114 s->next = 0;
115 s->prev = 0;
116 s->id = 0;
117 }
118}

120void close_all_sockets(atransport *t)
121{
122 asocket *s;
123
124 /* this is a little gross, but since s->close() *will* modify
125 ** the list out from under you, your options are limited.
126 */
127 adb_mutex_lock(&socket_list_lock);
128restart:
129 for(s = local_socket_list.next; s != &local_socket_list; s = s->next){
130 if(s->transport == t || (s->peer && s->peer->transport == t)) {
131 local_socket_close_locked(s);
132 goto restart;
133 }
134 }
135 adb_mutex_unlock(&socket_list_lock);
136}

138static int local_socket_enqueue(asocket *s, apacket *p)
139{
140 D("LS(%d): enqueue %d", s->id, p->len);
141
142 p->ptr = p->data;
143
144 /* if there is already data queue'd, we will receive
145 ** events when it's time to write. just add this to
146 ** the tail
147 */
148 if(s->pkt_first) {
149 goto enqueue;
150 }
151
152 /* write as much as we can, until we
153 ** would block or there is an error/eof
154 */
155 while(p->len > 0) {
156 int r = adb_write(s->fd, p->ptr, p->len);
157 if(r > 0) {
158 p->len -= r;
159 p->ptr += r;
160 continue;
161 }
162 if((r == 0) || (errno != EAGAIN)) {
163 D( "LS(%d): not ready, errno=%d: %s", s->id, errno, strerror(errno) );
164 put_apacket(p);
165 s->has_write_error = true;
166 s->close(s);
167 return 1; /* not ready (error) */
168 } else {
169 break;
170 }
171 }
172
173 if(p->len == 0) {
174 put_apacket(p);
175 return 0; /* ready for more data */
176 }
177
178enqueue:
179 p->next = 0;
180 if(s->pkt_first) {
181 s->pkt_last->next = p;
182 } else {
183 s->pkt_first = p;
184 }
185 s->pkt_last = p;
186
187 /* make sure we are notified when we can drain the queue */
188 fdevent_add(&s->fde, FDE_WRITE);
189
190 return 1; /* not ready (backlog) */
191}

193static void local_socket_ready(asocket *s)
194{
195 /* far side is ready for data, pay attention to
196 readable events */
197 fdevent_add(&s->fde, FDE_READ);
198}

200static void local_socket_close(asocket *s)
201{
202 adb_mutex_lock(&socket_list_lock);
203 local_socket_close_locked(s);
204 adb_mutex_unlock(&socket_list_lock);
205}

207// be sure to hold the socket list lock when calling this
208static void local_socket_destroy(asocket *s)
209{
210 apacket *p, *n;
211 int exit_on_close = s->exit_on_close;
212
213 D("LS(%d): destroying fde.fd=%d", s->id, s->fde.fd);
214
215 /* IMPORTANT: the remove closes the fd
216 ** that belongs to this socket
217 */
218 fdevent_remove(&s->fde);
219
220 /* dispose of any unwritten data */
221 for(p = s->pkt_first; p; p = n) {
222 D("LS(%d): discarding %d bytes", s->id, p->len);
223 n = p->next;
224 put_apacket(p);
225 }
226 remove_socket(s);
227 free(s);
228
229 if (exit_on_close) {
230 D("local_socket_destroy: exiting");
231 exit(1);
232 }
233}

236static void local_socket_close_locked(asocket *s)
237{
238 D("entered local_socket_close_locked. LS(%d) fd=%d", s->id, s->fd);
239 if(s->peer) {
240 D("LS(%d): closing peer. peer->id=%d peer->fd=%d",
241 s->id, s->peer->id, s->peer->fd);
242 /* Note: it's important to call shutdown before disconnecting from
243 * the peer, this ensures that remote sockets can still get the id
244 * of the local socket they're connected to, to send a CLOSE()
245 * protocol event. */
246 if (s->peer->shutdown)
247 s->peer->shutdown(s->peer);
248 s->peer->peer = 0;
249 // tweak to avoid deadlock
250 if (s->peer->close == local_socket_close) {
251 local_socket_close_locked(s->peer);
252 } else {
253 s->peer->close(s->peer);
254 }
255 s->peer = 0;
256 }
257
258 /* If we are already closing, or if there are no
259 ** pending packets, destroy immediately
260 */
261 if (s->closing || s->has_write_error || s->pkt_first == NULL) {
262 int id = s->id;
263 local_socket_destroy(s);
264 D("LS(%d): closed", id);
265 return;
266 }
267
268 /* otherwise, put on the closing list
269 */
270 D("LS(%d): closing", s->id);
271 s->closing = 1;
272 fdevent_del(&s->fde, FDE_READ);
273 remove_socket(s);
274 D("LS(%d): put on socket_closing_list fd=%d", s->id, s->fd);
275 insert_local_socket(s, &local_socket_closing_list);
276 CHECK_EQ(FDE_WRITE, s->fde.state & FDE_WRITE);
277}

279static void local_socket_event_func(int fd, unsigned ev, void* _s)
280{
281 asocket* s = reinterpret_cast<asocket*>(_s);
282 D("LS(%d): event_func(fd=%d(==%d), ev=%04x)", s->id, s->fd, fd, ev);
283
284 /* put the FDE_WRITE processing before the FDE_READ
285 ** in order to simplify the code.
286 */
287 if (ev & FDE_WRITE) {
288 apacket* p;
289 while ((p = s->pkt_first) != nullptr) {
290 while (p->len > 0) {
291 int r = adb_write(fd, p->ptr, p->len);
292 if (r == -1) {
293 /* returning here is ok because FDE_READ will
294 ** be processed in the next iteration loop
295 */
296 if (errno == EAGAIN) {
297 return;
298 }
299 } else if (r > 0) {
300 p->ptr += r;
301 p->len -= r;
302 continue;
303 }
304
305 D(" closing after write because r=%d and errno is %d", r, errno);
306 s->has_write_error = true;
307 s->close(s);
308 return;
309 }
310
311 if (p->len == 0) {
312 s->pkt_first = p->next;
313 if (s->pkt_first == 0) {
314 s->pkt_last = 0;
315 }
316 put_apacket(p);
317 }
318 }
319
320 /* if we sent the last packet of a closing socket,
321 ** we can now destroy it.
322 */
323 if (s->closing) {
324 D(" closing because 'closing' is set after write");
325 s->close(s);
326 return;
327 }
328
329 /* no more packets queued, so we can ignore
330 ** writable events again and tell our peer
331 ** to resume writing
332 */
333 fdevent_del(&s->fde, FDE_WRITE);
334 s->peer->ready(s->peer);
335 }
336
337
338 if (ev & FDE_READ) {
339 apacket *p = get_apacket();
340 unsigned char *x = p->data;
341 const size_t max_payload = s->get_max_payload();
342 size_t avail = max_payload;
343 int r = 0;
344 int is_eof = 0;
345
346 while (avail > 0) {
347 r = adb_read(fd, x, avail);
348 D("LS(%d): post adb_read(fd=%d,...) r=%d (errno=%d) avail=%zu",
349 s->id, s->fd, r, r < 0 ? errno : 0, avail);
350 if (r == -1) {
351 if (errno == EAGAIN) {
352 break;
353 }
354 } else if (r > 0) {
355 avail -= r;
356 x += r;
357 continue;
358 }
359
360 /* r = 0 or unhandled error */
361 is_eof = 1;
362 break;
363 }
364 D("LS(%d): fd=%d post avail loop. r=%d is_eof=%d forced_eof=%d",
365 s->id, s->fd, r, is_eof, s->fde.force_eof);
366 if ((avail == max_payload) || (s->peer == 0)) {
367 put_apacket(p);
368 } else {
369 p->len = max_payload - avail;
370
371 // s->peer->enqueue() may call s->close() and free s,
372 // so save variables for debug printing below.
373 unsigned saved_id = s->id;
374 int saved_fd = s->fd;
375 r = s->peer->enqueue(s->peer, p);
376 D("LS(%u): fd=%d post peer->enqueue(). r=%d", saved_id, saved_fd, r);
377
378 if (r < 0) {
379 /* error return means they closed us as a side-effect
380 ** and we must return immediately.
381 **
382 ** note that if we still have buffered packets, the
383 ** socket will be placed on the closing socket list.
384 ** this handler function will be called again
385 ** to process FDE_WRITE events.
386 */
387 return;
388 }
389
390 if (r > 0) {
391 /* if the remote cannot accept further events,
392 ** we disable notification of READs. They'll
393 ** be enabled again when we get a call to ready()
394 */
395 fdevent_del(&s->fde, FDE_READ);
396 }
397 }
398 /* Don't allow a forced eof if data is still there */
399 if ((s->fde.force_eof && !r) || is_eof) {
400 D(" closing because is_eof=%d r=%d s->fde.force_eof=%d",
401 is_eof, r, s->fde.force_eof);
402 s->close(s);
403 return;
404 }
405 }
406
407 if (ev & FDE_ERROR){
408 /* this should be caught be the next read or write
409 ** catching it here means we may skip the last few
410 ** bytes of readable data.
411 */
412 D("LS(%d): FDE_ERROR (fd=%d)", s->id, s->fd);
413 return;
414 }
415}

417asocket *create_local_socket(int fd)
418{
419 asocket *s = reinterpret_cast<asocket*>(calloc(1, sizeof(asocket)));
420 if (s == NULL) fatal("cannot allocate socket");
421 s->fd = fd;
422 s->enqueue = local_socket_enqueue;
423 s->ready = local_socket_ready;
424 s->shutdown = NULL;
425 s->close = local_socket_close;
426 install_local_socket(s);
427
428 fdevent_install(&s->fde, fd, local_socket_event_func, s);
429 D("LS(%d): created (fd=%d)", s->id, s->fd);
430 return s;
431}

433asocket *create_local_service_socket(const char *name,
434 const atransport* transport)
435{
436#if !ADB_HOST
437 if (!strcmp(name,"jdwp")) {
438 return create_jdwp_service_socket();

441 return create_jdwp_tracker_service_socket();

448 D("LS(%d): bound to '%s' via %d", s->id, name, fd);
449
450#if !ADB_HOST
451 char debug[PROPERTY_VALUE_MAX];
452 if (!strncmp(name, "root:", 5))
453 property_get("ro.debuggable", debug, "");
454
455 if ((!strncmp(name, "root:", 5) && getuid() != 0 && strcmp(debug, "1") == 0)
456 || (!strncmp(name, "unroot:", 7) && getuid() == 0)
457 || !strncmp(name, "usb:", 4)
458 || !strncmp(name, "tcpip:", 6)) {
459 D("LS(%d): enabling exit_on_close", s->id);
460 s->exit_on_close = 1;
461 }
462#endif
463
464 return s;
465}

483static int remote_socket_enqueue(asocket *s, apacket *p)
484{
485 D("entered remote_socket_enqueue RS(%d) WRITE fd=%d peer.fd=%d",
486 s->id, s->fd, s->peer->fd);
487 p->msg.command = A_WRTE;
488 p->msg.arg0 = s->peer->id;
489 p->msg.arg1 = s->id;
490 p->msg.data_length = p->len;
491 send_packet(p, s->transport);
492 return 1;
493}

506static void remote_socket_shutdown(asocket *s)
507{
508 D("entered remote_socket_shutdown RS(%d) CLOSE fd=%d peer->fd=%d",
509 s->id, s->fd, s->peer?s->peer->fd:-1);
510 apacket *p = get_apacket();
511 p->msg.command = A_CLSE;
512 if(s->peer) {
513 p->msg.arg0 = s->peer->id;
514 }
515 p->msg.arg1 = s->id;
516 send_packet(p, s->transport);
517}

519static void remote_socket_close(asocket *s)
520{
521 if (s->peer) {
522 s->peer->peer = 0;
523 D("RS(%d) peer->close()ing peer->id=%d peer->fd=%d",
524 s->id, s->peer->id, s->peer->fd);
525 s->peer->close(s->peer);
526 }
527 D("entered remote_socket_close RS(%d) CLOSE fd=%d peer->fd=%d",
528 s->id, s->fd, s->peer?s->peer->fd:-1);
529 D("RS(%d): closed", s->id);
530 free(s);
531}

533// Create a remote socket to exchange packets with a remote service through transport
534// |t|. Where |id| is the socket id of the corresponding service on the other
535// side of the transport (it is allocated by the remote side and _cannot_ be 0).
536// Returns a new non-NULL asocket handle.
537asocket *create_remote_socket(unsigned id, atransport *t)
538{
539 if (id == 0) fatal("invalid remote socket id (0)");
540 asocket* s = reinterpret_cast<asocket*>(calloc(1, sizeof(asocket)));
541
542 if (s == NULL) fatal("cannot allocate socket");
543 s->id = id;
544 s->enqueue = remote_socket_enqueue;
545 s->ready = remote_socket_ready;
546 s->shutdown = remote_socket_shutdown;
547 s->close = remote_socket_close;
548 s->transport = t;
549
550 D("RS(%d): created", s->id);
551 return s;
552}

554void connect_to_remote(asocket *s, const char *destination)
555{
556 D("Connect_to_remote call RS(%d) fd=%d", s->id, s->fd);
557 apacket *p = get_apacket();
558 size_t len = strlen(destination) + 1;
559
560 if(len > (s->get_max_payload()-1)) {
561 fatal("destination oversized");
562 }
563
564 D("LS(%d): connect('%s')", s->id, destination);
565 p->msg.command = A_OPEN;
566 p->msg.arg0 = s->id;
567 p->msg.data_length = len;
568 strcpy((char*) p->data, destination);
569 send_packet(p, s->transport);
570}

596static unsigned unhex(unsigned char *s, int len)
597{
598 unsigned n = 0, c;
599
600 while(len-- > 0) {
601 switch((c = *s++)) {
602 case '0': case '1': case '2':
603 case '3': case '4': case '5':
604 case '6': case '7': case '8':
605 case '9':
606 c -= '0';
607 break;
608 case 'a': case 'b': case 'c':
609 case 'd': case 'e': case 'f':
610 c = c - 'a' + 10;
611 break;
612 case 'A': case 'B': case 'C':
613 case 'D': case 'E': case 'F':
614 c = c - 'A' + 10;
615 break;
616 default:
617 return 0xffffffff;
618 }
619
620 n = (n << 4) | c;
621 }
622
623 return n;
624}

630// Returns the position in |service| following the target serial parameter. Serial format can be
631// any of:
632// * [tcp:|udp:]<serial>[:<port>]:<command>
633// * <prefix>:<serial>:<command>
634// Where <port> must be a base-10 number and <prefix> may be any of {usb,product,model,device}.
635//
636// The returned pointer will point to the ':' just before <command>, or nullptr if not found.
637char* skip_host_serial(char* service) {
638 static const std::vector<std::string>& prefixes =
639 *(new std::vector<std::string>{"usb:", "product:", "model:", "device:"});
640
641 for (const std::string& prefix : prefixes) {
642 if (!strncmp(service, prefix.c_str(), prefix.length())) {
643 return strchr(service + prefix.length(), ':');
644 }
645 }
646
647 // For fastboot compatibility, ignore protocol prefixes.
648 if (!strncmp(service, "tcp:", 4) || !strncmp(service, "udp:", 4)) {
649 service += 4;
650 }
651
652 char* first_colon = strchr(service, ':');
653 if (!first_colon) {
654 // No colon in service string.
655 return nullptr;
656 }
657
658 char* serial_end = first_colon;
659 if (isdigit(serial_end[1])) {
660 serial_end++;
661 while (*serial_end && isdigit(*serial_end)) {
662 serial_end++;
663 }
664 if (*serial_end != ':') {
665 // Something other than numbers was found, reset the end.
666 serial_end = first_colon;
667 }
668 }
669 return serial_end;
670}

676static int smart_socket_enqueue(asocket *s, apacket *p)
677{
678 unsigned len;
679#if ADB_HOST
680 char *service = nullptr;
681 char* serial = nullptr;
682 TransportType type = kTransportAny;
683#endif
684
685 D("SS(%d): enqueue %d", s->id, p->len);
686
687 if(s->pkt_first == 0) {
688 s->pkt_first = p;
689 s->pkt_last = p;
690 } else {
691 if((s->pkt_first->len + p->len) > s->get_max_payload()) {
692 D("SS(%d): overflow", s->id);
693 put_apacket(p);
694 goto fail;
695 }
696
697 memcpy(s->pkt_first->data + s->pkt_first->len,
698 p->data, p->len);
699 s->pkt_first->len += p->len;
700 put_apacket(p);
701
702 p = s->pkt_first;
703 }
704
705 /* don't bother if we can't decode the length */
706 if(p->len < 4) return 0;
707
708 len = unhex(p->data, 4);
709 if ((len < 1) || (len > MAX_PAYLOAD_V1)) {
710 D("SS(%d): bad size (%d)", s->id, len);
711 goto fail;
712 }
713
714 D("SS(%d): len is %d", s->id, len );
715 /* can't do anything until we have the full header */
716 if((len + 4) > p->len) {
717 D("SS(%d): waiting for %d more bytes", s->id, len+4 - p->len);
718 return 0;
719 }
720
721 p->data[len + 4] = 0;
722
723 D("SS(%d): '%s'", s->id, (char*) (p->data + 4));
724
725#if ADB_HOST
726 service = (char *)p->data + 4;
727 if(!strncmp(service, "host-serial:", strlen("host-serial:"))) {
728 char* serial_end;
729 service += strlen("host-serial:");
730
731 // serial number should follow "host:" and could be a host:port string.
732 serial_end = internal::skip_host_serial(service);
733 if (serial_end) {
734 *serial_end = 0; // terminate string
735 serial = service;
736 service = serial_end + 1;
737 }
738 } else if (!strncmp(service, "host-usb:", strlen("host-usb:"))) {
739 type = kTransportUsb;
740 service += strlen("host-usb:");
741 } else if (!strncmp(service, "host-local:", strlen("host-local:"))) {
742 type = kTransportLocal;
743 service += strlen("host-local:");
744 } else if (!strncmp(service, "host:", strlen("host:"))) {
745 type = kTransportAny;
746 service += strlen("host:");
747 } else {
748 service = nullptr;
749 }
750
751 if (service) {
752 asocket *s2;
753
754 /* some requests are handled immediately -- in that
755 ** case the handle_host_request() routine has sent
756 ** the OKAY or FAIL message and all we have to do
757 ** is clean up.
758 */
759 if(handle_host_request(service, type, serial, s->peer->fd, s) == 0) {
760 /* XXX fail message? */
761 D( "SS(%d): handled host service '%s'", s->id, service );
762 goto fail;
763 }
764 if (!strncmp(service, "transport", strlen("transport"))) {
765 D( "SS(%d): okay transport", s->id );
766 p->len = 0;
767 return 0;
768 }
769
770 /* try to find a local service with this name.
771 ** if no such service exists, we'll fail out
772 ** and tear down here.
773 */
774 s2 = create_host_service_socket(service, serial);
775 if(s2 == 0) {
776 D( "SS(%d): couldn't create host service '%s'", s->id, service );
777 SendFail(s->peer->fd, "unknown host service");
778 goto fail;
779 }
780
781 /* we've connected to a local host service,
782 ** so we make our peer back into a regular
783 ** local socket and bind it to the new local
784 ** service socket, acknowledge the successful
785 ** connection, and close this smart socket now
786 ** that its work is done.
787 */
788 SendOkay(s->peer->fd);
789
790 s->peer->ready = local_socket_ready;
791 s->peer->shutdown = nullptr;
792 s->peer->close = local_socket_close;
793 s->peer->peer = s2;
794 s2->peer = s->peer;
795 s->peer = 0;
796 D( "SS(%d): okay", s->id );
797 s->close(s);
798
799 /* initial state is "ready" */
800 s2->ready(s2);
801 return 0;
802 }
803#else /* !ADB_HOST */
804 if (s->transport == nullptr) {
805 std::string error_msg = "unknown failure";
806 s->transport = acquire_one_transport(kTransportAny, nullptr, nullptr, &error_msg);
807 if (s->transport == nullptr) {
808 SendFail(s->peer->fd, error_msg);
809 goto fail;
810 }
811 }
812#endif
813
814 if(!(s->transport) || (s->transport->connection_state == kCsOffline)) {
815 /* if there's no remote we fail the connection
816 ** right here and terminate it
817 */
818 SendFail(s->peer->fd, "device offline (x)");
819 goto fail;
820 }
821
822
823 /* instrument our peer to pass the success or fail
824 ** message back once it connects or closes, then
825 ** detach from it, request the connection, and
826 ** tear down
827 */
828 s->peer->ready = local_socket_ready_notify;
829 s->peer->shutdown = nullptr;
830 s->peer->close = local_socket_close_notify;
831 s->peer->peer = 0;
832 /* give him our transport and upref it */
833 s->peer->transport = s->transport;
834
835 connect_to_remote(s->peer, (char*) (p->data + 4));
836 s->peer = 0;
837 s->close(s);
838 return 1;
839
840fail:
841 /* we're going to close our peer as a side-effect, so
842 ** return -1 to signal that state to the local socket
843 ** who is enqueueing against us
844 */
845 s->close(s);
846 return -1;
847}

854static void smart_socket_close(asocket *s)
855{
856 D("SS(%d): closed", s->id);
857 if(s->pkt_first){
858 put_apacket(s->pkt_first);
859 }
860 if(s->peer) {
861 s->peer->peer = 0;
862 s->peer->close(s->peer);
863 s->peer = 0;
864 }
865 free(s);

868static asocket *create_smart_socket(void)
869{
870 D("Creating smart socket");
871 asocket *s = reinterpret_cast<asocket*>(calloc(1, sizeof(asocket)));
872 if (s == NULL) fatal("cannot allocate socket");
873 s->enqueue = smart_socket_enqueue;
874 s->ready = smart_socket_ready;
875 s->shutdown = NULL;
876 s->close = smart_socket_close;
877
878 D("SS(%d)", s->id);
879 return s;
880}
