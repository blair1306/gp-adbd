28typedef std::unordered_set<std::string> FeatureSet;
29
30const FeatureSet& supported_features();
31
32// Encodes and decodes FeatureSet objects into human-readable strings.
33std::string FeatureSetToString(const FeatureSet& features);
34FeatureSet StringToFeatureSet(const std::string& features_string);
35
36// Returns true if both local features and |feature_set| support |feature|.
37bool CanUseFeature(const FeatureSet& feature_set, const std::string& feature);
38
39// Do not use any of [:;=,] in feature strings, they have special meaning
40// in the connection banner.
41extern const char* const kFeatureShell2;
42// The 'cmd' command is available
43extern const char* const kFeatureCmd;
44
45class atransport {
46public:
47 // TODO(danalbert): We expose waaaaaaay too much stuff because this was
48 // historically just a struct, but making the whole thing a more idiomatic
49 // class in one go is a very large change. Given how bad our testing is,
50 // it's better to do this piece by piece.
51
52 atransport() {
53 transport_fde = {};
54 protocol_version = A_VERSION;
55 max_payload = MAX_PAYLOAD;
56 }
57
58 virtual ~atransport() {}
59
60 int (*read_from_remote)(apacket* p, atransport* t) = nullptr;
61 int (*write_to_remote)(apacket* p, atransport* t) = nullptr;
62 void (*close)(atransport* t) = nullptr;
63 void (*kick)(atransport* t) = nullptr;
64
65 int fd = -1;
66 int transport_socket = -1;
67 fdevent transport_fde;
68 size_t ref_count = 0;
69 uint32_t sync_token = 0;
70 ConnectionState connection_state = kCsOffline;
71 bool online = false;
72 TransportType type = kTransportAny;
73
74 // USB handle or socket fd as needed.
75 usb_handle* usb = nullptr;
76 int sfd = -1;
77
78 // Used to identify transports for clients.
79 char* serial = nullptr;
80 char* product = nullptr;
81 char* model = nullptr;
82 char* device = nullptr;
83 char* devpath = nullptr;
84 int adb_port = -1; // Use for emulators (local transport)
85 bool kicked = false;
86
87 void* key = nullptr;
88 unsigned char token[TOKEN_SIZE] = {};
89 size_t failed_auth_attempts = 0;
90
91 const std::string connection_state_name() const;
92
93 void update_version(int version, size_t payload);
94 int get_protocol_version() const;
95 size_t get_max_payload() const;
96
97 const FeatureSet& features() const {
98 return features_;
99 }
100
101 bool has_feature(const std::string& feature) const;
102
103 // Loads the transport's feature set from the given string.
104 void SetFeatures(const std::string& features_string);
105
106 void AddDisconnect(adisconnect* disconnect);
107 void RemoveDisconnect(adisconnect* disconnect);
108 void RunDisconnects();
109
110 // Returns true if |target| matches this transport. A matching |target| can be any of:
111 // * <serial>
112 // * <devpath>
113 // * product:<product>
114 // * model:<model>
115 // * device:<device>
116 //
117 // If this is a local transport, serial will also match [tcp:|udp:]<hostname>[:port] targets.
118 // For example, serial "100.100.100.100:5555" would match any of:
119 // * 100.100.100.100
120 // * tcp:100.100.100.100
121 // * udp:100.100.100.100:5555
122 // This is to make it easier to use the same network target for both fastboot and adb.
123 bool MatchesTarget(const std::string& target) const;
124
125private:
126 // A set of features transmitted in the banner with the initial connection.
127 // This is stored in the banner as 'features=feature0,feature1,etc'.
128 FeatureSet features_;
129 int protocol_version;
130 size_t max_payload;
131
132 // A list of adisconnect callbacks called when the transport is kicked.
133 std::list<adisconnect*> disconnects_;
134
135 DISALLOW_COPY_AND_ASSIGN(atransport);
136};

138/*
139 * Obtain a transport from the available transports.
140 * If serial is non-null then only the device with that serial will be chosen.
141 * If multiple devices/emulators would match, *is_ambiguous (if non-null)
142 * is set to true and nullptr returned.
143 * If no suitable transport is found, error is set and nullptr returned.
144 */
145atransport* acquire_one_transport(TransportType type, const char* serial,
146 bool* is_ambiguous, std::string* error_out);
147void kick_transport(atransport* t);
148void update_transports(void);
149
150void init_transport_registration(void);
151std::string list_transports(bool long_listing);
152atransport* find_transport(const char* serial);
153void kick_all_tcp_devices();
154
155void register_usb_transport(usb_handle* h, const char* serial,
156 const char* devpath, unsigned writeable);
157
158/* cause new transports to be init'd and added to the list */
159int register_socket_transport(int s, const char* serial, int port, int local);
160
161// This should only be used for transports with connection_state == kCsNoPerm.
162void unregister_usb_transport(usb_handle* usb);
163
164int check_header(apacket* p, atransport* t);
165int check_data(apacket* p);
166
167/* for MacOS X cleanup */
168void close_usb_devices();
169
170void send_packet(apacket* p, atransport* t);

172asocket* create_device_tracker(void);
