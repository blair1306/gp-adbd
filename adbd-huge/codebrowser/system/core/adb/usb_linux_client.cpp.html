42// Writes larger than 16k fail on some devices (seed with 3.10.49-g209ea2f in particular).
43#define USB_FFS_MAX_WRITE 16384
44
45// The kernel allocates a contiguous buffer for reads, which can fail for large ones due to
46// fragmentation. 16k chosen arbitrarily to match the write limit.
47#define USB_FFS_MAX_READ 16384

235static void usb_adb_open_thread(void* x) {
236 struct usb_handle *usb = (struct usb_handle *)x;
237 int fd;
238
239 adb_thread_setname("usb open");
240
241 while (true) {
242 // wait until the USB device needs opening
243 adb_mutex_lock(&usb->lock);
244 while (usb->fd != -1)
245 adb_cond_wait(&usb->notify, &usb->lock);
246 adb_mutex_unlock(&usb->lock);
247
248 D("[ usb_thread - opening device ]");
249 do {
250 /* XXX use inotify? */
251 fd = unix_open("/dev/android_adb", O_RDWR);
252 if (fd < 0) {
253 // to support older kernels
254 fd = unix_open("/dev/android", O_RDWR);
255 }
256 if (fd < 0) {
257 adb_sleep_ms(1000);
258 }
259 } while (fd < 0);
260 D("[ opening device succeeded ]");
261
262 close_on_exec(fd);
263 usb->fd = fd;
264
265 D("[ usb_thread - registering device ]");
266 register_usb_transport(usb, 0, 0, 1);
267 }
268
269 // never gets here
270 abort();
271}

273static int usb_adb_write(usb_handle *h, const void *data, int len)
274{
275 int n;
276
277 D("about to write (fd=%d, len=%d)", h->fd, len);
278 n = unix_write(h->fd, data, len);
279 if(n != len) {
280 D("ERROR: fd = %d, n = %d, errno = %d (%s)",
281 h->fd, n, errno, strerror(errno));
282 return -1;
283 }
284 D("[ done fd=%d ]", h->fd);
285 return 0;
286}
287
288static int usb_adb_read(usb_handle *h, void *data, int len)
289{
290 D("about to read (fd=%d, len=%d)", h->fd, len);
291 while (len > 0) {
292 // The kernel implementation of adb_read in f_adb.c doesn't support
293 // reads larger then 4096 bytes. Read the data in 4096 byte chunks to
294 // avoid the issue. (The ffs implementation doesn't have this limit.)
295 int bytes_to_read = len < 4096 ? len : 4096;
296 int n = unix_read(h->fd, data, bytes_to_read);
297 if (n != bytes_to_read) {
298 D("ERROR: fd = %d, n = %d, errno = %d (%s)",
299 h->fd, n, errno, strerror(errno));
300 return -1;
301 }
302 len -= n;
303 data = ((char*)data) + n;
304 }
305 D("[ done fd=%d ]", h->fd);
306 return 0;
307}

309static void usb_adb_kick(usb_handle *h)
310{
311 D("usb_kick");
312 adb_mutex_lock(&h->lock);
313 unix_close(h->fd);
314 h->fd = -1;
315
316 // notify usb_adb_open_thread that we are disconnected
317 adb_cond_signal(&h->notify);
318 adb_mutex_unlock(&h->lock);
319}

321static void usb_adb_init()
322{
323 usb_handle* h = reinterpret_cast<usb_handle*>(calloc(1, sizeof(usb_handle)));
324 if (h == nullptr) fatal("couldn't allocate usb_handle");
325
326 h->write = usb_adb_write;
327 h->read = usb_adb_read;
328 h->kick = usb_adb_kick;
329 h->fd = -1;
330
331 adb_cond_init(&h->notify, 0);
332 adb_mutex_init(&h->lock, 0);
333
334 // Open the file /dev/android_adb_enable to trigger
335 // the enabling of the adb USB function in the kernel.
336 // We never touch this file again - just leave it open
337 // indefinitely so the kernel will know when we are running
338 // and when we are not.
339 int fd = unix_open("/dev/android_adb_enable", O_RDWR);
340 if (fd < 0) {
341 D("failed to open /dev/android_adb_enable");
342 } else {
343 close_on_exec(fd);
344 }
345
346 D("[ usb_init - starting thread ]");
347 if (!adb_thread_create(usb_adb_open_thread, h)) {
348 fatal_errno("cannot create usb thread");
349 }
350}

353static void init_functionfs(struct usb_handle *h)
354{
355 ssize_t ret;
356 struct desc_v1 v1_descriptor;
357 struct desc_v2 v2_descriptor;
358
359 v2_descriptor.header.magic = cpu_to_le32(FUNCTIONFS_DESCRIPTORS_MAGIC_V2);
360 v2_descriptor.header.length = cpu_to_le32(sizeof(v2_descriptor));
361 v2_descriptor.header.flags = FUNCTIONFS_HAS_FS_DESC | FUNCTIONFS_HAS_HS_DESC |
362 FUNCTIONFS_HAS_SS_DESC | FUNCTIONFS_HAS_MS_OS_DESC;
363 v2_descriptor.fs_count = 3;
364 v2_descriptor.hs_count = 3;
365 v2_descriptor.ss_count = 5;
366 v2_descriptor.os_count = 1;
367 v2_descriptor.fs_descs = fs_descriptors;
368 v2_descriptor.hs_descs = hs_descriptors;
369 v2_descriptor.ss_descs = ss_descriptors;
370 v2_descriptor.os_header = os_desc_header;
371 v2_descriptor.os_desc = os_desc_compat;
372
373 if (h->control < 0) { // might have already done this before
374 D("OPENING %s", USB_FFS_ADB_EP0);
375 h->control = adb_open(USB_FFS_ADB_EP0, O_RDWR);
376 if (h->control < 0) {
377 D("[ %s: cannot open control endpoint: errno=%d]", USB_FFS_ADB_EP0, errno);
378 goto err;
379 }
380
381 ret = adb_write(h->control, &v2_descriptor, sizeof(v2_descriptor));
382 if (ret < 0) {
383 v1_descriptor.header.magic = cpu_to_le32(FUNCTIONFS_DESCRIPTORS_MAGIC);
384 v1_descriptor.header.length = cpu_to_le32(sizeof(v1_descriptor));
385 v1_descriptor.header.fs_count = 3;
386 v1_descriptor.header.hs_count = 3;
387 v1_descriptor.fs_descs = fs_descriptors;
388 v1_descriptor.hs_descs = hs_descriptors;
389 D("[ %s: Switching to V1_descriptor format errno=%d ]", USB_FFS_ADB_EP0, errno);
390 ret = adb_write(h->control, &v1_descriptor, sizeof(v1_descriptor));
391 if (ret < 0) {
392 D("[ %s: write descriptors failed: errno=%d ]", USB_FFS_ADB_EP0, errno);
393 goto err;
394 }
395 }
396
397 ret = adb_write(h->control, &strings, sizeof(strings));
398 if (ret < 0) {
399 D("[ %s: writing strings failed: errno=%d]", USB_FFS_ADB_EP0, errno);
400 goto err;
401 }
402 }
403
404 h->bulk_out = adb_open(USB_FFS_ADB_OUT, O_RDWR);
405 if (h->bulk_out < 0) {
406 D("[ %s: cannot open bulk-out ep: errno=%d ]", USB_FFS_ADB_OUT, errno);
407 goto err;
408 }
409
410 h->bulk_in = adb_open(USB_FFS_ADB_IN, O_RDWR);
411 if (h->bulk_in < 0) {
412 D("[ %s: cannot open bulk-in ep: errno=%d ]", USB_FFS_ADB_IN, errno);
413 goto err;
414 }
415
416 return;
417
418err:
419 if (h->bulk_in > 0) {
420 adb_close(h->bulk_in);
421 h->bulk_in = -1;
422 }
423 if (h->bulk_out > 0) {
424 adb_close(h->bulk_out);
425 h->bulk_out = -1;
426 }
427 if (h->control > 0) {
428 adb_close(h->control);
429 h->control = -1;
430 }
431 return;
432}

434static void usb_ffs_open_thread(void* x) {
435 struct usb_handle *usb = (struct usb_handle *)x;
436
437 adb_thread_setname("usb ffs open");
438
439 while (true) {
440 // wait until the USB device needs opening
441 adb_mutex_lock(&usb->lock);
442 while (usb->control != -1 && usb->bulk_in != -1 && usb->bulk_out != -1)
443 adb_cond_wait(&usb->notify, &usb->lock);
444 adb_mutex_unlock(&usb->lock);
445
446 while (true) {
447 init_functionfs(usb);
448
449 if (usb->control >= 0 && usb->bulk_in >= 0 && usb->bulk_out >= 0)
450 break;
451
452 adb_sleep_ms(1000);
453 }
454 property_set("sys.usb.ffs.ready", "1");
455
456 D("[ usb_thread - registering device ]");
457 register_usb_transport(usb, 0, 0, 1);
458 }
459
460 // never gets here
461 abort();
462}

464static int usb_ffs_write(usb_handle* h, const void* data, int len) {
465 D("about to write (fd=%d, len=%d)", h->bulk_in, len);
466
467 const char* buf = static_cast<const char*>(data);
468 while (len > 0) {
469 int write_len = std::min(USB_FFS_MAX_WRITE, len);
470 int n = adb_write(h->bulk_in, buf, write_len);
471 if (n < 0) {
472 D("ERROR: fd = %d, n = %d: %s", h->bulk_in, n, strerror(errno));
473 return -1;
474 }
475 buf += n;
476 len -= n;
477 }
478
479 D("[ done fd=%d ]", h->bulk_in);
480 return 0;
481}

483static int usb_ffs_read(usb_handle* h, void* data, int len) {
484 D("about to read (fd=%d, len=%d)", h->bulk_out, len);
485
486 char* buf = static_cast<char*>(data);
487 while (len > 0) {
488 int read_len = std::min(USB_FFS_MAX_READ, len);
489 int n = adb_read(h->bulk_out, buf, read_len);
490 if (n < 0) {
491 D("ERROR: fd = %d, n = %d: %s", h->bulk_out, n, strerror(errno));
492 return -1;
493 }
494 buf += n;
495 len -= n;
496 }
497
498 D("[ done fd=%d ]", h->bulk_out);
499 return 0;
500}

502static void usb_ffs_kick(usb_handle *h)
503{
504 int err;
505
506 err = ioctl(h->bulk_in, FUNCTIONFS_CLEAR_HALT);
507 if (err < 0) {
508 D("[ kick: source (fd=%d) clear halt failed (%d) ]", h->bulk_in, errno);
509 }
510
511 err = ioctl(h->bulk_out, FUNCTIONFS_CLEAR_HALT);
512 if (err < 0) {
513 D("[ kick: sink (fd=%d) clear halt failed (%d) ]", h->bulk_out, errno);
514 }
515
516 adb_mutex_lock(&h->lock);
517
518 // don't close ep0 here, since we may not need to reinitialize it with
519 // the same descriptors again. if however ep1/ep2 fail to re-open in
520 // init_functionfs, only then would we close and open ep0 again.
521 adb_close(h->bulk_out);
522 adb_close(h->bulk_in);
523 h->bulk_out = h->bulk_in = -1;
524
525 // notify usb_ffs_open_thread that we are disconnected
526 adb_cond_signal(&h->notify);
527 adb_mutex_unlock(&h->lock);
528}

530static void usb_ffs_init()
531{
532 D("[ usb_init - using FunctionFS ]");
533
534 usb_handle* h = reinterpret_cast<usb_handle*>(calloc(1, sizeof(usb_handle)));
535 if (h == nullptr) fatal("couldn't allocate usb_handle");
536
537 h->write = usb_ffs_write;
538 h->read = usb_ffs_read;
539 h->kick = usb_ffs_kick;
540 h->control = -1;
541 h->bulk_out = -1;
542 h->bulk_out = -1;
543
544 adb_cond_init(&h->notify, 0);
545 adb_mutex_init(&h->lock, 0);
546
547 D("[ usb_init - starting thread ]");
548 if (!adb_thread_create(usb_ffs_open_thread, h)) {
549 fatal_errno("[ cannot create usb thread ]\n");
550 }
551}

553void usb_init()
554{
555 if (access(USB_FFS_ADB_EP0, F_OK) == 0)
556 usb_ffs_init();
557 else
558 usb_adb_init();
559}
