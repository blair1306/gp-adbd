66/*
67 * gettime() - returns the time in seconds of the system's monotonic clock or
68 * zero on error.
69 */
70static time_t gettime(void)
71{
72 struct timespec ts;
73 int ret;
74
75 ret = clock_gettime(CLOCK_MONOTONIC, &ts);
76 if (ret < 0) {
77 ERROR("clock_gettime(CLOCK_MONOTONIC) failed: %s\n", strerror(errno));
78 return 0;
79 }
80
81 return ts.tv_sec;
82}

84static int wait_for_file(const char *filename, int timeout)
85{
86 struct stat info;
87 time_t timeout_time = gettime() + timeout;
88 int ret = -1;
89
90 while (gettime() < timeout_time && ((ret = stat(filename, &info)) < 0))
91 usleep(10000);
92
93 return ret;
94}

96static void check_fs(char *blk_device, char *fs_type, char *target)
97{
98 int status;
99 int ret;
100 long tmpmnt_flags = MS_NOATIME | MS_NOEXEC | MS_NOSUID;
101 char tmpmnt_opts[64] = "errors=remount-ro";
102 char *e2fsck_argv[] = {
103 E2FSCK_BIN,
104 "-y",
105 blk_device
106 };
107
108 /* Check for the types of filesystems we know how to check */
109 if (!strcmp(fs_type, "ext2") || !strcmp(fs_type, "ext3") || !strcmp(fs_type, "ext4")) {
110 /*
111 * First try to mount and unmount the filesystem. We do this because
112 * the kernel is more efficient than e2fsck in running the journal and
113 * processing orphaned inodes, and on at least one device with a
114 * performance issue in the emmc firmware, it can take e2fsck 2.5 minutes
115 * to do what the kernel does in about a second.
116 *
117 * After mounting and unmounting the filesystem, run e2fsck, and if an
118 * error is recorded in the filesystem superblock, e2fsck will do a full
119 * check. Otherwise, it does nothing. If the kernel cannot mount the
120 * filesytsem due to an error, e2fsck is still run to do a full check
121 * fix the filesystem.
122 */
123 errno = 0;
124 if (!strcmp(fs_type, "ext4")) {
125 // This option is only valid with ext4
126 strlcat(tmpmnt_opts, ",nomblk_io_submit", sizeof(tmpmnt_opts));
127 }
128 ret = mount(blk_device, target, fs_type, tmpmnt_flags, tmpmnt_opts);
129 INFO("%s(): mount(%s,%s,%s)=%d: %s\n",
130 __func__, blk_device, target, fs_type, ret, strerror(errno));
131 if (!ret) {
132 int i;
133 for (i = 0; i < 5; i++) {
134 // Try to umount 5 times before continuing on.
135 // Should we try rebooting if all attempts fail?
136 int result = umount(target);
137 if (result == 0) {
138 INFO("%s(): unmount(%s) succeeded\n", __func__, target);
139 break;
140 }
141 ERROR("%s(): umount(%s)=%d: %s\n", __func__, target, result, strerror(errno));
142 sleep(1);
143 }
144 }
145
146 /*
147 * Some system images do not have e2fsck for licensing reasons
148 * (e.g. recent SDK system images). Detect these and skip the check.
149 */
150 if (access(E2FSCK_BIN, X_OK)) {
151 INFO("Not running %s on %s (executable not in system image)\n",
152 E2FSCK_BIN, blk_device);
153 } else {
154 INFO("Running %s on %s\n", E2FSCK_BIN, blk_device);
155
156 ret = android_fork_execvp_ext(ARRAY_SIZE(e2fsck_argv), e2fsck_argv,
157 &status, true, LOG_KLOG | LOG_FILE,
158 true, FSCK_LOG_FILE, NULL, 0);
159
160 if (ret < 0) {
161 /* No need to check for error in fork, we can't really handle it now */
162 ERROR("Failed trying to run %s\n", E2FSCK_BIN);
163 }
164 }
165 } else if (!strcmp(fs_type, "f2fs")) {
166 char *f2fs_fsck_argv[] = {
167 F2FS_FSCK_BIN,
168 "-a",
169 blk_device
170 };
171 INFO("Running %s -a %s\n", F2FS_FSCK_BIN, blk_device);
172
173 ret = android_fork_execvp_ext(ARRAY_SIZE(f2fs_fsck_argv), f2fs_fsck_argv,
174 &status, true, LOG_KLOG | LOG_FILE,
175 true, FSCK_LOG_FILE, NULL, 0);
176 if (ret < 0) {
177 /* No need to check for error in fork, we can't really handle it now */
178 ERROR("Failed trying to run %s\n", F2FS_FSCK_BIN);
179 }
180 }
181
182 return;
183}

185static void remove_trailing_slashes(char *n)
186{
187 int len;
188
189 len = strlen(n) - 1;
190 while ((*(n + len) == '/') && len) {
191 *(n + len) = '\0';
192 len--;
193 }
194}

196/*
197 * Mark the given block device as read-only, using the BLKROSET ioctl.
198 * Return 0 on success, and -1 on error.
199 */
200int fs_mgr_set_blk_ro(const char *blockdev)
201{
202 int fd;
203 int rc = -1;
204 int ON = 1;
205
206 fd = TEMP_FAILURE_RETRY(open(blockdev, O_RDONLY | O_CLOEXEC));
207 if (fd < 0) {
208 // should never happen
209 return rc;
210 }
211
212 rc = ioctl(fd, BLKROSET, &ON);
213 close(fd);
214
215 return rc;
216}

218/*
219 * __mount(): wrapper around the mount() system call which also
220 * sets the underlying block device to read-only if the mount is read-only.
221 * See "man 2 mount" for return values.
222 */
223static int __mount(const char *source, const char *target, const struct fstab_rec *rec)
224{
225 unsigned long mountflags = rec->flags;
226 int ret;
227 int save_errno;
228
229 /* We need this because sometimes we have legacy symlinks
230 * that are lingering around and need cleaning up.
231 */
232 struct stat info;
233 if (!lstat(target, &info))
234 if ((info.st_mode & S_IFMT) == S_IFLNK)
235 unlink(target);
236 mkdir(target, 0755);
237 ret = mount(source, target, rec->fs_type, mountflags, rec->fs_options);
238 save_errno = errno;
239 INFO("%s(source=%s,target=%s,type=%s)=%d\n", __func__, source, target, rec->fs_type, ret);
240 if ((ret == 0) && (mountflags & MS_RDONLY) != 0) {
241 fs_mgr_set_blk_ro(source);
242 }
243 errno = save_errno;
244 return ret;
245}

247static int fs_match(char *in1, char *in2)
248{
249 char *n1;
250 char *n2;
251 int ret;
252
253 n1 = strdup(in1);
254 n2 = strdup(in2);
255
256 remove_trailing_slashes(n1);
257 remove_trailing_slashes(n2);
258
259 ret = !strcmp(n1, n2);
260
261 free(n1);
262 free(n2);
263
264 return ret;
265}

295/*
296 * Tries to mount any of the consecutive fstab entries that match
297 * the mountpoint of the one given by fstab->recs[start_idx].
298 *
299 * end_idx: On return, will be the last rec that was looked at.
300 * attempted_idx: On return, will indicate which fstab rec
301 * succeeded. In case of failure, it will be the start_idx.
302 * Returns
303 * -1 on failure with errno set to match the 1st mount failure.
304 * 0 on success.
305 */
306static int mount_with_alternatives(struct fstab *fstab, int start_idx, int *end_idx, int *attempted_idx)
307{
308 int i;
309 int mount_errno = 0;
310 int mounted = 0;
311
312 if (!end_idx || !attempted_idx || start_idx >= fstab->num_entries) {
313 errno = EINVAL;
314 if (end_idx) *end_idx = start_idx;
315 if (attempted_idx) *end_idx = start_idx;
316 return -1;
317 }
318
319 /* Hunt down an fstab entry for the same mount point that might succeed */
320 for (i = start_idx;
321 /* We required that fstab entries for the same mountpoint be consecutive */
322 i < fstab->num_entries && !strcmp(fstab->recs[start_idx].mount_point, fstab->recs[i].mount_point);
323 i++) {
324 /*
325 * Don't try to mount/encrypt the same mount point again.
326 * Deal with alternate entries for the same point which are required to be all following
327 * each other.
328 */
329 if (mounted) {
330 ERROR("%s(): skipping fstab dup mountpoint=%s rec[%d].fs_type=%s already mounted as %s.\n", __func__,
331 fstab->recs[i].mount_point, i, fstab->recs[i].fs_type, fstab->recs[*attempted_idx].fs_type);
332 continue;
333 }
334
335 if (fstab->recs[i].fs_mgr_flags & MF_CHECK) {
336 check_fs(fstab->recs[i].blk_device, fstab->recs[i].fs_type,
337 fstab->recs[i].mount_point);
338 }
339 if (!__mount(fstab->recs[i].blk_device, fstab->recs[i].mount_point, &fstab->recs[i])) {
340 *attempted_idx = i;
341 mounted = 1;
342 if (i != start_idx) {
343 ERROR("%s(): Mounted %s on %s with fs_type=%s instead of %s\n", __func__,
344 fstab->recs[i].blk_device, fstab->recs[i].mount_point, fstab->recs[i].fs_type,
345 fstab->recs[start_idx].fs_type);
346 }
347 } else {
348 /* back up errno for crypto decisions */
349 mount_errno = errno;
350 }
351 }
352
353 /* Adjust i for the case where it was still withing the recs[] */
354 if (i < fstab->num_entries) --i;
355
356 *end_idx = i;
357 if (!mounted) {
358 *attempted_idx = start_idx;
359 errno = mount_errno;
360 return -1;
361 }
362 return 0;
363}

365static int translate_ext_labels(struct fstab_rec *rec)
366{
367 DIR *blockdir = NULL;
368 struct dirent *ent;
369 char *label;
370 size_t label_len;
371 int ret = -1;
372
373 if (strncmp(rec->blk_device, "LABEL=", 6))
374 return 0;
375
376 label = rec->blk_device + 6;
377 label_len = strlen(label);
378
379 if (label_len > 16) {
380 ERROR("FS label is longer than allowed by filesystem\n");
381 goto out;
382 }
383
384
385 blockdir = opendir("/dev/block");
386 if (!blockdir) {
387 ERROR("couldn't open /dev/block\n");
388 goto out;
389 }
390
391 while ((ent = readdir(blockdir))) {
392 int fd;
393 char super_buf[1024];
394 struct ext4_super_block *sb;
395
396 if (ent->d_type != DT_BLK)
397 continue;
398
399 fd = openat(dirfd(blockdir), ent->d_name, O_RDONLY);
400 if (fd < 0) {
401 ERROR("Cannot open block device /dev/block/%s\n", ent->d_name);
402 goto out;
403 }
404
405 if (TEMP_FAILURE_RETRY(lseek(fd, 1024, SEEK_SET)) < 0 ||
406 TEMP_FAILURE_RETRY(read(fd, super_buf, 1024)) != 1024) {
407 /* Probably a loopback device or something else without a readable
408 * superblock.
409 */
410 close(fd);
411 continue;
412 }
413
414 sb = (struct ext4_super_block *)super_buf;
415 if (sb->s_magic != EXT4_SUPER_MAGIC) {
416 INFO("/dev/block/%s not ext{234}\n", ent->d_name);
417 continue;
418 }
419
420 if (!strncmp(label, sb->s_volume_name, label_len)) {
421 char *new_blk_device;
422
423 if (asprintf(&new_blk_device, "/dev/block/%s", ent->d_name) < 0) {
424 ERROR("Could not allocate block device string\n");
425 goto out;
426 }
427
428 INFO("resolved label %s to %s\n", rec->blk_device, new_blk_device);
429
430 free(rec->blk_device);
431 rec->blk_device = new_blk_device;
432 ret = 0;
433 break;
434 }
435 }
436
437out:
438 closedir(blockdir);
439 return ret;
440}

442// Check to see if a mountable volume has encryption requirements
443static int handle_encryptable(struct fstab *fstab, const struct fstab_rec* rec)
444{
445 /* If this is block encryptable, need to trigger encryption */
446 if ( (rec->fs_mgr_flags & MF_FORCECRYPT)
447 || (device_is_force_encrypted() && fs_mgr_is_encryptable(rec))) {
448 if (umount(rec->mount_point) == 0) {
449 return FS_MGR_MNTALL_DEV_NEEDS_ENCRYPTION;
450 } else {
451 WARNING("Could not umount %s (%s) - allow continue unencrypted\n",
452 rec->mount_point, strerror(errno));
453 return FS_MGR_MNTALL_DEV_NOT_ENCRYPTED;
454 }
455 }
456
457 // Deal with file level encryption
458 if (rec->fs_mgr_flags & MF_FILEENCRYPTION) {
459 // Default or not yet initialized encryption requires no more work here
460 if (!e4crypt_non_default_key(rec->mount_point)) {
461 INFO("%s is default file encrypted\n", rec->mount_point);
462 return FS_MGR_MNTALL_DEV_DEFAULT_FILE_ENCRYPTED;
463 }
464
465 INFO("%s is non-default file encrypted\n", rec->mount_point);
466
467 // Uses non-default key, so must unmount and set up temp file system
468 if (umount(rec->mount_point)) {
469 ERROR("Failed to umount %s - rebooting\n", rec->mount_point);
470 return FS_MGR_MNTALL_FAIL;
471 }
472
473 if (fs_mgr_do_tmpfs_mount(rec->mount_point) != 0) {
474 ERROR("Failed to mount a tmpfs at %s\n", rec->mount_point);
475 return FS_MGR_MNTALL_FAIL;
476 }
477
478 // Mount data temporarily so we can access unencrypted dir
479 char tmp_mnt[PATH_MAX];
480 strlcpy(tmp_mnt, rec->mount_point, sizeof(tmp_mnt));
481 strlcat(tmp_mnt, "/tmp_mnt", sizeof(tmp_mnt));
482 if (mkdir(tmp_mnt, 0700)) {
483 ERROR("Failed to create temp mount point\n");
484 return FS_MGR_MNTALL_FAIL;
485 }
486
487 if (fs_mgr_do_mount(fstab, rec->mount_point,
488 rec->blk_device, tmp_mnt)) {
489 ERROR("Error temp mounting encrypted file system\n");
490 return FS_MGR_MNTALL_FAIL;
491 }
492
493 return FS_MGR_MNTALL_DEV_NON_DEFAULT_FILE_ENCRYPTED;
494 }
495
496 return FS_MGR_MNTALL_DEV_NOT_ENCRYPTED;
497}

499/* When multiple fstab records share the same mount_point, it will
500 * try to mount each one in turn, and ignore any duplicates after a
501 * first successful mount.
502 * Returns -1 on error, and FS_MGR_MNTALL_* otherwise.
503 */
504int fs_mgr_mount_all(struct fstab *fstab)
505{
506 int i = 0;
507 int encryptable = FS_MGR_MNTALL_DEV_NOT_ENCRYPTED;
508 int error_count = 0;
509 int mret = -1;
510 int mount_errno = 0;
511 int attempted_idx = -1;
512
513 if (!fstab) {
514 return -1;
515 }
516
517 for (i = 0; i < fstab->num_entries; i++) {
518 /* Don't mount entries that are managed by vold */
519 if (fstab->recs[i].fs_mgr_flags & (MF_VOLDMANAGED | MF_RECOVERYONLY)) {
520 continue;
521 }
522
523 /* Skip swap and raw partition entries such as boot, recovery, etc */
524 if (!strcmp(fstab->recs[i].fs_type, "swap") ||
525 !strcmp(fstab->recs[i].fs_type, "emmc") ||
526 !strcmp(fstab->recs[i].fs_type, "mtd")) {
527 continue;
528 }
529
530 /* Skip mounting the root partition, as it will already have been mounted */
531 if (!strcmp(fstab->recs[i].mount_point, "/")) {
532 if ((fstab->recs[i].fs_mgr_flags & MS_RDONLY) != 0) {
533 fs_mgr_set_blk_ro(fstab->recs[i].blk_device);
534 }
535 continue;
536 }
537
538 /* Translate LABEL= file system labels into block devices */
539 if (!strcmp(fstab->recs[i].fs_type, "ext2") ||
540 !strcmp(fstab->recs[i].fs_type, "ext3") ||
541 !strcmp(fstab->recs[i].fs_type, "ext4")) {
542 int tret = translate_ext_labels(&fstab->recs[i]);
543 if (tret < 0) {
544 ERROR("Could not translate label to block device\n");
545 continue;
546 }
547 }
548
549 if (fstab->recs[i].fs_mgr_flags & MF_WAIT) {
550 wait_for_file(fstab->recs[i].blk_device, WAIT_TIMEOUT);
551 }
552
553 if ((fstab->recs[i].fs_mgr_flags & MF_VERIFY) && device_is_secure()) {
554 int rc = fs_mgr_setup_verity(&fstab->recs[i]);
555 if (device_is_debuggable() && rc == FS_MGR_SETUP_VERITY_DISABLED) {
556 INFO("Verity disabled");
557 } else if (rc != FS_MGR_SETUP_VERITY_SUCCESS) {
558 ERROR("Could not set up verified partition, skipping!\n");
559 continue;
560 }
561 }
562 int last_idx_inspected;
563 int top_idx = i;
564
565 mret = mount_with_alternatives(fstab, i, &last_idx_inspected, &attempted_idx);
566 i = last_idx_inspected;
567 mount_errno = errno;
568
569 /* Deal with encryptability. */
570 if (!mret) {
571 int status = handle_encryptable(fstab, &fstab->recs[attempted_idx]);
572
573 if (status == FS_MGR_MNTALL_FAIL) {
574 /* Fatal error - no point continuing */
575 return status;
576 }
577
578 if (status != FS_MGR_MNTALL_DEV_NOT_ENCRYPTED) {
579 if (encryptable != FS_MGR_MNTALL_DEV_NOT_ENCRYPTED) {
580 // Log and continue
581 ERROR("Only one encryptable/encrypted partition supported\n");
582 }
583 encryptable = status;
584 }
585
586 /* Success! Go get the next one */
587 continue;
588 }
589
590 /* mount(2) returned an error, handle the encryptable/formattable case */
591 bool wiped = partition_wiped(fstab->recs[top_idx].blk_device);
592 if (mret && mount_errno != EBUSY && mount_errno != EACCES &&
593 fs_mgr_is_formattable(&fstab->recs[top_idx]) && wiped) {
594 /* top_idx and attempted_idx point at the same partition, but sometimes
595 * at two different lines in the fstab. Use the top one for formatting
596 * as that is the preferred one.
597 */
598 ERROR("%s(): %s is wiped and %s %s is formattable. Format it.\n", __func__,
599 fstab->recs[top_idx].blk_device, fstab->recs[top_idx].mount_point,
600 fstab->recs[top_idx].fs_type);
601 if (fs_mgr_is_encryptable(&fstab->recs[top_idx]) &&
602 strcmp(fstab->recs[top_idx].key_loc, KEY_IN_FOOTER)) {
603 int fd = open(fstab->recs[top_idx].key_loc, O_WRONLY);
604 if (fd >= 0) {
605 INFO("%s(): also wipe %s\n", __func__, fstab->recs[top_idx].key_loc);
606 wipe_block_device(fd, get_file_size(fd));
607 close(fd);
608 } else {
609 ERROR("%s(): %s wouldn't open (%s)\n", __func__,
610 fstab->recs[top_idx].key_loc, strerror(errno));
611 }
612 }
613 if (fs_mgr_do_format(&fstab->recs[top_idx]) == 0) {
614 /* Let's replay the mount actions. */
615 i = top_idx - 1;
616 continue;
617 }
618 }
619 if (mret && mount_errno != EBUSY && mount_errno != EACCES &&
620 fs_mgr_is_encryptable(&fstab->recs[attempted_idx])) {
621 if (wiped) {
622 ERROR("%s(): %s is wiped and %s %s is encryptable. Suggest recovery...\n", __func__,
623 fstab->recs[attempted_idx].blk_device, fstab->recs[attempted_idx].mount_point,
624 fstab->recs[attempted_idx].fs_type);
625 encryptable = FS_MGR_MNTALL_DEV_NEEDS_RECOVERY;
626 continue;
627 } else {
628 /* Need to mount a tmpfs at this mountpoint for now, and set
629 * properties that vold will query later for decrypting
630 */
631 ERROR("%s(): possibly an encryptable blkdev %s for mount %s type %s )\n", __func__,
632 fstab->recs[attempted_idx].blk_device, fstab->recs[attempted_idx].mount_point,
633 fstab->recs[attempted_idx].fs_type);
634 if (fs_mgr_do_tmpfs_mount(fstab->recs[attempted_idx].mount_point) < 0) {
635 ++error_count;
636 continue;
637 }
638 }
639 encryptable = FS_MGR_MNTALL_DEV_MIGHT_BE_ENCRYPTED;
640 } else {
641 ERROR("Failed to mount an un-encryptable or wiped partition on"
642 "%s at %s options: %s error: %s\n",
643 fstab->recs[attempted_idx].blk_device, fstab->recs[attempted_idx].mount_point,
644 fstab->recs[attempted_idx].fs_options, strerror(mount_errno));
645 ++error_count;
646 continue;
647 }
648 }
649
650 if (error_count) {
651 return -1;
652 } else {
653 return encryptable;
654 }
655}

657/* If tmp_mount_point is non-null, mount the filesystem there. This is for the
658 * tmp mount we do to check the user password
659 * If multiple fstab entries are to be mounted on "n_name", it will try to mount each one
660 * in turn, and stop on 1st success, or no more match.
661 */
662int fs_mgr_do_mount(struct fstab *fstab, char *n_name, char *n_blk_device,
663 char *tmp_mount_point)
664{
665 int i = 0;
666 int ret = FS_MGR_DOMNT_FAILED;
667 int mount_errors = 0;
668 int first_mount_errno = 0;
669 char *m;
670
671 if (!fstab) {
672 return ret;
673 }
674
675 for (i = 0; i < fstab->num_entries; i++) {
676 if (!fs_match(fstab->recs[i].mount_point, n_name)) {
677 continue;
678 }
679
680 /* We found our match */
681 /* If this swap or a raw partition, report an error */
682 if (!strcmp(fstab->recs[i].fs_type, "swap") ||
683 !strcmp(fstab->recs[i].fs_type, "emmc") ||
684 !strcmp(fstab->recs[i].fs_type, "mtd")) {
685 ERROR("Cannot mount filesystem of type %s on %s\n",
686 fstab->recs[i].fs_type, n_blk_device);
687 goto out;
688 }
689
690 /* First check the filesystem if requested */
691 if (fstab->recs[i].fs_mgr_flags & MF_WAIT) {
692 wait_for_file(n_blk_device, WAIT_TIMEOUT);
693 }
694
695 if (fstab->recs[i].fs_mgr_flags & MF_CHECK) {
696 check_fs(n_blk_device, fstab->recs[i].fs_type,
697 fstab->recs[i].mount_point);
698 }
699
700 if ((fstab->recs[i].fs_mgr_flags & MF_VERIFY) && device_is_secure()) {
701 int rc = fs_mgr_setup_verity(&fstab->recs[i]);
702 if (device_is_debuggable() && rc == FS_MGR_SETUP_VERITY_DISABLED) {
703 INFO("Verity disabled");
704 } else if (rc != FS_MGR_SETUP_VERITY_SUCCESS) {
705 ERROR("Could not set up verified partition, skipping!\n");
706 continue;
707 }
708 }
709
710 /* Now mount it where requested */
711 if (tmp_mount_point) {
712 m = tmp_mount_point;
713 } else {
714 m = fstab->recs[i].mount_point;
715 }
716 if (__mount(n_blk_device, m, &fstab->recs[i])) {
717 if (!first_mount_errno) first_mount_errno = errno;
718 mount_errors++;
719 continue;
720 } else {
721 ret = 0;
722 goto out;
723 }
724 }
725 if (mount_errors) {
726 ERROR("Cannot mount filesystem on %s at %s. error: %s\n",
727 n_blk_device, m, strerror(first_mount_errno));
728 if (first_mount_errno == EBUSY) {
729 ret = FS_MGR_DOMNT_BUSY;
730 } else {
731 ret = FS_MGR_DOMNT_FAILED;
732 }
733 } else {
734 /* We didn't find a match, say so and return an error */
735 ERROR("Cannot find mount point %s in fstab\n", fstab->recs[i].mount_point);
736 }
737
738out:
739 return ret;
740}

743 * mount a tmpfs filesystem at the given point.
744 * return 0 on success, non-zero on failure.
745 */
746int fs_mgr_do_tmpfs_mount(char *n_name)
747{
748 int ret;

750 ret = mount("tmpfs", n_name, "tmpfs",

762{
763 int i = 0;
764 int ret = 0;
765
766 if (!fstab) {
767 return -1;
768 }
769
770 while (fstab->recs[i].blk_device) {
771 if (umount(fstab->recs[i].mount_point)) {

781/* This must be called after mount_all, because the mkswap command needs to be
782 * available.
783 */
784int fs_mgr_swapon_all(struct fstab *fstab)
785{
786 int i = 0;
787 int flags = 0;
788 int err = 0;
789 int ret = 0;
790 int status;
791 char *mkswap_argv[2] = {
792 MKSWAP_BIN,
793 NULL
794 };
795
796 if (!fstab) {
797 return -1;
798 }
799
800 for (i = 0; i < fstab->num_entries; i++) {
801 /* Skip non-swap entries */
802 if (strcmp(fstab->recs[i].fs_type, "swap")) {
803 continue;
804 }
805
806 if (fstab->recs[i].zram_size > 0) {
807 /* A zram_size was specified, so we need to configure the
808 * device. There is no point in having multiple zram devices
809 * on a system (all the memory comes from the same pool) so
810 * we can assume the device number is 0.
811 */
812 FILE *zram_fp;
813
814 zram_fp = fopen(ZRAM_CONF_DEV, "r+");
815 if (zram_fp == NULL) {
816 ERROR("Unable to open zram conf device %s\n", ZRAM_CONF_DEV);
817 ret = -1;
818 continue;
819 }
820 fprintf(zram_fp, "%d\n", fstab->recs[i].zram_size);
821 fclose(zram_fp);
822 }
823
824 if (fstab->recs[i].fs_mgr_flags & MF_WAIT) {
825 wait_for_file(fstab->recs[i].blk_device, WAIT_TIMEOUT);
826 }
827
828 /* Initialize the swap area */
829 mkswap_argv[1] = fstab->recs[i].blk_device;
830 err = android_fork_execvp_ext(ARRAY_SIZE(mkswap_argv), mkswap_argv,
831 &status, true, LOG_KLOG, false, NULL,
832 NULL, 0);
833 if (err) {
834 ERROR("mkswap failed for %s\n", fstab->recs[i].blk_device);
835 ret = -1;
836 continue;
837 }
838
839 /* If -1, then no priority was specified in fstab, so don't set
840 * SWAP_FLAG_PREFER or encode the priority */
841 if (fstab->recs[i].swap_prio >= 0) {
842 flags = (fstab->recs[i].swap_prio << SWAP_FLAG_PRIO_SHIFT) &
843 SWAP_FLAG_PRIO_MASK;
844 flags |= SWAP_FLAG_PREFER;
845 } else {
846 flags = 0;
847 }
848 err = swapon(fstab->recs[i].blk_device, flags);
849 if (err) {
850 ERROR("swapon failed for %s\n", fstab->recs[i].blk_device);
851 ret = -1;
852 }
853 }
854
855 return ret;
856}

858/*
859 * key_loc must be at least PROPERTY_VALUE_MAX bytes long
860 *
861 * real_blk_device must be at least PROPERTY_VALUE_MAX bytes long
862 */
863int fs_mgr_get_crypt_info(struct fstab *fstab, char *key_loc, char *real_blk_device, int size)
864{
865 int i = 0;
866
867 if (!fstab) {
868 return -1;
869 }
870 /* Initialize return values to null strings */
871 if (key_loc) {
872 *key_loc = '\0';
873 }
874 if (real_blk_device) {
875 *real_blk_device = '\0';
876 }
877
878 /* Look for the encryptable partition to find the data */
879 for (i = 0; i < fstab->num_entries; i++) {
880 /* Don't deal with vold managed enryptable partitions here */
881 if (fstab->recs[i].fs_mgr_flags & MF_VOLDMANAGED) {
882 continue;
883 }
884 if (!(fstab->recs[i].fs_mgr_flags & (MF_CRYPT | MF_FORCECRYPT))) {
885 continue;
886 }
887
888 /* We found a match */
889 if (key_loc) {
890 strlcpy(key_loc, fstab->recs[i].key_loc, size);
891 }
892 if (real_blk_device) {
893 strlcpy(real_blk_device, fstab->recs[i].blk_device, size);
894 }
895 break;
896 }
897
898 return 0;
899}
