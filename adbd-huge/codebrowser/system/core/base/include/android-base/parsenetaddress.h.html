17#ifndef ANDROID_BASE_PARSENETADDRESS_H

20#include <string>
21
22namespace android {
23namespace base {
24
25// Parses |address| into |host| and |port|.
26//
27// If |address| doesn't contain a port number, the default value is taken from
28// |port|. If |canonical_address| is non-null it will be set to "host:port" or
29// "[host]:port" as appropriate.
30//
31// On failure, returns false and fills |error|.
32bool ParseNetAddress(const std::string& address, std::string* host, int* port,
33 std::string* canonical_address, std::string* error);
34
35} // namespace base
36} // namespace android
