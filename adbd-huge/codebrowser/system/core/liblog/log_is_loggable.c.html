28static pthread_mutex_t lock_loggable = PTHREAD_MUTEX_INITIALIZER;
29
30static int lock()
31{
32 /*
33 * If we trigger a signal handler in the middle of locked activity and the
34 * signal handler logs a message, we could get into a deadlock state.
35 */
36 /*
37 * Any contention, and we can turn around and use the non-cached method
38 * in less time than the system call associated with a mutex to deal with
39 * the contention.
40 */
41 return pthread_mutex_trylock(&lock_loggable);
42}
43
44static void unlock()
45{
46 pthread_mutex_unlock(&lock_loggable);
47}

49struct cache {
50 const prop_info *pinfo;
51 uint32_t serial;
52 unsigned char c;
53};
54
55static int check_cache(struct cache *cache)
56{
57 return cache->pinfo
58 && __system_property_serial(cache->pinfo) != cache->serial;
59}
60
61#define BOOLEAN_TRUE 0xFF
62#define BOOLEAN_FALSE 0xFE
63
64static void refresh_cache(struct cache *cache, const char *key)
65{
66 uint32_t serial;
67 char buf[PROP_VALUE_MAX];
68
69 if (!cache->pinfo) {
70 cache->pinfo = __system_property_find(key);
71 if (!cache->pinfo) {
72 return;
73 }
74 cache->serial = -1;
75 }
76 serial = __system_property_serial(cache->pinfo);
77 if (serial == cache->serial) {
78 return;
79 }
80 cache->serial = serial;
81 __system_property_read(cache->pinfo, 0, buf);
82 switch(buf[0]) {
83 case 't': case 'T':
84 cache->c = strcasecmp(buf + 1, "rue") ? buf[0] : BOOLEAN_TRUE;
85 break;
86 case 'f': case 'F':
87 cache->c = strcasecmp(buf + 1, "alse") ? buf[0] : BOOLEAN_FALSE;
88 break;
89 default:
90 cache->c = buf[0];
91 }
92}

94static int __android_log_level(const char *tag, int default_prio)
95{
96 /* sizeof() is used on this array below */
97 static const char log_namespace[] = "persist.log.tag.";
98 static const size_t base_offset = 8; /* skip "persist." */
99 /* calculate the size of our key temporary buffer */
100 const size_t taglen = (tag && *tag) ? strlen(tag) : 0;
101 /* sizeof(log_namespace) = strlen(log_namespace) + 1 */
102 char key[sizeof(log_namespace) + taglen]; /* may be > PROPERTY_KEY_MAX */
103 char *kp;
104 size_t i;
105 char c = 0;
106 /*
107 * Single layer cache of four properties. Priorities are:
108 * log.tag.<tag>
109 * persist.log.tag.<tag>
110 * log.tag
111 * persist.log.tag
112 * Where the missing tag matches all tags and becomes the
113 * system global default. We do not support ro.log.tag* .
114 */
115 static char last_tag[PROP_NAME_MAX];
116 static uint32_t global_serial;
117 /* some compilers erroneously see uninitialized use. !not_locked */
118 uint32_t current_global_serial = 0;
119 static struct cache tag_cache[2];
120 static struct cache global_cache[2];
121 int change_detected;
122 int global_change_detected;
123 int not_locked;
124
125 strcpy(key, log_namespace);
126
127 global_change_detected = change_detected = not_locked = lock();
128
129 if (!not_locked) {
130 /*
131 * check all known serial numbers to changes.
132 */
133 for (i = 0; i < (sizeof(tag_cache) / sizeof(tag_cache[0])); ++i) {
134 if (check_cache(&tag_cache[i])) {
135 change_detected = 1;
136 }
137 }
138 for (i = 0; i < (sizeof(global_cache) / sizeof(global_cache[0])); ++i) {
139 if (check_cache(&global_cache[i])) {
140 global_change_detected = 1;
141 }
142 }
143
144 current_global_serial = __system_property_area_serial();
145 if (current_global_serial != global_serial) {
146 change_detected = 1;
147 global_change_detected = 1;
148 }
149 }
150
151 if (taglen) {
152 int local_change_detected = change_detected;
153 if (!not_locked) {
154 if (!last_tag[0]
155 || (last_tag[0] != tag[0])
156 || strncmp(last_tag + 1, tag + 1, sizeof(last_tag) - 1)) {
157 /* invalidate log.tag.<tag> cache */
158 for (i = 0; i < (sizeof(tag_cache) / sizeof(tag_cache[0])); ++i) {
159 tag_cache[i].pinfo = NULL;
160 tag_cache[i].c = '\0';
161 }
162 last_tag[0] = '\0';
163 local_change_detected = 1;
164 }
165 if (!last_tag[0]) {
166 strncpy(last_tag, tag, sizeof(last_tag));
167 }
168 }
169 strcpy(key + sizeof(log_namespace) - 1, tag);
170
171 kp = key;
172 for (i = 0; i < (sizeof(tag_cache) / sizeof(tag_cache[0])); ++i) {
173 struct cache *cache = &tag_cache[i];
174 struct cache temp_cache;
175
176 if (not_locked) {
177 temp_cache.pinfo = NULL;
178 temp_cache.c = '\0';
179 cache = &temp_cache;
180 }
181 if (local_change_detected) {
182 refresh_cache(cache, kp);
183 }
184
185 if (cache->c) {
186 c = cache->c;
187 break;
188 }
189
190 kp = key + base_offset;
191 }
192 }
193
194 switch (toupper(c)) { /* if invalid, resort to global */
195 case 'V':
196 case 'D':
197 case 'I':
198 case 'W':
199 case 'E':
200 case 'F': /* Not officially supported */
201 case 'A':
202 case 'S':
203 case BOOLEAN_FALSE: /* Not officially supported */
204 break;
205 default:
206 /* clear '.' after log.tag */
207 key[sizeof(log_namespace) - 2] = '\0';
208
209 kp = key;
210 for (i = 0; i < (sizeof(global_cache) / sizeof(global_cache[0])); ++i) {
211 struct cache *cache = &global_cache[i];
212 struct cache temp_cache;
213
214 if (not_locked) {
215 temp_cache = *cache;
216 if (temp_cache.pinfo != cache->pinfo) { /* check atomic */
217 temp_cache.pinfo = NULL;
218 temp_cache.c = '\0';
219 }
220 cache = &temp_cache;
221 }
222 if (global_change_detected) {
223 refresh_cache(cache, kp);
224 }
225
226 if (cache->c) {
227 c = cache->c;
228 break;
229 }
230
231 kp = key + base_offset;
232 }
233 break;
234 }
235
236 if (!not_locked) {
237 global_serial = current_global_serial;
238 unlock();
239 }
240
241 switch (toupper(c)) {
242 case 'V': return ANDROID_LOG_VERBOSE;
243 case 'D': return ANDROID_LOG_DEBUG;
244 case 'I': return ANDROID_LOG_INFO;
245 case 'W': return ANDROID_LOG_WARN;
246 case 'E': return ANDROID_LOG_ERROR;
247 case 'F': /* FALLTHRU */ /* Not officially supported */
248 case 'A': return ANDROID_LOG_FATAL;
249 case BOOLEAN_FALSE: /* FALLTHRU */ /* Not Officially supported */
250 case 'S': return -1; /* ANDROID_LOG_SUPPRESS */
251 }
252 return default_prio;
253}

255LIBLOG_ABI_PUBLIC int __android_log_is_loggable(int prio, const char *tag,
256 int default_prio)
257{
258 int logLevel = __android_log_level(tag, default_prio);
259 return logLevel >= 0 && prio >= logLevel;
260}

262LIBLOG_HIDDEN int __android_log_is_debuggable()
263{
264 static uint32_t serial;
265 static struct cache tag_cache;
266 static const char key[] = "ro.debuggable";
267 int ret;
268
269 if (tag_cache.c) { /* ro property does not change after set */
270 ret = tag_cache.c == '1';
271 } else if (lock()) {
272 struct cache temp_cache = { NULL, -1, '\0' };
273 refresh_cache(&temp_cache, key);
274 ret = temp_cache.c == '1';
275 } else {
276 int change_detected = check_cache(&tag_cache);
277 uint32_t current_serial = __system_property_area_serial();
278 if (current_serial != serial) {
279 change_detected = 1;
280 }
281 if (change_detected) {
282 refresh_cache(&tag_cache, key);
283 serial = current_serial;
284 }
285 ret = tag_cache.c == '1';
286
287 unlock();
288 }
289
290 return ret;
291}

293/*
294 * For properties that are read often, but generally remain constant.
295 * Since a change is rare, we will accept a trylock failure gracefully.
296 * Use a separate lock from is_loggable to keep contention down b/25563384.
297 */
298struct cache2 {
299 pthread_mutex_t lock;
300 uint32_t serial;
301 const char *key_persist;
302 struct cache cache_persist;
303 const char *key_ro;
304 struct cache cache_ro;
305 unsigned char (*const evaluate)(const struct cache2 *self);
306};
307
308static inline unsigned char do_cache2(struct cache2 *self)
309{
310 uint32_t current_serial;
311 int change_detected;
312 unsigned char c;
313
314 if (pthread_mutex_trylock(&self->lock)) {
315 /* We are willing to accept some race in this context */
316 return self->evaluate(self);
317 }
318
319 change_detected = check_cache(&self->cache_persist)
320 || check_cache(&self->cache_ro);
321 current_serial = __system_property_area_serial();
322 if (current_serial != self->serial) {
323 change_detected = 1;
324 }
325 if (change_detected) {
326 refresh_cache(&self->cache_persist, self->key_persist);
327 refresh_cache(&self->cache_ro, self->key_ro);
328 self->serial = current_serial;
329 }
330 c = self->evaluate(self);
331
332 pthread_mutex_unlock(&self->lock);
333
334 return c;
335}
336
337static unsigned char evaluate_persist_ro(const struct cache2 *self)
338{
339 unsigned char c = self->cache_persist.c;
340
341 if (c) {
342 return c;
343 }
344
345 return self->cache_ro.c;
346}

348/*
349 * Timestamp state generally remains constant, but can change at any time
350 * to handle developer requirements.
351 */
352LIBLOG_ABI_PUBLIC clockid_t android_log_clockid()
353{
354 static struct cache2 clockid = {
355 PTHREAD_MUTEX_INITIALIZER,
356 0,
357 "persist.logd.timestamp",
358 { NULL, -1, '\0' },
359 "ro.logd.timestamp",
360 { NULL, -1, '\0' },
361 evaluate_persist_ro
362 };
363
364 return (tolower(do_cache2(&clockid)) == 'm')
365 ? CLOCK_MONOTONIC
366 : CLOCK_REALTIME;
367}

369/*
370 * Security state generally remains constant, but the DO must be able
371 * to turn off logging should it become spammy after an attack is detected.
372 */
373static unsigned char evaluate_security(const struct cache2 *self)
374{
375 unsigned char c = self->cache_ro.c;
376
377 return (c != BOOLEAN_FALSE) && c && (self->cache_persist.c == BOOLEAN_TRUE);
378}
379
380LIBLOG_ABI_PUBLIC int __android_log_security()
381{
382 static struct cache2 security = {
383 PTHREAD_MUTEX_INITIALIZER,
384 0,
385 "persist.logd.security",
386 { NULL, -1, BOOLEAN_FALSE },
387 "ro.device_owner",
388 { NULL, -1, BOOLEAN_FALSE },
389 evaluate_security
390 };
391
392 return do_cache2(&security);
393}
