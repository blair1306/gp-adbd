17#ifndef _CUTILS_KLOG_H_
18#define _CUTILS_KLOG_H_
19
20#include <sys/cdefs.h>
21#include <sys/uio.h>
22#include <stdarg.h>
23
24__BEGIN_DECLS
25
26void klog_init(void);
27int klog_get_level(void);
28void klog_set_level(int level);
29/* TODO: void klog_close(void); - and make klog_fd users thread safe. */
30
31void klog_write(int level, const char *fmt, ...)
32 __attribute__ ((format(printf, 2, 3)));
33void klog_writev(int level, const struct iovec* iov, int iov_count);
34
35__END_DECLS
36
37#define KLOG_ERROR_LEVEL 3
38#define KLOG_WARNING_LEVEL 4
39#define KLOG_NOTICE_LEVEL 5
40#define KLOG_INFO_LEVEL 6
41#define KLOG_DEBUG_LEVEL 7
42
43#define KLOG_ERROR(tag,x...) klog_write(KLOG_ERROR_LEVEL, "<3>" tag ": " x)
44#define KLOG_WARNING(tag,x...) klog_write(KLOG_WARNING_LEVEL, "<4>" tag ": " x)
45#define KLOG_NOTICE(tag,x...) klog_write(KLOG_NOTICE_LEVEL, "<5>" tag ": " x)
46#define KLOG_INFO(tag,x...) klog_write(KLOG_INFO_LEVEL, "<6>" tag ": " x)
47#define KLOG_DEBUG(tag,x...) klog_write(KLOG_DEBUG_LEVEL, "<7>" tag ": " x)
48
49#define KLOG_DEFAULT_LEVEL 3 /* messages <= this level are logged */
50
51#endif
