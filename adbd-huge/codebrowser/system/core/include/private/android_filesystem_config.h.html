97#define AID_SHELL 2000 /* adb and debug shell user */
98#define AID_CACHE 2001 /* cache access */
99#define AID_DIAG 2002 /* access to diagnostic resources */
100
101/* The range 2900-2999 is reserved for OEM, and must never be
102 * used here */
103#define AID_OEM_RESERVED_START 2900
104#define AID_OEM_RESERVED_END 2999
105
106/* The 3000 series are intended for use as supplemental group id's only.
107 * They indicate special Android capabilities that the kernel is aware of. */
108#define AID_NET_BT_ADMIN 3001 /* bluetooth: create any socket */
109#define AID_NET_BT 3002 /* bluetooth: create sco, rfcomm or l2cap sockets */
110#define AID_INET 3003 /* can create AF_INET and AF_INET6 sockets */
111#define AID_NET_RAW 3004 /* can create raw INET sockets */
112#define AID_NET_ADMIN 3005 /* can configure interfaces and routing tables. */
113#define AID_NET_BW_STATS 3006 /* read bandwidth statistics */
114#define AID_NET_BW_ACCT 3007 /* change bandwidth statistics accounting */
115#define AID_NET_BT_STACK 3008 /* bluetooth: access config files */
116#define AID_READPROC 3009 /* Allow /proc read access */
117#define AID_WAKELOCK 3010 /* Allow system wakelock read/write access */
118
119/* The range 5000-5999 is also reserved for OEM, and must never be used here. */
120#define AID_OEM_RESERVED_2_START 5000
121#define AID_OEM_RESERVED_2_END 5999
122
123#define AID_EVERYBODY 9997 /* shared between all apps in the same profile */
124#define AID_MISC 9998 /* access to misc storage */
125#define AID_NOBODY 9999
126
127#define AID_APP 10000 /* first app user */
128
129#define AID_ISOLATED_START 99000 /* start of uids for fully isolated sandboxed processes */
130#define AID_ISOLATED_END 99999 /* end of uids for fully isolated sandboxed processes */
131
132#define AID_USER 100000 /* offset for uid ranges for each user */
133
134#define AID_SHARED_GID_START 50000 /* start of gids for apps in each user to share */
135#define AID_SHARED_GID_END 59999 /* start of gids for apps in each user to share */
