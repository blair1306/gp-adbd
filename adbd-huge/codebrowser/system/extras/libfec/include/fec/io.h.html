116/* C++ wrappers for fec_handle and operations */
117namespace fec {
118 using handle = std::unique_ptr<fec_handle, decltype(&fec_close)>;
119
120 class io {
121 public:
122 io() : handle_(nullptr, fec_close) {}
123
124 io(const std::string& fn, int mode = O_RDONLY, int flags = 0,
125 int roots = FEC_DEFAULT_ROOTS) : handle_(nullptr, fec_close) {
126 open(fn, mode, flags, roots);
127 }
128
129 explicit operator bool() const {
130 return !!handle_;
131 }
132
133 bool open(const std::string& fn, int mode = O_RDONLY, int flags = 0,
134 int roots = FEC_DEFAULT_ROOTS)
135 {
136 fec_handle *fh = nullptr;
137 int rc = fec_open(&fh, fn.c_str(), mode, flags, roots);
138 if (!rc) {
139 handle_.reset(fh);
140 }
141 return !rc;
142 }
143
144 bool close() {
145 return !fec_close(handle_.release());
146 }
147
148 bool seek(int64_t offset, int whence) {
149 return !fec_seek(handle_.get(), offset, whence);
150 }
151
152 ssize_t read(void *buf, size_t count) {
153 return fec_read(handle_.get(), buf, count);
154 }
155
156 ssize_t pread(void *buf, size_t count, uint64_t offset) {
157 return fec_pread(handle_.get(), buf, count, offset);
158 }
159
160 bool get_status(fec_status& status) {
161 return !fec_get_status(handle_.get(), &status);
162 }
163
164 bool get_verity_metadata(fec_verity_metadata& data) {
165 return !fec_verity_get_metadata(handle_.get(), &data);
166 }
167
168 bool has_verity() {
169 fec_verity_metadata data;
170 return get_verity_metadata(data);
171 }
172
173 bool get_ecc_metadata(fec_ecc_metadata& data) {
174 return !fec_ecc_get_metadata(handle_.get(), &data);
175 }
176
177 bool has_ecc() {
178 fec_ecc_metadata data;
179 return get_ecc_metadata(data) && data.valid;
180 }
181
182 bool set_verity_status(bool enabled) {
183 return !fec_verity_set_status(handle_.get(), enabled);
184 }
185
186 private:
187 handle handle_;
188 };
189}
