88static int filter_dot(const struct dirent *d)
89{
90 return (strcmp(d->d_name, "..") && strcmp(d->d_name, "."));

93static u32 build_default_directory_structure(const char *dir_path,
94 struct selabel_handle *sehnd)
95{
96 u32 inode;
97 u32 root_inode;
98 struct dentry dentries = {
99 .filename = "lost+found",
100 .file_type = EXT4_FT_DIR,
101 .mode = S_IRWXU,
102 .uid = 0,
103 .gid = 0,
104 .mtime = 0,
105 };
106 root_inode = make_directory(0, 1, &dentries, 1);
107 inode = make_directory(root_inode, 0, NULL, 0);
108 *dentries.inode = inode;
109 inode_set_permissions(inode, dentries.mode,
110 dentries.uid, dentries.gid, dentries.mtime);
111
112#ifndef USE_MINGW
113 if (sehnd) {
114 char *path = NULL;
115 char *secontext = NULL;
116
117 asprintf(&path, "%slost+found", dir_path);
118 if (selabel_lookup(sehnd, &secontext, path, S_IFDIR) < 0) {
119 error("cannot lookup security context for %s", path);
120 } else {
121 inode_set_selinux(inode, secontext);
122 freecon(secontext);
123 }
124 free(path);
125 }
126#endif
127
128 return root_inode;
129}

131#ifndef USE_MINGW
132/* Read a local directory and create the same tree in the generated filesystem.
133 Calls itself recursively with each directory in the given directory.
134 full_path is an absolute or relative path, with a trailing slash, to the
135 directory on disk that should be copied, or NULL if this is a directory
136 that does not exist on disk (e.g. lost+found).
137 dir_path is an absolute path, with trailing slash, to the same directory
138 if the image were mounted at the specified mount point */
139static u32 build_directory_structure(const char *full_path, const char *dir_path, const char *target_out_path,
140 u32 dir_inode, fs_config_func_t fs_config_func,
141 struct selabel_handle *sehnd, int verbose, time_t fixed_time)
142{
143 int entries = 0;
144 struct dentry *dentries;
145 struct dirent **namelist = NULL;
146 struct stat stat;
147 int ret;
148 int i;
149 u32 inode;
150 u32 entry_inode;
151 u32 dirs = 0;
152 bool needs_lost_and_found = false;
153
154 if (full_path) {
155 entries = scandir(full_path, &namelist, filter_dot, (void*)alphasort);
156 if (entries < 0) {
157#ifdef __GLIBC__
158 /* The scandir function implemented in glibc has a bug that makes it
159 erroneously fail with ENOMEM under certain circumstances.
160 As a workaround we can retry the scandir call with the same arguments.
161 GLIBC BZ: https://sourceware.org/bugzilla/show_bug.cgi?id=17804 */
162 if (errno == ENOMEM)
163 entries = scandir(full_path, &namelist, filter_dot, (void*)alphasort);
164#endif
165 if (entries < 0) {
166 error_errno("scandir");
167 return EXT4_ALLOCATE_FAILED;
168 }
169 }
170 }
171
172 if (dir_inode == 0) {
173 /* root directory, check if lost+found already exists */
174 for (i = 0; i < entries; i++)
175 if (strcmp(namelist[i]->d_name, "lost+found") == 0)
176 break;
177 if (i == entries)
178 needs_lost_and_found = true;
179 }
180
181 dentries = calloc(entries, sizeof(struct dentry));
182 if (dentries == NULL)
183 critical_error_errno("malloc");
184
185 for (i = 0; i < entries; i++) {
186 dentries[i].filename = strdup(namelist[i]->d_name);
187 if (dentries[i].filename == NULL)
188 critical_error_errno("strdup");
189
190 asprintf(&dentries[i].path, "%s%s", dir_path, namelist[i]->d_name);
191 asprintf(&dentries[i].full_path, "%s%s", full_path, namelist[i]->d_name);
192
193 free(namelist[i]);
194
195 ret = lstat(dentries[i].full_path, &stat);
196 if (ret < 0) {
197 error_errno("lstat");
198 i--;
199 entries--;
200 continue;
201 }
202
203 dentries[i].size = stat.st_size;
204 dentries[i].mode = stat.st_mode & (S_ISUID|S_ISGID|S_ISVTX|S_IRWXU|S_IRWXG|S_IRWXO);
205 if (fixed_time == -1) {
206 dentries[i].mtime = stat.st_mtime;
207 } else {
208 dentries[i].mtime = fixed_time;
209 }
210 uint64_t capabilities;
211 if (fs_config_func != NULL) {
212#ifdef ANDROID
213 unsigned int mode = 0;
214 unsigned int uid = 0;
215 unsigned int gid = 0;
216 int dir = S_ISDIR(stat.st_mode);
217 fs_config_func(dentries[i].path, dir, target_out_path, &uid, &gid, &mode, &capabilities);
218 dentries[i].mode = mode;
219 dentries[i].uid = uid;
220 dentries[i].gid = gid;
221 dentries[i].capabilities = capabilities;
222#else
223 error("can't set android permissions - built without android support");
224#endif
225 }
226#ifndef USE_MINGW
227 if (sehnd) {
228 if (selabel_lookup(sehnd, &dentries[i].secon, dentries[i].path, stat.st_mode) < 0) {
229 error("cannot lookup security context for %s", dentries[i].path);
230 }
231
232 if (dentries[i].secon && verbose)
233 printf("Labeling %s as %s\n", dentries[i].path, dentries[i].secon);
234 }
235#endif
236
237 if (S_ISREG(stat.st_mode)) {
238 dentries[i].file_type = EXT4_FT_REG_FILE;
239 } else if (S_ISDIR(stat.st_mode)) {
240 dentries[i].file_type = EXT4_FT_DIR;
241 dirs++;
242 } else if (S_ISCHR(stat.st_mode)) {
243 dentries[i].file_type = EXT4_FT_CHRDEV;
244 } else if (S_ISBLK(stat.st_mode)) {
245 dentries[i].file_type = EXT4_FT_BLKDEV;
246 } else if (S_ISFIFO(stat.st_mode)) {
247 dentries[i].file_type = EXT4_FT_FIFO;
248 } else if (S_ISSOCK(stat.st_mode)) {
249 dentries[i].file_type = EXT4_FT_SOCK;
250 } else if (S_ISLNK(stat.st_mode)) {
251 dentries[i].file_type = EXT4_FT_SYMLINK;
252 dentries[i].link = calloc(info.block_size, 1);
253 readlink(dentries[i].full_path, dentries[i].link, info.block_size - 1);
254 } else {
255 error("unknown file type on %s", dentries[i].path);
256 i--;
257 entries--;
258 }
259 }
260 free(namelist);
261
262 if (needs_lost_and_found) {
263 /* insert a lost+found directory at the beginning of the dentries */
264 struct dentry *tmp = calloc(entries + 1, sizeof(struct dentry));
265 memset(tmp, 0, sizeof(struct dentry));
266 memcpy(tmp + 1, dentries, entries * sizeof(struct dentry));
267 dentries = tmp;
268
269 dentries[0].filename = strdup("lost+found");
270 asprintf(&dentries[0].path, "%slost+found", dir_path);
271 dentries[0].full_path = NULL;
272 dentries[0].size = 0;
273 dentries[0].mode = S_IRWXU;
274 dentries[0].file_type = EXT4_FT_DIR;
275 dentries[0].uid = 0;
276 dentries[0].gid = 0;
277 if (sehnd) {
278 if (selabel_lookup(sehnd, &dentries[0].secon, dentries[0].path, dentries[0].mode) < 0)
279 error("cannot lookup security context for %s", dentries[0].path);
280 }
281 entries++;
282 dirs++;
283 }
284
285 inode = make_directory(dir_inode, entries, dentries, dirs);
286
287 for (i = 0; i < entries; i++) {
288 if (dentries[i].file_type == EXT4_FT_REG_FILE) {
289 entry_inode = make_file(dentries[i].full_path, dentries[i].size);
290 } else if (dentries[i].file_type == EXT4_FT_DIR) {
291 char *subdir_full_path = NULL;
292 char *subdir_dir_path;
293 if (dentries[i].full_path) {
294 ret = asprintf(&subdir_full_path, "%s/", dentries[i].full_path);
295 if (ret < 0)
296 critical_error_errno("asprintf");
297 }
298 ret = asprintf(&subdir_dir_path, "%s/", dentries[i].path);
299 if (ret < 0)
300 critical_error_errno("asprintf");
301 entry_inode = build_directory_structure(subdir_full_path, subdir_dir_path, target_out_path,
302 inode, fs_config_func, sehnd, verbose, fixed_time);
303 free(subdir_full_path);
304 free(subdir_dir_path);
305 } else if (dentries[i].file_type == EXT4_FT_SYMLINK) {
306 entry_inode = make_link(dentries[i].link);
307 } else {
308 error("unknown file type on %s", dentries[i].path);
309 entry_inode = 0;
310 }
311 *dentries[i].inode = entry_inode;
312
313 ret = inode_set_permissions(entry_inode, dentries[i].mode,
314 dentries[i].uid, dentries[i].gid,
315 dentries[i].mtime);
316 if (ret)
317 error("failed to set permissions on %s\n", dentries[i].path);
318
319 /*
320 * It's important to call inode_set_selinux() before
321 * inode_set_capabilities(). Extended attributes need to
322 * be stored sorted order, and we guarantee this by making
323 * the calls in the proper order.
324 * Please see xattr_assert_sane() in contents.c
325 */
326 ret = inode_set_selinux(entry_inode, dentries[i].secon);
327 if (ret)
328 error("failed to set SELinux context on %s\n", dentries[i].path);
329 ret = inode_set_capabilities(entry_inode, dentries[i].capabilities);
330 if (ret)
331 error("failed to set capability on %s\n", dentries[i].path);
332
333 free(dentries[i].path);
334 free(dentries[i].full_path);
335 free(dentries[i].link);
336 free((void *)dentries[i].filename);
337 free(dentries[i].secon);
338 }
339
340 free(dentries);
341 return inode;
342}
343#endif

345static u32 compute_block_size()
346{
347 return 4096;
348}

350static u32 compute_journal_blocks()
351{
352 u32 journal_blocks = DIV_ROUND_UP(info.len, info.block_size) / 64;
353 if (journal_blocks < 1024)
354 journal_blocks = 1024;
355 if (journal_blocks > 32768)
356 journal_blocks = 32768;
357 return journal_blocks;
358}
359
360static u32 compute_blocks_per_group()
361{
362 return info.block_size * 8;
363}
364
365static u32 compute_inodes()
366{
367 return DIV_ROUND_UP(info.len, info.block_size) / 4;
368}
369
370static u32 compute_inodes_per_group()
371{
372 u32 blocks = DIV_ROUND_UP(info.len, info.block_size);
373 u32 block_groups = DIV_ROUND_UP(blocks, info.blocks_per_group);
374 u32 inodes = DIV_ROUND_UP(info.inodes, block_groups);
375 inodes = EXT4_ALIGN(inodes, (info.block_size / info.inode_size));
376
377 /* After properly rounding up the number of inodes/group,
378 * make sure to update the total inodes field in the info struct.
379 */
380 info.inodes = inodes * block_groups;
381
382 return inodes;
383}
384
385static u32 compute_bg_desc_reserve_blocks()
386{
387 u32 blocks = DIV_ROUND_UP(info.len, info.block_size);
388 u32 block_groups = DIV_ROUND_UP(blocks, info.blocks_per_group);
389 u32 bg_desc_blocks = DIV_ROUND_UP(block_groups * sizeof(struct ext2_group_desc),
390 info.block_size);
391
392 u32 bg_desc_reserve_blocks =
393 DIV_ROUND_UP(block_groups * 1024 * sizeof(struct ext2_group_desc),
394 info.block_size) - bg_desc_blocks;
395
396 if (bg_desc_reserve_blocks > info.block_size / sizeof(u32))
397 bg_desc_reserve_blocks = info.block_size / sizeof(u32);
398
399 return bg_desc_reserve_blocks;
400}

402void reset_ext4fs_info() {
403 // Reset all the global data structures used by make_ext4fs so it
404 // can be called again.
405 memset(&info, 0, sizeof(info));
406 memset(&aux_info, 0, sizeof(aux_info));
407

410 ext4_sparse_file = NULL;

418 info.len = len;

423int make_ext4fs(const char *filename, long long len,
424 const char *mountpoint, struct selabel_handle *sehnd)
425{
426 int fd;
427 int status;
428
429 reset_ext4fs_info();
430 info.len = len;
431
432 fd = open(filename, O_WRONLY | O_CREAT | O_TRUNC | O_BINARY, 0644);
433 if (fd < 0) {
434 error_errno("open");
435 return EXIT_FAILURE;
436 }
437
438 status = make_ext4fs_internal(fd, NULL, NULL, mountpoint, NULL, 0, 0, 0, 1, 0, sehnd, 0, -1, NULL);
439 close(fd);
440
441 return status;
442}

444/* return a newly-malloc'd string that is a copy of str. The new string
445 is guaranteed to have a trailing slash. If absolute is true, the new string
446 is also guaranteed to have a leading slash.
447*/
448static char *canonicalize_slashes(const char *str, bool absolute)
449{
450 char *ret;
451 int len = strlen(str);
452 int newlen = len;
453 char *ptr;
454
455 if (len == 0) {
456 if (absolute)
457 return strdup("/");
458 else
459 return strdup("");
460 }
461
462 if (str[0] != '/' && absolute) {
463 newlen++;
464 }
465 if (str[len - 1] != '/') {
466 newlen++;
467 }
468 ret = malloc(newlen + 1);
469 if (!ret) {
470 critical_error("malloc");
471 }
472
473 ptr = ret;
474 if (str[0] != '/' && absolute) {
475 *ptr++ = '/';
476 }
477
478 strcpy(ptr, str);
479 ptr += len;
480
481 if (str[len - 1] != '/') {
482 *ptr++ = '/';
483 }
484
485 if (ptr != ret + newlen) {
486 critical_error("assertion failed\n");
487 }
488
489 *ptr = '\0';
490
491 return ret;
492}
493
494static char *canonicalize_abs_slashes(const char *str)
495{
496 return canonicalize_slashes(str, true);
497}
498
499static char *canonicalize_rel_slashes(const char *str)
500{
501 return canonicalize_slashes(str, false);
502}

504int make_ext4fs_internal(int fd, const char *_directory, const char *_target_out_directory,
505 const char *_mountpoint, fs_config_func_t fs_config_func, int gzip,
506 int sparse, int crc, int wipe, int real_uuid,
507 struct selabel_handle *sehnd, int verbose, time_t fixed_time,
508 FILE* block_list_file)
509{
510 u32 root_inode_num;
511 u16 root_mode;
512 char *mountpoint;
513 char *directory = NULL;
514 char *target_out_directory = NULL;
515
516 if (setjmp(setjmp_env))
517 return EXIT_FAILURE; /* Handle a call to longjmp() */
518
519 info.block_device = is_block_device_fd(fd);
520
521 if (info.block_device && (sparse || gzip || crc)) {
522 fprintf(stderr, "No sparse/gzip/crc allowed for block device\n");
523 return EXIT_FAILURE;
524 }
525
526 if (_mountpoint == NULL) {
527 mountpoint = strdup("");
528 } else {
529 mountpoint = canonicalize_abs_slashes(_mountpoint);
530 }
531
532 if (_directory) {
533 directory = canonicalize_rel_slashes(_directory);
534 }
535
536 if (_target_out_directory) {
537 target_out_directory = canonicalize_rel_slashes(_target_out_directory);
538 }
539
540 if (info.len <= 0)
541 info.len = get_file_size(fd);
542
543 if (info.len <= 0) {
544 fprintf(stderr, "Need size of filesystem\n");
545 return EXIT_FAILURE;
546 }
547
548 if (info.block_size <= 0)
549 info.block_size = compute_block_size();
550
551 /* Round down the filesystem length to be a multiple of the block size */
552 info.len &= ~((u64)info.block_size - 1);
553
554 if (info.journal_blocks == 0)
555 info.journal_blocks = compute_journal_blocks();
556
557 if (info.no_journal == 0)
558 info.feat_compat = EXT4_FEATURE_COMPAT_HAS_JOURNAL;
559 else
560 info.journal_blocks = 0;
561
562 if (info.blocks_per_group <= 0)
563 info.blocks_per_group = compute_blocks_per_group();
564
565 if (info.inodes <= 0)
566 info.inodes = compute_inodes();
567
568 if (info.inode_size <= 0)
569 info.inode_size = 256;
570
571 if (info.label == NULL)
572 info.label = "";
573
574 info.inodes_per_group = compute_inodes_per_group();
575
576 info.feat_compat |=
577 EXT4_FEATURE_COMPAT_RESIZE_INODE |
578 EXT4_FEATURE_COMPAT_EXT_ATTR;
579
580 info.feat_ro_compat |=
581 EXT4_FEATURE_RO_COMPAT_SPARSE_SUPER |
582 EXT4_FEATURE_RO_COMPAT_LARGE_FILE |
583 EXT4_FEATURE_RO_COMPAT_GDT_CSUM;
584
585 info.feat_incompat |=
586 EXT4_FEATURE_INCOMPAT_EXTENTS |
587 EXT4_FEATURE_INCOMPAT_FILETYPE;
588
589
590 info.bg_desc_reserve_blocks = compute_bg_desc_reserve_blocks();
591
592 printf("Creating filesystem with parameters:\n");
593 printf(" Size: %"PRIu64"\n", info.len);
594 printf(" Block size: %d\n", info.block_size);
595 printf(" Blocks per group: %d\n", info.blocks_per_group);
596 printf(" Inodes per group: %d\n", info.inodes_per_group);
597 printf(" Inode size: %d\n", info.inode_size);
598 printf(" Journal blocks: %d\n", info.journal_blocks);
599 printf(" Label: %s\n", info.label);
600
601 ext4_create_fs_aux_info();
602
603 printf(" Blocks: %"PRIu64"\n", aux_info.len_blocks);
604 printf(" Block groups: %d\n", aux_info.groups);
605 printf(" Reserved block group size: %d\n", info.bg_desc_reserve_blocks);
606
607 ext4_sparse_file = sparse_file_new(info.block_size, info.len);
608
609 block_allocator_init();
610
611 ext4_fill_in_sb(real_uuid);
612
613 if (reserve_inodes(0, 10) == EXT4_ALLOCATE_FAILED)
614 error("failed to reserve first 10 inodes");
615
616 if (info.feat_compat & EXT4_FEATURE_COMPAT_HAS_JOURNAL)
617 ext4_create_journal_inode();
618
619 if (info.feat_compat & EXT4_FEATURE_COMPAT_RESIZE_INODE)
620 ext4_create_resize_inode();
621
622#ifdef USE_MINGW
623 // Windows needs only 'create an empty fs image' functionality
624 assert(!directory);
625 root_inode_num = build_default_directory_structure(mountpoint, sehnd);
626#else
627 if (directory)
628 root_inode_num = build_directory_structure(directory, mountpoint, target_out_directory, 0,
629 fs_config_func, sehnd, verbose, fixed_time);
630 else
631 root_inode_num = build_default_directory_structure(mountpoint, sehnd);
632#endif
633
634 root_mode = S_IRWXU | S_IRGRP | S_IXGRP | S_IROTH | S_IXOTH;
635 inode_set_permissions(root_inode_num, root_mode, 0, 0, 0);
636
637#ifndef USE_MINGW
638 if (sehnd) {
639 char *secontext = NULL;
640
641 if (selabel_lookup(sehnd, &secontext, mountpoint, S_IFDIR) < 0) {
642 error("cannot lookup security context for %s", mountpoint);
643 }
644 if (secontext) {
645 if (verbose) {
646 printf("Labeling %s as %s\n", mountpoint, secontext);
647 }
648 inode_set_selinux(root_inode_num, secontext);
649 }
650 freecon(secontext);
651 }
652#endif
653
654 ext4_update_free();
655
656 if (block_list_file) {
657 size_t dirlen = directory ? strlen(directory) : 0;
658 struct block_allocation* p = get_saved_allocation_chain();
659 while (p) {
660 if (directory && strncmp(p->filename, directory, dirlen) == 0) {
661 // substitute mountpoint for the leading directory in the filename, in the output file
662 fprintf(block_list_file, "%s%s", mountpoint, p->filename + dirlen);
663 } else {
664 fprintf(block_list_file, "%s", p->filename);
665 }
666 print_blocks(block_list_file, p);
667 struct block_allocation* pn = p->next;
668 free_alloc(p);
669 p = pn;
670 }
671 }
672
673 printf("Created filesystem with %d/%d inodes and %d/%d blocks\n",
674 aux_info.sb->s_inodes_count - aux_info.sb->s_free_inodes_count,
675 aux_info.sb->s_inodes_count,
676 aux_info.sb->s_blocks_count_lo - aux_info.sb->s_free_blocks_count_lo,
677 aux_info.sb->s_blocks_count_lo);
678
679 if (wipe && WIPE_IS_SUPPORTED) {
680 wipe_block_device(fd, info.len);
681 }
682
683 write_ext4_image(fd, gzip, sparse, crc);
684
685 sparse_file_destroy(ext4_sparse_file);
686 ext4_sparse_file = NULL;
687
688 free(mountpoint);
689 free(directory);
690
691 return 0;
