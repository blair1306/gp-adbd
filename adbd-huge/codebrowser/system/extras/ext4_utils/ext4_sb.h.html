30struct fs_info {
31 int64_t len; /* If set to 0, ask the block device for the size,
32 * if less than 0, reserve that much space at the
33 * end of the partition, else use the size given. */
34 uint32_t block_size;
35 uint32_t blocks_per_group;
36 uint32_t inodes_per_group;
37 uint32_t inode_size;
38 uint32_t inodes;
39 uint32_t journal_blocks;
40 uint16_t feat_ro_compat;
41 uint16_t feat_compat;
42 uint16_t feat_incompat;
43 uint32_t bg_desc_reserve_blocks;
44 const char *label;
45 uint8_t no_journal;
46 bool block_device; /* target fd is a block device? */
47};
