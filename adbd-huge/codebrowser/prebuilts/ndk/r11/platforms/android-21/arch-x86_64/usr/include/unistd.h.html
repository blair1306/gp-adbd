187/* Used to retry syscalls that can return EINTR. */
188#define TEMP_FAILURE_RETRY(exp) ({ \
189 __typeof__(exp) _rc; \
190 do { \
191 _rc = (exp); \
192 } while (_rc == -1 && errno == EINTR); \
193 _rc; })

195#if defined(__BIONIC_FORTIFY)
196extern ssize_t __read_chk(int, void*, size_t, size_t);
197__errordecl(__read_dest_size_error, "read called with size bigger than destination");
198__errordecl(__read_count_toobig_error, "read called with count > SSIZE_MAX");
199extern ssize_t __read_real(int, void*, size_t)
200 __asm__(__USER_LABEL_PREFIX__ "read");
201
202__BIONIC_FORTIFY_INLINE
203ssize_t read(int fd, void* buf, size_t count) {
204 size_t bos = __bos0(buf);
205
206#if !defined(__clang__)
207 if (__builtin_constant_p(count) && (count > SSIZE_MAX)) {
208 __read_count_toobig_error();
209 }
210
211 if (bos == __BIONIC_FORTIFY_UNKNOWN_SIZE) {
212 return __read_real(fd, buf, count);
213 }
214
215 if (__builtin_constant_p(count) && (count > bos)) {
216 __read_dest_size_error();
217 }
218
219 if (__builtin_constant_p(count) && (count <= bos)) {
220 return __read_real(fd, buf, count);
221 }
222#endif
223
224 return __read_chk(fd, buf, count, bos);
225}
226#endif /* defined(__BIONIC_FORTIFY) */
