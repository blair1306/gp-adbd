42#include <sys/types.h>
43#include <sys/cdefs.h>
44
45__BEGIN_DECLS
46#if defined(__BIONIC_FORTIFY)
47#define bcopy(b1, b2, len) \
48 (void)(__builtin___memmove_chk((b2), (b1), (len), __bos0(b2)))
49#define bzero(b, len) \
50 (void)(__builtin___memset_chk((b), '\0', (len), __bos0(b)))
51#else
52#define bcopy(b1, b2, len) (void)(__builtin_memmove((b2), (b1), (len)))
53#define bzero(b, len) (void)(__builtin_memset((b), '\0', (len)))
54#endif
55
56int ffs(int);
57int strcasecmp(const char *, const char *);
58int strncasecmp(const char *, const char *, size_t);
59
60__END_DECLS

62#endif /* !defined(_STRINGS_H_) */
