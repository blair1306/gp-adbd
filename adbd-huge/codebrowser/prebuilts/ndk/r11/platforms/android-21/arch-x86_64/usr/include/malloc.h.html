17#ifndef LIBC_INCLUDE_MALLOC_H_
18#define LIBC_INCLUDE_MALLOC_H_
19
20/*
21 * Declaration of malloc routines. Bionic uses dlmalloc (see
22 * upstream-dlmalloc) but doesn't directly include it here to keep the
23 * defined malloc.h interface small.
24 */
25#include <sys/cdefs.h>
26#include <stddef.h>
27
28__BEGIN_DECLS
29
30extern void* malloc(size_t byte_count) __mallocfunc __wur __attribute__((alloc_size(1)));
31extern void* calloc(size_t item_count, size_t item_size) __mallocfunc __wur __attribute__((alloc_size(1,2)));
32extern void* realloc(void* p, size_t byte_count) __wur __attribute__((alloc_size(2)));
33extern void free(void* p);
34
35extern void* memalign(size_t alignment, size_t byte_count) __mallocfunc __wur __attribute__((alloc_size(2)));
36extern size_t malloc_usable_size(const void* p);
37
38#ifndef STRUCT_MALLINFO_DECLARED
39#define STRUCT_MALLINFO_DECLARED 1
40struct mallinfo {
41 size_t arena; /* Total number of non-mmapped bytes currently allocated from OS. */
42 size_t ordblks; /* Number of free chunks. */
43 size_t smblks; /* (Unused.) */
44 size_t hblks; /* (Unused.) */
45 size_t hblkhd; /* Total number of bytes in mmapped regions. */
46 size_t usmblks; /* Maximum total allocated space; greater than total if trimming has occurred. */
47 size_t fsmblks; /* (Unused.) */
48 size_t uordblks; /* Total allocated space (normal or mmapped.) */
49 size_t fordblks; /* Total free space. */
50 size_t keepcost; /* Upper bound on number of bytes releasable by malloc_trim. */
51};
52#endif /* STRUCT_MALLINFO_DECLARED */
53
54extern struct mallinfo mallinfo(void);
55
56__END_DECLS
57
58#endif /* LIBC_INCLUDE_MALLOC_H_ */
