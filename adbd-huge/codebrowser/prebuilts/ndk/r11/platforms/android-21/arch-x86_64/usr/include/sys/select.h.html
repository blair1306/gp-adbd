29#ifndef _SYS_SELECT_H_
30#define _SYS_SELECT_H_
31
32#include <sys/cdefs.h>
33#include <sys/time.h>
34#include <sys/types.h>
35#include <signal.h>
36#include <string.h>
37
38__BEGIN_DECLS
39
40#define FD_SETSIZE 1024
41#define NFDBITS (8 * sizeof(unsigned long))
42#define __FDSET_LONGS (FD_SETSIZE/NFDBITS)
43
44typedef struct {
45 unsigned long fds_bits[__FDSET_LONGS];
46} fd_set;
47
48#define __FDELT(fd) ((fd) / NFDBITS)
49#define __FDMASK(fd) (1UL << ((fd) % NFDBITS))
50#define __FDS_BITS(set) (((fd_set*)(set))->fds_bits)
51
52#define FD_ZERO(set) (memset(set, 0, sizeof(*(fd_set*)(set))))
53
54#if defined(__BIONIC_FORTIFY)
55extern void __FD_CLR_chk(int, fd_set*, size_t);
56extern void __FD_SET_chk(int, fd_set*, size_t);
57extern int __FD_ISSET_chk(int, fd_set*, size_t);
58#define FD_CLR(fd, set) __FD_CLR_chk(fd, set, __bos(set))
59#define FD_SET(fd, set) __FD_SET_chk(fd, set, __bos(set))
60#define FD_ISSET(fd, set) __FD_ISSET_chk(fd, set, __bos(set))
61#else
62#define FD_CLR(fd, set) (__FDS_BITS(set)[__FDELT(fd)] &= ~__FDMASK(fd))
63#define FD_SET(fd, set) (__FDS_BITS(set)[__FDELT(fd)] |= __FDMASK(fd))
64#define FD_ISSET(fd, set) ((__FDS_BITS(set)[__FDELT(fd)] & __FDMASK(fd)) != 0)
65#endif /* defined(__BIONIC_FORTIFY) */
66
67extern int select(int, fd_set*, fd_set*, fd_set*, struct timeval*);
68extern int pselect(int, fd_set*, fd_set*, fd_set*, const struct timespec*, const sigset_t*);
69
70__END_DECLS
71
72#endif /* _SYS_SELECT_H_ */
