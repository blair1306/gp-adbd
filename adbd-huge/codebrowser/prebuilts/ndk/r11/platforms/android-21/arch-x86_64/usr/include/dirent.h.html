37#ifndef DT_UNKNOWN

41#define DT_DIR 4
42#define DT_BLK 6

46#define DT_WHT 14

49#define __DIRENT64_BODY \
50 uint64_t d_ino; \
51 int64_t d_off; \
52 unsigned short d_reclen; \
53 unsigned char d_type; \
54 char d_name[256]; \

56struct dirent { __DIRENT64_BODY };

59#undef __DIRENT64_BODY

73extern int dirfd(DIR*);
74extern int alphasort(const struct dirent**, const struct dirent**);
75extern int alphasort64(const struct dirent64**, const struct dirent64**);
